52  {

54  return T(1);

64  template <typename T>
65  T binomial( int n, int j )
66  {
67  if( n < j )
68  return T(0);
69 
70  return fact <T> (n) / static_cast<double>( fact <T> (j) * fact <T> (n-j) );
