107  /***********************************************************************
108  ****** The partition vector.
109  ****** Based on continuity, the domain is partitioned.
110  ****** Here a knot vector is partitioned in intervals where each
111  ****** partition is expressed by a pair of indexes in the knot vector.

114  template <typename T>
115  class VisPart : public std::vector<int> {
116  T tol; // Can be static - and depending on the type T
117  bool eq(T a, T b) { return a-b < tol; }
118 
119  public:
120  VisPart(const DVector<T>& t, int k, int dis) : tol(1e-5) {
121  // t - is the knot vector
122  // k - is the order of the spline
123  // dis = 0 - split where we have a discontinue function
124  // dis = 1 - split where we have a discontinue 1. derivative
125  // dis = 2 - split where we have a discontinue 2. derivative ...
126  // dis must not be bigger than d=k-1 !!!
127 // tol = 1e-5;
128  int d = k-1;
129  if(test1(t, d))
130  comp(t, d, d);
131  else if(test2(t, d)) {
132  comp(t, d, dis);
133  corr(t,2*d,t.getDim()-2*d-1);
134  }
135  else
136  comp(t, d, dis);
137  }
138 
139 
140  private:
141  void comp(const DVector<T>& t, int d, int dis) {
142  int n = t.getDim()-d-1;
143  int i = d;
144  while(eq(t(i+1), t(i))) ++i;
145  this->push_back(i);
146  for(++i; i <= n; i++) {
147  int j=1;
148  while (eq(t(i+j), t(i))) ++j;
149  if(i+j-1 >= n) {
150  this->push_back(i);
151  break;
152  }
153  if(j > d-dis) {
154  this->push_back(i);
155  this->push_back(i+=j-1);
156  }
157  }
158  }
159 
160 
161  void corr(const DVector<T>& t, int a, int b) {
162  unsigned int i=1;
163  for(; i < this->size(); i++)
164  if((*this)[i] > b) break;
165  if((*this)[i-1] < a) {
166  for(int j = b; j>=a; j--)
167  if(t(j-1) < t(j)) break;
168  this->insert(this->begin()+i,2,b);
169  }
170  }
171 
172 
173  bool test1( const DVector<T>& t, int d ) {
174  if(eq(t(d),t(0))) return false;
175  else return t.getDim() < (4*d+1);
176  }
177 
178 
179  bool test2( const DVector<T>& t, int d ) {
180  if(eq(t(d),t(0))) return false;
181  else return t.getDim() >= (4*d+1);
182  }
183  }; // end class VisPart

192  template <typename T>
193  class SampNr : public std::vector<int> {
194  public:
195  SampNr(const DVector<T>& t, const VisPart<T>& vp, int m) {
196  // t - is the knot vector
197  // vp - is the partition vector
198  // m - is the total number of samples (at least)
199  T tot = t(vp.back())- t(vp.front());
200  int n = vp.size()/2;
201  this->resize(n);
202 
203  for(int i=0; i < n; i++) {
204  T tmp = T(m+n-1)*(t(vp[2*i+1]) - t(vp[2*i]))/tot + T(0.5);
205  (*this)[i] = int(tmp);
206  }
207  }
208  }; // end class SampNr

244  inline
245  void IndexBsp::_init( int i, int k, int n ) {
246  // i - the index of the knot on left side of the value: t[i] <= t < t[i+1]
247  // k - the order of the spline
248  // n - the number of control points
249  i -= (k-1);
250  if( i+k > n) { // closed, and go circular
251  int j, s = n-i;
252  for (j=0; j < s; j++)
253  (*this)[j] = i++;
254  while(j < k)
255  for (i=0 ; i < n; i++) {
256  (*this)[j++] = i;
257  if(j==k) break;
258  }
259  }
260  else // ordinary, only a sequence.
261  for(int j=0; j<k; j++)
262  (*this)[j]= i+j;
263  }

358  // Sampling for pre-evaluation, independent of direction
359  //*******************************************************
360  template <typename T>
361  void preSamples( DVector< PreMat<T> >& p, const DVector<T>& t, int m, int d, int n, T start, T end ) {
362 
363  // compute dt (step in parameter)
364  const T dt = ( end - start ) / T(m-1);
365 
366  // Set the dimension of the Bernstein-Hermite Polynomial DVector
367  p.setDim(m);
368 
369  // Compute the Bernstein-Hermite Polynomiale, for the B-spline Surface
370  for( int j = 0; j < m-1; j++ ) {
371  int i = EvaluatorStatic<T>::evaluateBSp( p[j].m, start+j*dt, t, d, false ) - d;
372  p[j].ind.init( i, d+1, n);
373  }
374  int i = EvaluatorStatic<T>::evaluateBSp( p[m-1].m, end, t, d, true ) - d;
375  p[m-1].ind.init( i, d+1, n);
376  }
