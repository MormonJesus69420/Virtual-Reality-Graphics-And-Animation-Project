146  void DefaultSelectRenderer::select(int what) {
147 
148  const Camera *cam = getCamera();
149 
150  GL_CHECK(::glViewport(0,0,_size(0),_size(1)));
151 
152  // Clear buffers
153  _fbo.clear( GL_DEPTH_BUFFER_BIT );
154  _fbo.clearColorBuffer( GMcolor::black() );
155 
156  // Render selection
157  GLboolean depth_test_state;
158  GL_CHECK(::glGetBooleanv( GL_DEPTH_TEST, &depth_test_state ));
159  GL_CHECK(::glEnable( GL_DEPTH_TEST ));
160 
161  GL_CHECK(::glPolygonMode(GL_FRONT_AND_BACK,GL_FILL));
162 
163  _fbo.bind(); {
164 
165  for( int i=0; i < _objs.getSize(); ++i ) {
166 
167  const SceneObject *obj = _objs(i);
168  if( obj != cam && ( what == 0 || what == obj->getTypeId() || ( what < 0 && what + obj->getTypeId() != 0 ) ) ) {
169 
170  if(obj->isCollapsed()) {
171 
172  VisualizerStdRep::getInstance()->renderGeometry(obj,this,obj->getVirtualName());
173  }
174  else {
175 
176  const Array<Visualizer*>& visus = obj->getVisualizers();
177  for( int j = 0; j < visus.getSize(); ++j )
178  visus(j)->renderGeometry(obj,this,obj->getVirtualName());
179 
180  obj->localSelect(this,obj->getVirtualName());
181  }
182  }
183  }
184 
185  } _fbo.unbind();
186 
187  if( !depth_test_state )
188  GL_CHECK(::glDisable( GL_DEPTH_TEST ));
189  }
