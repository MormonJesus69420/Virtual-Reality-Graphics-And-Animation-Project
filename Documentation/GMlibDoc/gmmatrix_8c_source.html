1083  Vector<int,n> indx;
1084  Vector<double,n> vv;
1085 
1086  const double TINY=1.0e-20;
1087  int i,j,imax,k;
1088  double big,dum,sum,temp;
1089 
1090  double d=1.0;
1091  for (i=0;i<n;i++)
1092  {
1093  big=0.0;
1094  for (j=0;j<n;j++) if ((temp = std::fabs(a[i][j])) > big) big=temp;
1095  // if (big == 0.0) nrerror("Singular matrix in routine ludcmp");
1096  vv[i] = 1.0/big;
1097  }
1098  imax = 0;
1099  for (j=0;j<n;j++)
1100  {
1101  big=0.0;
1102  for (i=0;i<j;i++)
1103  {
1104  sum = a[i][j];
1105  for (k=0;k<i;k++) sum -= a[i][k]*a[k][j];
1106  a[i][j] = sum;
1107  }
1108  for (i=j;i<n;i++)
1109  {
1110  sum = a[i][j];
1111  for (k=0;k<j;k++) sum -= a[i][k]*a[k][j];
1112  a[i][j]=sum;
1113  if ((dum=vv[i]*std::fabs(sum)) >= big) { big=dum; imax=i; }
1114  }
1115  if (j != imax)
1116  {
1117  for (k=0;k<n;k++) a.swap(imax,k,j,k);
1118  d = -d;
1119  vv[imax]=vv[j];
1120  }
1121  indx[j] = imax;
1122  if (a[j][j] == 0.0) a[j][j] = TINY;
1123  if (j != n-1)
1124  {
1125  dum=1.0/(a[j][j]);
1126  for (i=j+1;i<n;i++) a[i][j] *= dum;
1127  }
1128  } // LU-decomp. finished, stored in a[][]
1129 
1130  for(int cols=0; cols<n; cols++) // LU-back subst. begins
1131  {
1132  int ii=0, ip;
1133  vv = T(0.0); vv[cols] = T(1.0);
1134  for (i=0;i<n;i++)
1135  {
1136  ip = indx[i];
1137  T sum2 = vv[ip];
1138  vv[ip] = vv[i];
1139  if (ii != 0) for (j=ii-1;j<i;j++) sum2 -= a[i][j]*vv[j];
1140  else if (sum2 != 0.0) ii = i+1;
1141  vv[i] = sum2;
1142  }
1143  for (i=n-1;i>=0;i--)
1144  {
1145  T sum2=vv[i];
1146  for (j=i+1;j<n;j++) sum2 -= a[i][j]*vv[j];
