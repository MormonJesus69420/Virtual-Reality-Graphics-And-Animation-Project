75  bool Texture::genTexture() {
76 
77  // Check if Data != 0
78  if( _data == 0 )
79  return false;
80 
81  // Gen Texture ID
82  if( _texture_id ) {
83 
84  _texture_id_map[_texture_id]--;
85 
86  if( _texture_id_map[_texture_id] < 1 )
87  glDeleteTextures( 1, &_texture_id );
88  }
89 
90  glGenTextures( 1, &_texture_id );
91 
92 
93  // Bind Texture ID
94  glBindTexture( _texture_dimension, _texture_id );
95 
96 
97  // Texture parameters;
98  // Minifying and Magnifying Policy
99  glTexParameteri( _texture_dimension, GL_TEXTURE_MIN_FILTER, GL_LINEAR );
100  glTexParameteri( _texture_dimension, GL_TEXTURE_MAG_FILTER, GL_LINEAR );
101 
102 
103  // S/T/R Wrap Policy
104  if( _texture_dimension == GL::GM_TEXTURE_1D || _texture_dimension == GL::GM_TEXTURE_2D )
105  glTexParameteri( _texture_dimension, GL_TEXTURE_WRAP_S, GL_REPEAT );
106 
107  if( _texture_dimension == GL::GM_TEXTURE_2D)
108  glTexParameteri( _texture_dimension, GL_TEXTURE_WRAP_T, GL_REPEAT );
109 
110 
111  // S/T/R Texture Gen Mode
112  if( _texture_dimension == GL::GM_TEXTURE_1D || _texture_dimension == GL::GM_TEXTURE_2D )
113  glTexGeni( GL_S, GL_TEXTURE_GEN_MODE, _texture_gen_mode );
114 
115  if( _texture_dimension == GL::GM_TEXTURE_2D )
116  glTexGeni( GL_T, GL_TEXTURE_GEN_MODE, _texture_gen_mode );
117 
118 
119  // Texture Coordinate Generation
120  if( _texture_dimension == GL::GM_TEXTURE_1D ) {
121 
122  if ( _depth == GM_TEXTURE_DEPTH_24 )
123  gluBuild1DMipmaps( _texture_dimension, 3, _width, GL_RGB, GL_UNSIGNED_BYTE, _data );
124  else if( _depth == GM_TEXTURE_DEPTH_32 )
125  gluBuild1DMipmaps( _texture_dimension, 3, _width, GL_RGBA, GL_UNSIGNED_BYTE, _data );
126  }
127  else if ( _texture_dimension == GL::GM_TEXTURE_2D ) {
128 
129  if ( _depth == GM_TEXTURE_DEPTH_24 )
130  gluBuild2DMipmaps( _texture_dimension, 3, _width, _height, GL_RGB, GL_UNSIGNED_BYTE, _data );
131  else if( _depth == GM_TEXTURE_DEPTH_32 )
132  gluBuild2DMipmaps( _texture_dimension, 3, _width, _height, GL_RGBA, GL_UNSIGNED_BYTE, _data );
133  }
134 
135  // "Release" Texture ID
136  glBindTexture( _texture_dimension, 0 );
137 
138  // Add Texture ID to Texture ID Map
139  _texture_id_map[_texture_id] = 1;
140 
141  // Return Success
142  return true;
143  }

214  Texture& Texture::operator = ( const Texture& texture ) {
215

218

239  glDeleteTextures( 1, &old_texture_id );
