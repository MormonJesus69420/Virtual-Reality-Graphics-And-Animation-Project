124  template <typename T>
125  void PBezierCurveSurf<T>::eval(T u, T v, int d1,int d2, bool /*lu*/, bool /*lv*/ ) const {
126 
127  if(_swap_par) {
128  std::swap(u,v);
129  std::swap(d1,d2);
130  }
131 
132  DMatrix<Vector<T,3> > c(_cu.getSize(), d2+1);
133 
134  for(int i=0; i < _cu.getSize(); i++)
135  c[i] = _cu(i)->evaluateParent(v, d2);
136 
137  // Compute the Bernstein-Hermite Polynomials
138  DMatrix< T > bhp;
139  EvaluatorStatic<T>::evaluateBhp( bhp, _cu.getSize()-1, u, 1);
140 
141  DMatrix< Vector<T,3> > p = bhp * c;
142  p.resetDim(d1+1,d2+1);
143 
144  if(_swap_par) this->_p = p.transpose();
145  else this->_p = p;
146 
147  this->_p.resetDim(d1+1,d2+1);
148  }
