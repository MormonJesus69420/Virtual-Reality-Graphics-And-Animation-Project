84 
85  this->_p.setDim( d1+1, d2+1 );

87  this->_p[0][0][0] = cos(u)*(_b*cos(v)+_a);

90 
91

102  this->_p[2][0][2] = T(0);

107  this->_p[0][1][2] = _c*cos(v);

113  }

116  this->_p[1][1][1] = -_b*cos(u)*sin(v);
117  this->_p[1][1][2] = T(0);
118  }

123  }
