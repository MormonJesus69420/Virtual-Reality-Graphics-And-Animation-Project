177  DVector<T> nt(_t.getDim()-_d-_d);
178  for(int i=0; i<nt.getDim(); i++)
179  nt[i]=_t[i+_d];

187  template <typename T_Stream>

191

193  if(_k == 1) { // closed
194  in >> _d;
195  _k = _d+1;
196  DVector<T> u = _t;
197  spesKnotClose( _t, _cl, u, _c.getDim(), _d);
198  }
199  else // open
