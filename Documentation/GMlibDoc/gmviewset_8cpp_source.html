151  void ViewSet::prepareGraphics() {
152 
153  _no_borders = _borders.size();
154 
155  DVector<GLBorder> data(_no_borders);
156  for( int i = 0; i < _no_borders; ++i ) {
157 
158  const float w_fact = 1.0f/float(_vp_w);
159  const float h_fact = 1.0f/float(_vp_h);
160  const Box<int,2> &bounds = _borders[i]->getBounds();
161 
162  data[i].v[0].x = w_fact * bounds.getPointMin()(0);
163  data[i].v[0].y = h_fact * bounds.getPointMin()(1);
164 
165  data[i].v[1].x = w_fact * bounds.getPointMin()(0);
166  data[i].v[1].y = h_fact * bounds.getPointMax()(1);
167 
168  data[i].v[2].x = w_fact * bounds.getPointMax()(0);
169  data[i].v[2].y = h_fact * bounds.getPointMax()(1);
170 
171  data[i].v[3].x = w_fact * bounds.getPointMax()(0);
172  data[i].v[3].y = h_fact * bounds.getPointMin()(1);
173  }
174 
175  _vbo.bind();
176  _vbo.bufferData(_no_borders * sizeof(GLBorder), data.getPtr(), GL_STATIC_DRAW);
177  _vbo.unbind();
