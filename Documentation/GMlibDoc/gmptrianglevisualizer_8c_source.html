38  template <typename T, int n>
39  void PTriangleVisualizer<T,n>::fillStandardVBO(GL::VertexBufferObject vbo, const DVector<DVector<Vector<T,n> > > &p) {
40 
41  int no_dp = p.getDim();
42  DVector<GL::GLVertexNormal> dp(no_dp);
43 
44  const GMlib::Vector<float,3> e1 {-1.0,1.0,0.0}; // Directional derivative
45  const GMlib::Vector<float,3> e2 {-1.0,0.0,1.0}; // Directional derivative (linear independent of e1
46 
47  GMlib::Vector<float,3> v1, v2;
48  GMlib::UnitVector<float,3> N;
49 
50  for( int i = 0; i < p.getDim(); i++ ) {
51 
52  const Vector<float,3> p0 = p(i)(0);
53  const Vector<float,3> p1 = p(i)(1);
54  const Vector<float,3> p2 = p(i)(2);
55  const Vector<float,3> p3 = p(i)(3);
56 
57  v1 = e1(0) * p1 + e1(1) * p2 + e1(2) * p3;
58  v2 = e2(0) * p1 + e2(1) * p2 + e2(2) * p3;
59  N = v1 ^ v2;
60 
61  dp[i].x = p0(0);
62  dp[i].y = p0(1);
63  dp[i].z = p0(2);
64  dp[i].nx = N(0);
65  dp[i].ny = N(1);
66  dp[i].nz = N(2);
67  }
68 
69  vbo.bind();
70  vbo.bufferData( no_dp * sizeof( GL::GLVertexNormal ), dp.getPtr(), GL_STATIC_DRAW );
71  vbo.unbind();
72  }

74  template <typename T, int n>
75  void PTriangleVisualizer<T,n>::fillTriangleIBO(GL::IndexBufferObject ibo, int m ) {
76 
77  int no_indices = (m-1)*(m-1)*3;
78 
79  DVector<GLuint> indices(no_indices);
80  GLuint *iptr = indices.getPtr();
81  for( int i = 0; i < m-1; i++ ) {
82 
83  // Index row i and row i+1
84  const int o1 = (i*(i+1))/2;
85  const int o2 = ((i+1)*(i+2))/2;
86 
87  // Upper triangles (pointing down)
88  for( int j = 1; j <= i; j++ ) {
89  *iptr++ = o1 + j;
90  *iptr++ = o1 + j - 1;
91  *iptr++ = o2 + j;
92  }
93 
94  // Lower triangles (pointing up)
95  for( int j = 1; j < i+2; j++ ) {
96  *iptr++ = o2 + j - 1;
97  *iptr++ = o2 + j;
98  *iptr++ = o1 + j - 1;
99  }
100  }
101 
102  ibo.bind();
103  ibo.bufferData( sizeof(GLuint)*no_indices, indices.getPtr(), GL_STATIC_DRAW );
104  ibo.unbind();
105  }

107  template <typename T, int n>
108  void PTriangleVisualizer<T,n>::fillTriangleStripIBO( GLuint ibo_id, int m ) {
109 
110  m -= 1;
111 
112  int no_indices = m*(m+2);
113 
114  DVector<GLushort> indices(no_indices);
115  for( int i = 0; i < m; i++ ) {
116 
117  const int o1 = 0.5 * i * (i+1);
118  const int o2 = 0.5 * (i+1) * (i+2);
119  const int o = i*(i+2);
120  const int is = 2 * i + 3;
121 
122  for( int j = 0; j < is/2; j++ ) {
123 
124  indices[o+j*2] = o2 +j;
125  indices[o+j*2+1] = o1 +j;
126  }
127 
128  indices[o+is-1] = o1+is-1;
129  }
130 
131  glBindBuffer( GL_ELEMENT_ARRAY_BUFFER, ibo_id );
132  glBufferData( GL_ELEMENT_ARRAY_BUFFER, no_indices * sizeof(GLushort), indices.getPtr(), GL_STATIC_DRAW );
133  glBindBuffer( GL_ELEMENT_ARRAY_BUFFER, 0x0 );
