68  inline
69  Point<T,n> SubSpace<T,n,m>::getClosestPoint( const Point<T,n>& p ) const {
70 
71  Point<T,n> cp = getPosC();
72  Vector<T,n> d = p - cp;
73 
74  for(int i = 0; i < m; i++) {
75  cp += ( (d * _matrix[i]) / (_matrix[i] * _matrix[i]) ) * _matrix[i];
76  }
77  return cp;
78  }
