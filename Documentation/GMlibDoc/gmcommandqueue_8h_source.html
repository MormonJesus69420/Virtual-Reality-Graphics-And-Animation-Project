176  cl_int
177  CommandQueue::enqueueReadBuffer(const Buffer &buffer, cl_bool blocking,
178  ::size_t offset, DMatrix<T>& data) const {
179 
180  DVector<T> vec(data.getDim1() * data.getDim2());
181 
182  cl_int res;
183  cl::Event event;
184  res = enqueueReadBuffer( buffer, blocking, offset, vec, 0x0, &event );
185 
186  event.wait();
187  data = vec.getPtr();
188 
189  return res;
190  }
