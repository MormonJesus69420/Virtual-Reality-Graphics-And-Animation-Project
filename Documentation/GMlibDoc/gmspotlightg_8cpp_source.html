77  void SpotLightG::init() {
78 
79  setSurroundingSphere( Sphere<float,3>( Point<float,3>( 0.0 ), 1.0 ) );
80 
81  _color_prog.acquire("color");
82  _shading_prog.acquire("directional_lighting");
83 
84 
85  // Create geometry
86  int i;
87  double k = 3.3;
88  double b = k/sqrt(1+pow(tan(_cutoff.getRad()),2));
89  double a = sqrt(k*k-b*b);
90  if(_cutoff.getDeg()>90) b = -b;
91 
92  Vector<float,3> d = _up;
93  Vector<float,3> n = d^d.getLinIndVec();
94  d.setLength(b);
95  n.setLength(a);
96  UnitVector<float,3> n2 = -d;
97  if(_cutoff.getDeg()!=90) n2 = Vector<float,3>((b/a)*n-(a/b)*d);
98 
99  Vector<float,3> v = d ^ n;
100 
101  HqMatrix<float,3> m1(Angle(30),n,v,_pos);
102  HqMatrix<float,3> m2(Angle(-30),n,v,_pos);
103 
104  Array<Point<float,3> > p1;
105  p1 += _pos;
106  p1 += _pos+d+n;
107  for(i=1; i<12; i++) p1 += m1*p1.back();
108 
109  Vector<float,3> bb=0.6*d;
110  if(b<0.0) bb.setLength(0.0);
111  Array<Point<float,3> > p2;
112  p2 += _pos+bb+0.1*_dir;
113  p2 += _pos+0.6*d+0.6*n;
114  for(i=1; i<12; i++) p2 += m2*p2.back();
115 
116  Array<Vector<float,3> > v1;
117  v1 += UnitVector<float,3>(-d);
118  v1 += n2;
119  for(i=1; i<12; i++) v1 += m1*v1.back();
120 
121  Array<Vector<float,3> > v2;
122  v2 += UnitVector<float,3>(d);
123  v2 += UnitVector<float,3>(d+n);
124  for(i=1; i<12; i++) v2 += m2*v2.back();
125 
126 
127  // the light box geometry
128  DVector<GL::GLVertexNormal> dp_box(p1.size()+1);
129  for(i=0; i<p1.size(); i++) {
130  GL::GLVertexNormal box_p;
131  box_p.x = p1(i)(0);
132  box_p.y = p1(i)(1);
133  box_p.z = p1(i)(2);
134  box_p.nx = v1(i)(0);
135  box_p.ny = v1(i)(1);
136  box_p.nz = v1(i)(2);
137  dp_box[i] = box_p;
138  }
139  dp_box[dp_box.getDim()-1] = dp_box[1];
140 
141  _light_box_geom_elements = dp_box.getDim();
142  _light_box_geom_vbo.create();
143  _light_box_geom_vbo.bufferData( _light_box_geom_elements * sizeof(GL::GLVertexNormal), dp_box.getPtr(), GL_STATIC_DRAW );
144 
145  // the light geometry
146  DVector<GL::GLVertexNormal> dp_light(p2.getSize()+1);
147  for(i=0; i<p2.size(); i++) {
148  GL::GLVertexNormal light_p;
149  light_p.x = p2(i)(0);
150  light_p.y = p2(i)(1);
151  light_p.z = p2(i)(2);
152  light_p.nx = v2(i)(0);
153  light_p.ny = v2(i)(1);
154  light_p.nz = v2(i)(2);
155  dp_light[i] = light_p;
156  }
157  dp_light[dp_light.getDim()-1] = dp_light[1];
158 
159  _light_geom_elements = dp_light.getDim();
160  _light_geom_vbo.create();
161  _light_geom_vbo.bufferData( _light_geom_elements * sizeof(GL::GLVertexNormal), dp_light.getPtr(), GL_STATIC_DRAW );
162  }
