68  template <typename T, int n>
69  void PSurfDefaultVisualizer<T,n>::render( const SceneObject* obj, const DefaultRenderer* renderer ) const {
70 
71  const Camera* cam = renderer->getCamera();
72  const HqMatrix<float,3> &mvmat = obj->getModelViewMatrix(cam);
73  const HqMatrix<float,3> &pmat = obj->getProjectionMatrix(cam);
74 // const SqMatrix<float,3> &nmat = obj->getNormalMatrix(cam);
75 
76  SqMatrix<float,3> nmat = mvmat.getRotationMatrix();
77 // nmat.invertOrthoNormal();
78 // nmat.transpose();
79 
80  this->glSetDisplayMode();
81 
82  _prog.bind(); {
83 
84  // Model view and projection matrices
85  _prog.uniform( "u_mvmat", mvmat );
86  _prog.uniform( "u_mvpmat", pmat * mvmat );
87  _prog.uniform( "u_nmat", nmat );
88 
89  // Lights
90  _prog.bindBufferBase( "DirectionalLights", renderer->getDirectionalLightUBO(), 0 );
91  _prog.bindBufferBase( "PointLights", renderer->getPointLightUBO(), 1 );
92  _prog.bindBufferBase( "SpotLights", renderer->getSpotLightUBO(), 2 );
93 
94  // Material
95  const Material &m = obj->getMaterial();
96  _prog.uniform( "u_mat_amb", m.getAmb() );
97  _prog.uniform( "u_mat_dif", m.getDif() );
98  _prog.uniform( "u_mat_spc", m.getSpc() );
99  _prog.uniform( "u_mat_shi", m.getShininess() );
100 
101  // Normal map
102  _prog.uniform( "u_nmap", _nmap, GLenum(GL_TEXTURE0), 0 );
103 
104  // Get vertex and texture attrib locations
105  GL::AttributeLocation vert_loc = _prog.getAttributeLocation( "in_vertex" );
106  GL::AttributeLocation tex_loc = _prog.getAttributeLocation( "in_tex" );
107 
108  // Bind and draw
109  _vbo.bind();
110  _vbo.enable( vert_loc, 3, GL_FLOAT, GL_FALSE, sizeof(GL::GLVertexTex2D), reinterpret_cast<const GLvoid *>(0x0) );
111  _vbo.enable( tex_loc, 2, GL_FLOAT, GL_FALSE, sizeof(GL::GLVertexTex2D), reinterpret_cast<const GLvoid *>(3*sizeof(GLfloat)) );
112  draw();
113  _vbo.disable( vert_loc );
114  _vbo.disable( tex_loc );
115  _vbo.unbind();
116 
117  } _prog.unbind();
118  }

164  template<typename T,int n>
165  void PSurfDefaultVisualizer<T,n>::initShaderProgram() {
166 
167  const std::string prog_name = "psurf_default_prog";
168  if( _prog.acquire(prog_name) ) return;
169 
170 
171  std::string vs_src =
172  GL::OpenGLManager::glslDefHeaderVersionSource() +
173 
174  "uniform mat4 u_mvmat, u_mvpmat;\n"
175  "\n"
176  "in vec4 in_vertex;\n"
177  "in vec2 in_tex;\n"
178  "\n"
179  "out vec4 gl_Position;\n"
180  "\n"
181  "smooth out vec3 ex_pos;\n"
182  "smooth out vec2 ex_tex;\n"
183  "\n"
184  "void main() {\n"
185  "\n"
186  " vec4 v_pos = u_mvmat * in_vertex;\n"
187  " ex_pos = v_pos.xyz * v_pos.w;\n"
188  "\n"
189  " ex_tex = in_tex;\n"
190  "\n"
191  " gl_Position = u_mvpmat * in_vertex;\n"
192  "}\n"
193  ;
194 
195  std::string fs_src =
196  GL::OpenGLManager::glslDefHeaderVersionSource() +
197  GL::OpenGLManager::glslFnComputeBlinnPhongLightingSource() +
198 
199  "uniform sampler2D u_nmap;\n"
200  "uniform mat4 u_mvmat;\n"
201  "uniform mat3 u_nmat;\n"
202  "\n"
203  "uniform vec4 u_mat_amb;\n"
204  "uniform vec4 u_mat_dif;\n"
205  "uniform vec4 u_mat_spc;\n"
206  "uniform float u_mat_shi;\n"
207  "\n"
208  "smooth in vec3 ex_pos;\n"
209  "smooth in vec2 ex_tex;\n"
210  "\n"
211  "out vec4 gl_FragColor;\n"
212  "\n"
213  "void main() {\n"
214  "\n"
215  " vec3 nmap_normal = texture( u_nmap, ex_tex.ts).xyz;\n"
216  " vec3 normal = normalize( u_nmat * nmap_normal );\n"
217  "\n"
218  " Material mat;\n"
219  " mat.ambient = u_mat_amb;\n"
220  " mat.diffuse = u_mat_dif;\n"
221  " mat.specular = u_mat_spc;\n"
222  " mat.shininess = u_mat_shi;\n"
223  "\n"
224  " gl_FragColor = computeBlinnPhongLighting( mat, ex_pos, normal );\n"
225  "\n"
226  "}\n"
227  ;
228 
229  bool compile_ok, link_ok;
230 
231  GL::VertexShader vshader;
232  vshader.create("psurf_default_vs");
233  vshader.setPersistent(true);
234  vshader.setSource(vs_src);
235  compile_ok = vshader.compile();
236  if( !compile_ok ) {
237  std::cout << "Src:" << std::endl << vshader.getSource() << std::endl << std::endl;
238  std::cout << "Error: " << vshader.getCompilerLog() << std::endl;
239  }
240  assert(compile_ok);
241 
242  GL::FragmentShader fshader;
243  fshader.create("psurf_default_fs");
244  fshader.setPersistent(true);
245  fshader.setSource(fs_src);
246  compile_ok = fshader.compile();
247  if( !compile_ok ) {
248  std::cout << "Src:" << std::endl << fshader.getSource() << std::endl << std::endl;
249  std::cout << "Error: " << fshader.getCompilerLog() << std::endl;
250  }
251  assert(compile_ok);
252 
253  _prog.create(prog_name);
254  _prog.setPersistent(true);
255  _prog.attachShader(vshader);
256  _prog.attachShader(fshader);
257  link_ok = _prog.link();
258  if( !link_ok ) {
259  std::cout << "Error: " << _prog.getLinkerLog() << std::endl;
260  }
261  assert(link_ok);
262  }
