135  template <typename T>
136  void PHermiteSurface<T>::eval(T u, T v, int d1, int d2, bool /*lu*/, bool /*lv*/) const {
137 
138  // set result set dim
139  this->_p.setDim( d1+1, d2+1 );
140 
141  // vars
142  Point<float,3> p(0.0f);
143 
144  // interpolate u in v direction
145  for( int i = 0; i < _c1.getSize(); i++ )
146  p += getH( _c2.getSize()/2, i, v ) * (_c1(i)->evaluateParent(u, 1))(0);
147 
148  // interpolate v in u direction
149  for( int i = 0; i < _c2.getSize(); i++ )
150  p += getH( _c1.getSize()/2, i, u ) * (_c2(i)->evaluateParent(v, 1))(0);
151 
152  // bi-linear interpolation
153  p -=
154  _b(0)(0) * (1.0f - u) * (1.0f - v) +
155  _b(0)(1) * (1.0f - u) * v +
156  _b(1)(0) * u * (1.0f - v) +
157  _b(1)(1) * u * v;
158 
159  this->_p[0][0] = p;
160 
161 
162  if( this->_dm == GM_DERIVATION_EXPLICIT ) {
163 
164  if( d1 > 0 ) { // Su
165 
166  p = Point<float,3>(0.0f);
167 
168  // interpolate u in v direction
169  for( int i = 0; i < _c1.getSize(); i++ )
170  p += getH( _c2.getSize()/2, i, v ) * (_c1(i)->evaluateParent(u, 1))(1);
171 
172  // interpolate v in u direction
173  for( int i = 0; i < _c2.getSize(); i++ )
174  p += getHder( _c1.getSize()/2, i, u ) * (_c2(i)->evaluateParent(v, 1))(0);
175 
176  // bi-linear interpolation
177  p -=
178  _b(0)(0) * ( v - 1.0f ) +
179  _b(0)(1) * ( -v ) +
180  _b(1)(0) * (1.0f - v) +
181  _b(1)(1) * v;
182 
183  this->_p[1][0] = p;
184  }
185 
186 
187 
188  if( d2 > 0 ) { // Sv
189 
190  p = Point<float,3>(0.0f);
191 
192  // interpolate u in v direction
193  for( int i = 0; i < _c1.getSize(); i++ )
194  p += getHder( _c2.getSize()/2, i, v ) * (_c1(i)->evaluateParent(u, 1))(0);
195 
196  // interpolate v in u direction
197  for( int i = 0; i < _c2.getSize(); i++ )
198  p += getH( _c1.getSize()/2, i, u ) * (_c2(i)->evaluateParent(v, 1))(1);
199 
200  // bi-linear interpolation
201  p -=
202  _b(0)(0) * (u - 1.0f) +
203  _b(0)(1) * (1.0f - u) +
204  _b(1)(0) * ( -u ) +
205  _b(1)(1) * u;
206 
207  this->_p[0][1] = p;
208  }
209 
210 
211 
212  if( d1 > 0 && d2 > 0 ) {
213 
214  p = Point<float,3>(0.0f);
215 
216  // interpolate u in v direction
217  for( int i = 0; i < _c1.getSize(); i++ )
218  p += getHder( _c2.getSize()/2, i, v ) * (_c1(i)->evaluateParent(u, 1))(1);
219 
220  // interpolate v in u direction
221  for( int i = 0; i < _c2.getSize(); i++ )
222  p += getHder( _c1.getSize()/2, i, u ) * (_c2(i)->evaluateParent(v, 1))(1);
223 
224  // bi-linear interpolation
225  p -=
226  _b(0)(0) * (1.0f - u) * (1.0f - v) +
227  _b(0)(1) * (1.0f - u) * v +
228  _b(1)(0) * u * (1.0f - v) +
229  _b(1)(1) * u * v;
230 
231  this->_p[1][1] = p;
232  }

280  inline
281  T PHermiteSurface<T>::getH( int d, int k, T t ) const {
282 
283  if(d==2)
284  {
285  switch (k)
286  {
287  case 1: return (2*t-3)*t*t+1;
288  case 2: return (3-2*t)*t*t;
289  case 3: return (t*t-2*t+1)*t;
290  case 4: return (t-1)*t*t;
291  }
292  }
293 
294  // if(d==1)
295  if(k==1) return t;
296  else return 1.0-t;
297  }

311 
312  if(d==2)
313  {
314  switch (k)
315  {
316  case 1: return (t-1)*t*6;
317  case 2: return (1-t)*t*6;
318  case 3: return 3*t*t-4*t+1;
319  case 4: return (3*t-2)*t;

344  template <typename T>
345  bool PHermiteSurface<T>::isValidCoonsPatch() {
346 
347  // Check number of curves
348  if( _c1.getSize() < 2 || _c2.getSize() < 2 )
349  return false;
350 
351  // Check that the corner points are joined together (b00, b10, b01 and b11)
352 
353  // _c1 -> c1/c2, _c2 -> f1/f2
354  Point<T,3> c1_s, c1_e, c2_s, c2_e;
355  Point<T,3> f1_s, f1_e, f2_s, f2_e;
356  c1_s = _c1[0]->evaluateParent( _c1[0]->getParStart(), 0 )[0];
357  c1_e = _c1[0]->evaluateParent( _c1[0]->getParEnd(), 0 )[0];
358  c2_s = _c1[1]->evaluateParent( _c1[1]->getParStart(), 0 )[0];
359  c2_e = _c1[1]->evaluateParent( _c1[1]->getParEnd(), 0 )[0];
360  f1_s = _c2[0]->evaluateParent( _c2[0]->getParStart(), 0 )[0];
361  f1_e = _c2[0]->evaluateParent( _c2[0]->getParEnd(), 0 )[0];
362  f2_s = _c2[1]->evaluateParent( _c2[1]->getParStart(), 0 )[0];
363  f2_e = _c2[1]->evaluateParent( _c2[1]->getParEnd(), 0 )[0];
364 
365 
366  // Junction c1 x c2
367  if( std::fabs( (c1_s - f1_s).getLength() ) > 1e-5 ) return false;
368 
369  // Junction c1 x c4
370  if( std::fabs( (c1_e - f2_s).getLength() ) > 1e-5 ) return false;
371 
372  // Junction c2 x c3
373  if( std::fabs( (f1_e - c2_s).getLength() ) > 1e-5 ) return false;
374 
375  // Junction c3 x c4
376  if( std::fabs( (c2_e - f2_e).getLength() ) > 1e-5 ) return false;
377 
378  // return true if all criterias is met
379  return true;
380  }
