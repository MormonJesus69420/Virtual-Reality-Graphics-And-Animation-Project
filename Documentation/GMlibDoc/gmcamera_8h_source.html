206  int isInsideFrustum(const Sphere<float,3>& s) const {
207 
208  if(!s.isValid()) return -1;
209  int ret = 1;
210 
211  Vector<float,3> d = s.getPos()-_frustum_p[0];
212  float dv = d*_frustum_v[1]; // Høyre
213  if(dv >= s.getRadius()) return -1;
214  else if(dv > -s.getRadius()) ret = 0;
215  dv = d*_frustum_v[2]; // Opp
216  if(dv >= s.getRadius()) return -1;
217  else if(dv > -s.getRadius()) ret = 0;
218  dv = d*_frustum_v[4]; // Bak
219  if(dv >= s.getRadius()) return -1;
220  else if(dv > -s.getRadius()) ret = 0;
221 
222  d = s.getPos()-_frustum_p[1];
223  dv = d*_frustum_v[0]; // Venstre
224  if(dv >= s.getRadius()) return -1;
225  else if(dv > -s.getRadius()) ret = 0;
226  dv = d*_frustum_v[3]; // Ned
227  if(dv >= s.getRadius()) return -1;
228  else if(dv > -s.getRadius()) ret = 0;
229  dv = d*_frustum_v[5]; // Fram
230  if(dv >= s.getRadius()) return -1;
231  else if(dv > -s.getRadius()) ret = 0;
232  return ret;
233  }

237  virtual void computeProjectionMatrix() {
238 
239  float hh = _frustum_near* _frustum_angle_tan;
240  float rr = getAspectRatio()*hh;
241 
242  float l, r, b, t, n, f;
243  l = -rr;
244  r = rr;
245  b = -hh;
246  t = hh;
247  n = _frustum_near;
248  f = _frustum_far;
249 
250  float A, B, C, D;
251  A = ( r + l ) / ( r - l );
252  B = ( t + b ) / ( t - b );
253  C = - ( f + n ) / ( f - n );
254  D = - 2 * f * n / ( f - n );
255 
256  _projection_matrix[0][0] = 2.0f / (r - l);
257  _projection_matrix[0][1] = 0.0f;
258  _projection_matrix[0][2] = A;
259  _projection_matrix[0][3] = 0.0f;
260 
261  _projection_matrix[1][0] = 0.0f;
262  _projection_matrix[1][1] = 2.0f / ( t - b );
263  _projection_matrix[1][2] = B;
264  _projection_matrix[1][3] = 0.0f;
265 
266  _projection_matrix[2][0] = 0.0f;
267  _projection_matrix[2][1] = 0.0f;
268  _projection_matrix[2][2] = C;
269  _projection_matrix[2][3] = D;
270 
271  _projection_matrix[3][0] = 0.0f;
272  _projection_matrix[3][1] = 0.0f;
273  _projection_matrix[3][2] = -1.0f;
274  _projection_matrix[3][3] = 0.0f;
275  }

277  virtual void computeFrustumBounds() {
278 
279  float ratio = getAspectRatio() * _frustum_angle_tan;
280  double rr = sqrt(1+ratio*ratio);
281  double tt = sqrt(1+_frustum_angle_tan*_frustum_angle_tan);
282  Vector<float,3> f = _matrix_scene*_dir;
283  Vector<float,3> oe = _matrix_scene*_up;
284  Vector<float,3> ve = _matrix_scene*_side;
285  Vector<float,3> pp = ratio*ve-_frustum_angle_tan*oe;
286  _frustum_p[0] = _matrix_scene*_pos; // Venstre, høyre, opp, ned (posisjon)
287  _frustum_p[1] = _frustum_p[0]+_frustum_far*(f+pp);
288  _frustum_p[0] += _frustum_near*(f-pp);
289 
290  _frustum_v[0] = ve-ratio*f; // Venstre (normal)
291  _frustum_v[1] = (_frustum_v[0]-2*ve)/rr; // Høyre (normal)
292  _frustum_v[0] /= rr;
293  _frustum_v[2] = oe-_frustum_angle_tan*f; // Opp (normal)
294  _frustum_v[3] = (_frustum_v[2]-2*oe)/tt; // ned (normal)
295  _frustum_v[2] /= tt;
296  _frustum_v[4] = -f; // Bak (normal)
297  _frustum_v[5] = f; // Fram (normal)
298 
299  }

301  virtual void computeFrustumFrame() {
302 
303  float ratio = getAspectRatio();
304 
305  Point<float,3> p1(0,0,-_frustum_near);
306  Point<float,3> p2(0,0,-_frustum_far);
307 
308  Vector<float,3> v1(0,_frustum_angle_tan*_frustum_near,0);
309  Vector<float,3> v2(-ratio*_frustum_angle_tan*_frustum_near,0,0);
310  Vector<float,3> v3(0,_frustum_angle_tan*_frustum_far,0);
311  Vector<float,3> v4(-ratio*_frustum_angle_tan*_frustum_far,0,0);
312 
313  Point<float,3> p1m(p1-v1);
314  Point<float,3> p2m(p2-v3);
315  p1 += v1;
316  p2 += v3;
317 
318  _frustum_frame[0] = p1 + v2;
319  _frustum_frame[1] = p1 - v2;
320  _frustum_frame[2] = p1m - v2;
321  _frustum_frame[3] = p1m + v2;
322  _frustum_frame[4] = p2 + v4;
323  _frustum_frame[5] = p2 - v4;
324  _frustum_frame[6] = p2m - v4;
325  _frustum_frame[7] = p2m + v4;
326  }
