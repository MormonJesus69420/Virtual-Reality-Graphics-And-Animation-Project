91  void PChrysanthemumCurve<T>::eval( T t, int d, bool /*l*/ ) const {
92 
93  this->_p.setDim( d+1 );
94 
95  const double p4 = sin(17*t/3);
96  const double p8 = sin(2*cos(3*t) - 28*t);
97  const double r = 5*(1+sin(11*t/5)) - 4*pow(p4,4) * pow(p8,8);
98 
99  this->_p[0][0] = _r * T(r * cos(t));
100  this->_p[0][1] = _r * T(r * sin(t));
101  this->_p[0][2] = _r * T((r/20 + _trans) * sin(_scale * r));
102  }
