52  void SurroundingSphereVisualizer::render(const SceneObject *obj, const DefaultRenderer */*renderer*/) const{
53 
54  int poly_mode;
55  glGetIntegerv( GL_POLYGON_MODE, &poly_mode );
56  if( _wireframe )
57  glPolygonMode(GL_FRONT_AND_BACK, GL_LINE);

59  glPolygonMode(GL_FRONT_AND_BACK, GL_FILL);
60 
61  // Display Clean Surrounding Sphere

63 
64  displaySphere( obj->getSurroundingSphere() );

66 
67  // Display Surrounding Sphere

69

71  }
72 
73  glPolygonMode(GL_FRONT_AND_BACK, poly_mode);
74  }
