158  template <typename T, int n>
159  void PTriangle<T,n>::_fuForm( T u, T v, T& E, T& F, T& G, T& e, T& f, T& g) const
160  {
161  _eval( u, v, 2);
162  Vector<T,n> du = _p[2]-_p[1];
163  Vector<T,n> dv = _p[3]-_p[1];
164  UnitVector<T,n> N = du^dv;
165 
166  Vector<T,n> duu = _p[6] - 2*_p[9] + _p[4];
167  Vector<T,n> duv = getDerUV(u,v);
168  Vector<T,n> dvv = _p[5] - 2*_p[8] + _p[4];
169 
170  E = du*du;
171  F = du*dv;
172  G = dv*dv;
173  e = N*duu;
174  f = N*duv;
175  g = N*dvv;
176  }

332  template <typename T, int n>
333  bool PTriangle<T,n>::isClosestPoint( const Point<T,n>& q, T& u, T& v ) const {
334 
335  T a11,a12,a21,a22,b1,b2;
336  T du,dv,det;
337 
338  HqMatrix<float,3> invmat = this->_present;
339  invmat.invertOrthoNormal();
340  Point<T,n> p = (invmat * q).template toType<T>(); // Egentlig _present
341 
342  for(int i=0; i<20;i++) {
343 
344  _eval( u, v, 1 );
345 
346  Vector<T,n> d = p - _p[0];
347 
348  a11 = d*(_p[7]-2*_p[5]+_p[4])-(_p[2]-_p[1])*(_p[2]-_p[1]);
349  a12 = d*(_p[8]-_p[6]-_p[5]+_p[4])-(_p[2]-_p[1])*(_p[3]-_p[1]);
350  a21 = a12;
351  a22 = d*(_p[9]-2*_p[6]+_p[4])-(_p[3]-_p[1])*(_p[3]-_p[1]);
352 
353  b1 = -(d*(_p[2]-_p[1]));
354  b2 = -(d*(_p[3]-_p[1]));
355 
356  det = a11*a22-a12*a21;
357  du = (b1*a22-a12*b2)/det;
358  dv = (a11*b2-b1*a21)/det;
359  if(fabs(du)< 1e-6 && fabs(dv) < 1e-6) break;
360  u += du;
361  v += dv;
362  }
363  return true;
364  }

408  template <typename T, int n>
409  inline
410  void PTriangle<T,n>::resample1( DVector< DVector < Vector<T,n> > >& p, int m, int d )
411  {
412  T u,v,du = T(1)/(m-1);
413  p.setDim(_sum(m));
414  int i,j,k;
415  for(k=0,i=0; i<m; i++) {
416  for( j=0; j<=i; j++ )
417  {
418  v = j*du;
419  u = (i-j)*du;
420  eval(u, v, 1-u-v, d);
421  p[k++] = _p;
422  }
423  }
424  }

427  template <typename T, int n>
428  inline
429  void PTriangle<T,n>::resample2( DVector< DVector < Vector<T,n> > >& p, int m, int a, int b)
430  {
431  T u,v,w,du = T(1)/(m-1);
432  p.setDim(_sum(m));
433  int i,j,k;
434  for(k=0,i=0; i<m; i++) {
435  for( j=0; j<=i; j++ )
436  {
437  v = j*du;
438  u = i*du;
439  w = 1 - u;
440  u -= v;
441  Point<T,n> pr = u*_pt[0] + v*_pt[a] + w*_pt[b];
442  eval(pr, 1);
443  p[k++] = _p;
444  }
445  }
