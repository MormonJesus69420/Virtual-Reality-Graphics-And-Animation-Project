77  template <typename T>
78  inline
79  void BasisTriangleERBS<T>::computeB1( T u, T v, T w )
80  {
81  DVector<T> Bu; getB( Bu, u, 2 );
82  DVector<T> Bv; getB( Bv, v, 2 );
83  DVector<T> Bw; getB( Bw, w, 2 );
84 
85  T sum = Bu[0] + Bv[0] + Bw[0];
86  T sum2 = sum*sum;
87  T sum3 = sum2*sum;
88 
89  _A[0][0] = Bu[0]/sum; // B1
90  _A[0][1] = Bv[0]/sum; // B2
91  _A[0][2] = Bw[0]/sum; // B3
92 
93  T aa = Bu[1]/sum2;
94  _A[1][0] = aa * ( sum - Bu[0] ); // du B1
95  _A[1][1] = -aa * Bv[0]; // B2
96  _A[1][2] = -aa * Bw[0]; // B3
97  aa = Bv[1]/sum2;
98  _A[2][0] = -aa * Bu[0]; // dv B1
99  _A[2][1] = aa * ( sum - Bv[0] ); // B2
100  _A[2][2] = -aa * Bw[0]; // B3
101  aa = Bw[1]/sum2;
102  _A[3][0] = -aa * Bu[0]; // dw B1
103  _A[3][1] = -aa * Bv[0]; // B2
104  _A[3][2] = aa * ( sum - Bw[0] ); // B3
105 
106  aa = (2*Bu[1]*Bu[1]/sum - Bu[2])/sum2;
107  _A[4][0] = aa * (Bu[0]-sum); // duu B1
108  _A[4][1] = aa * (Bv[0]); // B2
109  _A[4][2] = aa * (Bw[0]); // B3
110  aa = Bu[1]*Bv[1]/sum3;
111  _A[5][0] = aa*(2*Bu[0]-sum); // duv B1
112  _A[5][1] = aa*(2*Bv[0]-sum); // B2
113  _A[5][2] = aa*(2*Bw[0]); // B3
114  aa = Bu[1]*Bw[1]/sum3;
115  _A[6][0] = aa*(2*Bu[0]-sum); // duw B1
116  _A[6][1] = aa*(2*Bv[0]); // B2
117  _A[6][2] = aa*(2*Bw[0]-sum); // B3
118  aa = (2*Bv[1]*Bv[1]/sum - Bv[2])/sum2;
119  _A[7][0] = aa * (Bu[0]); // dvv B1
120  _A[7][1] = aa * (Bv[0]-sum); // B2
121  _A[7][2] = aa * (Bw[0]); // B3
122  aa = Bv[1]*Bw[1]/sum3;
123  _A[8][0] = aa*(2*Bu[0]); // dvw B1
124  _A[8][1] = aa*(2*Bv[0]-sum); // B2
125  _A[8][2] = aa*(2*Bw[0]-sum); // B3
126  aa = (2*Bw[1]*Bw[1]/sum - Bw[2])/sum2;
127  _A[9][0] = aa * (Bu[0]); // dww B1
128  _A[9][1] = aa * (Bv[0]); // B2
129  _A[9][2] = aa * (Bw[0]-sum); // B3
130  }

136  template <typename T>
137  inline
138  void BasisTriangleERBS<T>::computeB2( T u, T v, T w )
139  {
140  int k=2;
141 
142  T uu = pow(u,k);
143  T vv = pow(v,k);
144  T ww = pow(w,k);
145  T sum = uu + vv + ww;
146  T sum2 = pow(sum,k);
147 
148  _A[0][0] = uu/sum;
149  _A[0][1] = vv/sum;
150  _A[0][2] = ww/sum;
151 
152  _A[1][0] = k*pow(u,k-1)*(sum - uu)/sum2;
153  _A[1][1] = k*pow(v,k-1)*(-uu)/sum2;
154  _A[1][2] = k*pow(w,k-1)*(-uu)/sum2;
155 
156  _A[2][0] = k*pow(u,k-1)*(-vv)/sum2;
157  _A[2][1] = k*pow(v,k-1)*(sum - vv)/sum2;
158  _A[2][2] = k*pow(w,k-1)*(-vv)/sum2;
159 
160  _A[3][0] = k*pow(u,k-1)*(-ww)/sum2;
161  _A[3][1] = k*pow(v,k-1)*(-ww)/sum2;
162  _A[3][2] = k*pow(w,k-1)*(sum - ww)/sum2;
163 
164  _A[4][0] = k*(k-1)*pow(u,k-2)*(sum - uu)/sum2 ;
165  _A[4][1] = k*pow(v,k-1)*(-uu)/sum2;
166  _A[4][2] = k*pow(w,k-1)*(-uu)/sum2;
167  }
