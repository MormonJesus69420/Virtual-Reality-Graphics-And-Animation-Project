91  void computeProjectionMatrix() override {
92 
93  float hh = _horizontal;
94  float rr = getAspectRatio()*_horizontal;
95 
96  float l, r, b, t, n, f;
97  l = -rr;
98  r = rr;
99  b = -hh;
100  t = hh;
101  n = getNearPlane();
102  f = getFarPlane();
103 
104  float tx, ty, tz;
105  tx = -(r+l)/(r-l);
106  ty = -(t+b)/(t-b);
107  tz = -(f+n)/(f-n);
108 
109  _projection_matrix[0][0] = 2.0f / (r - l);
110  _projection_matrix[0][1] = 0.0f;
111  _projection_matrix[0][2] = 0.0f;
112  _projection_matrix[0][3] = tx;
113 
114  _projection_matrix[1][0] = 0.0f;
115  _projection_matrix[1][1] = 2.0f / ( t - b );
116  _projection_matrix[1][2] = 0.0f;
117  _projection_matrix[1][3] = ty;
118 
119  _projection_matrix[2][0] = 0.0f;
120  _projection_matrix[2][1] = 0.0f;
121  _projection_matrix[2][2] = (-2.0f)/(f-n);
122  _projection_matrix[2][3] = tz;
123 
124  _projection_matrix[3][0] = 0.0f;
125  _projection_matrix[3][1] = 0.0f;
126  _projection_matrix[3][2] = 0.0f;
127  _projection_matrix[3][3] = 1.0f;
128  }

131  void computeFrustumBounds() override {
132 
133  float ratio = getAspectRatio();
134 
135  Vector<float,3> f = _matrix_scene*_dir;
136  Vector<float,3> oe = _matrix_scene*_up;
137  Vector<float,3> ve = _matrix_scene*_side;
138  _frustum_p[0] = _matrix_scene*_pos;
139  _frustum_p[1] = _frustum_p[0]+_frustum_far*f+_horizontal*(ratio*ve-oe);
140  _frustum_p[0] += _frustum_near*f-_horizontal*(ratio*ve-oe);
141  _frustum_v[0] = ve; // Venstre (normal)
142  _frustum_v[1] = -ve; // HÃ¸yre (normal)
143  _frustum_v[2] = oe; // Opp (normal)
144  _frustum_v[3] = -oe; // ned (normal)
145  _frustum_v[4] = -f; // Bak (normal)
146  _frustum_v[5] = f; // Fram (normal)
147  }

150  void computeFrustumFrame() override {
151 
152  float ratio = getAspectRatio();
153 
154  Point<float,3> p1(0,0,-_frustum_near);
155  Point<float,3> p2(0,0,-_frustum_far);
156 
157  Vector<float,3> v1(0,_horizontal,0);
158  Vector<float,3> v2(-ratio*_horizontal,0,0);
159  Vector<float,3> v3(0,_horizontal,0);
160  Vector<float,3> v4(-ratio*_horizontal,0,0);
161 
162 
163  Point<float,3> p1m(p1-v1);
164  Point<float,3> p2m(p2-v3);
165  p1 += v1;
166  p2 += v3;
167 
168  _frustum_frame[0] = p1 + v2;
169  _frustum_frame[1] = p1 - v2;
170  _frustum_frame[2] = p1m - v2;
171  _frustum_frame[3] = p1m + v2;
172  _frustum_frame[4] = p2 + v4;
173  _frustum_frame[5] = p2 - v4;
174  _frustum_frame[6] = p2m - v4;
175  _frustum_frame[7] = p2m + v4;
176 
177  }
