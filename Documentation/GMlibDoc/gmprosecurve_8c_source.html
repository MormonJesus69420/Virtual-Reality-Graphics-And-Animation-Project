131  template <typename T>
132  void PRoseCurve<T>::eval( T t, int d, bool /*l*/ ) const {
133 
134  this->_p.setDim(d + 1);
135 
136  this->_p[0][0] = _r * T(cos(t)*cos(1.75*t));
137  this->_p[0][1] = _r * T(sin(t)*cos(1.75*t));
138  this->_p[0][2] = _rize*(this->_p[0][0]*this->_p[0][0] + this->_p[0][1]*this->_p[0][1]);
139 
140  if( this->_dm == GM_DERIVATION_EXPLICIT ) {
141  if(d > 0) {
142  this->_p[1][0] = _r * T(-1.75*cos(t) * sin(1.75*t) - sin(t)*cos(1.75*t));
143  this->_p[1][1] = _r * T(-1.75*sin(t) * sin(1.75*t) + cos(t)*cos(1.75*t));
144  this->_p[1][2] = 2*_rize*(this->_p[0][0]*this->_p[1][0] + this->_p[0][1]*this->_p[1][1]);
145  if(d > 1) {
146  this->_p[2][0]= _r * T( 3.5*sin(t)*sin(1.75*t) - 4.0625*cos(t)*cos(1.75*t));
147  this->_p[2][1]= _r * T(-3.5*cos(t)*sin(1.75*t) - 4.0625*sin(t)*cos(1.75*t));
148  this->_p[2][2]= 2*_rize*(this->_p[1][0]*this->_p[1][0] + this->_p[0][0]*this->_p[2][0]+
149  this->_p[1][1]*this->_p[1][1] + this->_p[0][1]*this->_p[2][1]);
150  }
