69 template <typename T, int n>
70 void PSurfParamLinesVisualizer<T,n>::render( const SceneObject* obj, const DefaultRenderer * renderer) const {
71 
72  const Camera* cam = renderer->getCamera();
73 
74  const HqMatrix<float,3> &mvmat = obj->getModelViewMatrix(cam);
75  const HqMatrix<float,3> &pmat = obj->getProjectionMatrix(cam);
76 
77  this->glSetDisplayMode();
78 
79  _prog.bind(); {
80 
81  // Model view and projection matrices
82  _prog.uniform( "u_mvmat", mvmat );
83  _prog.uniform( "u_mvpmat", pmat * mvmat );
84 
85  // Lights
86  _prog.bindBufferBase( "DirectionalLights", renderer->getDirectionalLightUBO(), 0 );
87  _prog.bindBufferBase( "PointLights", renderer->getPointLightUBO(), 1 );
88  _prog.bindBufferBase( "SpotLights", renderer->getSpotLightUBO(), 2 );
89 
90  // Base material
91  _prog.uniform( "u_mat_amb", obj->getMaterial().getAmb() );
92  _prog.uniform( "u_mat_dif", obj->getMaterial().getDif() );
93  _prog.uniform( "u_mat_spc", obj->getMaterial().getSpc() );
94  _prog.uniform( "u_mat_shi", obj->getMaterial().getShininess() );
95 
96  // Line Material
97  _prog.uniform( "u_mat_line_amb", _mat.getAmb() );
98  _prog.uniform( "u_mat_line_dif", _mat.getDif() );
99  _prog.uniform( "u_mat_line_spc", _mat.getSpc() );
100  _prog.uniform( "u_mat_line_shi", _mat.getShininess() );
101 
102  // Normal map
103  _prog.uniform( "u_nmap", _nmap, (GLenum)GL_TEXTURE0, 0 );
104  _prog.uniform( "u_ptex_u", _ptex_u, (GLenum)GL_TEXTURE1, 1 );
105  _prog.uniform( "u_ptex_v", _ptex_v, (GLenum)GL_TEXTURE2, 2 );
106 
107  // Get vertex and texture attrib locations
108  GL::AttributeLocation vert_loc = _prog.getAttributeLocation( "in_vertex" );
109  GL::AttributeLocation tex_loc = _prog.getAttributeLocation( "in_tex" );
110 
111  // Bind and draw
112  _vbo.bind();
113  _vbo.enable( vert_loc, 3, GL_FLOAT, GL_FALSE, sizeof(GL::GLVertexTex2D), reinterpret_cast<const GLvoid *>(0x0) );
114  _vbo.enable( tex_loc, 2, GL_FLOAT, GL_FALSE, sizeof(GL::GLVertexTex2D), reinterpret_cast<const GLvoid *>(3*sizeof(GLfloat)) );
115 
116  draw();
117 
118  _vbo.disable( vert_loc );
119  _vbo.disable( tex_loc );
120  _vbo.unbind();
121 
122  } _prog.unbind();
123 }

160 template <typename T, int n>
161 void PSurfParamLinesVisualizer<T,n>::generatePTex(int m1, int m2, int s1, int s2, bool closed_u, bool closed_v) {
162 
163  DVector<float> tex_data_u, tex_data_v;
164 
165  if( m1 < 2 || m2 < 2 || s1 ==0 || s2 == 0 ) {
166 
167  tex_data_u = DVector<float>(1,0.0f);
168  tex_data_v = DVector<float>(1,0.0f);
169  }
170  else {
171 
172  int m1_t, m2_t;
173  m1_t = (m1-1)*(s1+1) + 1;
174  m2_t = (m2-1)*(s2+1) + 1;
175 
176  tex_data_u = DVector<float>(m1_t,0.0f);
177  tex_data_v = DVector<float>(m2_t,0.0f);
178 
179  for( int i = 0; i < m1_t; i += (s1+1) ) tex_data_u[i] = 1.0f;
180  for( int i = 0; i < m2_t; i += (s2+1) ) tex_data_v[i] = 1.0f;
181  }
182 
183  // Create U line tex
184  _ptex_u.texImage2D( 0, GL_R8, tex_data_u.getDim(), 1, 0, GL_RED, GL_FLOAT, tex_data_u.getPtr() );
185  _ptex_u.texParameteri( GL_TEXTURE_MIN_FILTER, GL_LINEAR );
186  _ptex_u.texParameteri( GL_TEXTURE_MAG_FILTER, GL_LINEAR );
187  if( closed_u ) _ptex_u.texParameterf(GL_TEXTURE_WRAP_S, GL_CLAMP_TO_EDGE);
188  else _ptex_u.texParameterf(GL_TEXTURE_WRAP_S, GL_CLAMP_TO_EDGE);
189  if( closed_v ) _ptex_v.texParameterf(GL_TEXTURE_WRAP_T, GL_CLAMP_TO_EDGE);
190  else _ptex_v.texParameterf(GL_TEXTURE_WRAP_T, GL_CLAMP_TO_EDGE);
191 
192  // Create V line tex
193  _ptex_v.texImage2D( 0, GL_R8, 1, tex_data_v.getDim(), 0, GL_RED, GL_FLOAT, tex_data_v.getPtr() );
194  _ptex_v.texParameteri( GL_TEXTURE_MIN_FILTER, GL_LINEAR );
195  _ptex_v.texParameteri( GL_TEXTURE_MAG_FILTER, GL_LINEAR );
196  if( closed_u ) _ptex_u.texParameterf(GL_TEXTURE_WRAP_S, GL_CLAMP_TO_EDGE);
197  else _ptex_u.texParameterf(GL_TEXTURE_WRAP_S, GL_CLAMP_TO_EDGE);
198  if( closed_v ) _ptex_v.texParameterf(GL_TEXTURE_WRAP_T, GL_CLAMP_TO_EDGE);
199  else _ptex_v.texParameterf(GL_TEXTURE_WRAP_T, GL_CLAMP_TO_EDGE);
200 }

202 template <typename T, int n>
203 void PSurfParamLinesVisualizer<T,n>::initShaderProgram() {
204 
205  const std::string prog_name = "psurf_paramlines_prog";
206  if( _prog.acquire(prog_name) ) return;
207 
208 
209  std::string vs_src =
210  GL::OpenGLManager::glslDefHeaderVersionSource() +
211 
212  "uniform mat4 u_mvmat, u_mvpmat;\n"
213  "\n"
214  "in vec4 in_vertex;\n"
215  "in vec2 in_tex;\n"
216  "\n"
217  "out vec4 gl_Position;\n"
218  "\n"
219  "smooth out vec3 ex_pos;\n"
220  "smooth out vec2 ex_tex;\n"
221  "\n"
222  "void main() {\n"
223  "\n"
224  " vec4 v_pos = u_mvmat * in_vertex;\n"
225  " ex_pos = v_pos.xyz * v_pos.w;\n"
226  "\n"
227  " ex_tex = in_tex;\n"
228  "\n"
229  " gl_Position = u_mvpmat * in_vertex;\n"
230  "}\n"
231  ;
232 
233  std::string fs_src =
234  GL::OpenGLManager::glslDefHeaderVersionSource() +
235  GL::OpenGLManager::glslFnComputeBlinnPhongLightingSource() +
236 
237  "uniform sampler2D u_nmap;\n"
238  "uniform sampler2D u_ptex_u;\n"
239  "uniform sampler2D u_ptex_v;\n"
240  "uniform mat4 u_mvmat;\n"
241  "\n"
242  "uniform vec4 u_mat_amb;\n"
243  "uniform vec4 u_mat_dif;\n"
244  "uniform vec4 u_mat_spc;\n"
245  "uniform float u_mat_shi;\n"
246  "\n"
247  "uniform vec4 u_mat_line_amb;\n"
248  "uniform vec4 u_mat_line_dif;\n"
249  "uniform vec4 u_mat_line_spc;\n"
250  "uniform float u_mat_line_shi;\n"
251  "\n"
252  "smooth in vec3 ex_pos;\n"
253  "smooth in vec2 ex_tex;\n"
254  "\n"
255  "out vec4 gl_FragColor;\n"
256  "\n"
257  "void main() {\n"
258  "\n"
259  " mat3 nmat = inverse( transpose( mat3( u_mvmat ) ) );\n"
260  " vec3 nmap_normal = texture( u_nmap, ex_tex.st).xyz;\n"
261  " vec3 normal = normalize( nmat * nmap_normal );\n"
262  "\n"
263  " float pf_u = texture( u_ptex_u, ex_tex.st ).r;\n"
264  " float pf_v = texture( u_ptex_v, ex_tex.st ).r;\n"
265  "\n"
266  " Material mat;\n"
267  "\n"
268  " if( max(pf_u,pf_v) < 0.5 ) {\n"
269  " mat.ambient = u_mat_amb;\n"
270  " mat.diffuse = u_mat_dif;\n"
271  " mat.specular = u_mat_spc;\n"
272  " mat.shininess = u_mat_shi;\n"
273  " }\n"
274  " else {\n"
275  " mat.ambient = u_mat_line_amb;\n"
276  " mat.diffuse = u_mat_line_dif;\n"
277  " mat.specular = u_mat_line_spc;\n"
278  " mat.shininess = u_mat_line_shi;\n"
279  " }\n"
280  "\n"
281  " vec4 light_color = vec4(0.0);\n"
282  "\n"
283  " gl_FragColor = computeBlinnPhongLighting( mat, ex_pos, normal );\n"
284  "\n"
285  "}\n"
286  ;
287 
288  bool compile_ok, link_ok;
289 
290  GL::VertexShader vshader;
291  vshader.create("psurf_paramlines_vs");
292  vshader.setPersistent(true);
293  vshader.setSource(vs_src);
294  compile_ok = vshader.compile();
295  assert(compile_ok);
296 
297  GL::FragmentShader fshader;
298  fshader.create("psurf_paramlines_fs");
299  fshader.setPersistent(true);
300  fshader.setSource(fs_src);
301  compile_ok = fshader.compile();
302  assert(compile_ok);
303 
304  _prog.create(prog_name);
305  _prog.setPersistent(true);
306  _prog.attachShader(vshader);
307  _prog.attachShader(fshader);
308  link_ok = _prog.link();
309  assert(link_ok);
310 }
