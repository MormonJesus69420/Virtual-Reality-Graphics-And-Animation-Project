117  void PTriangleColorPointVisualizer<T>::replot(DVector<DMatrix<Vector<T, 3> > > &p, int m) {
118 
119  PTriangleVisualizer<T>::replot( p, m );
120 
121  // Allocate GPU memory
122  glBindBuffer( GL_ARRAY_BUFFER, _vbo_verts );
123  glBufferData( GL_ARRAY_BUFFER, PTRIANGLEVERTEX_SIZE * _points.getDim(), 0x0, GL_DYNAMIC_DRAW );
124  glBindBuffer( GL_ARRAY_BUFFER, _vbo_color );
125  glBufferData( GL_ARRAY_BUFFER, PTRIANGLEVERTEXATTRIB_SIZE * _points.getDim(), 0x0, GL_DYNAMIC_DRAW );
126 
127 
128  // Fill GPU memory
129 
130  glBindBuffer( GL_ARRAY_BUFFER, _vbo_verts );
131  PTriangleVertex *ptr_v = (PTriangleVertex*)glMapBuffer( GL_ARRAY_BUFFER, GL_WRITE_ONLY );
132 
133  glBindBuffer( GL_ARRAY_BUFFER, _vbo_color );
134  PTriangleVertexAttributes *ptr_c = (PTriangleVertexAttributes*)glMapBuffer( GL_ARRAY_BUFFER, GL_WRITE_ONLY );
135 
136 
137  if( ptr_v && ptr_c && this->_triangle ) {
138 
139  Color c1 = _c1;
140  Color c2 = _c2;
141  Color c3 = _c3;
142 
143  if( _color_mode == GM_PTRIANGLE_COLORPOINTVISUALIZER_HSV ) {
144 
145  c1.toHSV();
146  c2.toHSV();
147  c3.toHSV();
148  }
149 
150  DVector< Color > colors( p.getDim() );
151 
152 
153  for( int i = 0; i < _points.getDim(); i++ ) {
154 
155  Point<T,3> p = this->_triangle->evaluateLocal( _points[i][0], _points[i][1], 0, 0 )(0)(0);
156  Color c = _points[i][0] * c1 + _points[i][1] * c2 + _points[i][2] * c3;
157  if( _color_mode == GM_PTRIANGLE_COLORPOINTVISUALIZER_HSV )
158  c.toRGB();
159 
160  (*ptr_v).x = p[0];
161  (*ptr_v).y = p[1];
162  (*ptr_v).z = p[2];
163 
164  (*ptr_c).r = c.getRedC();
165  (*ptr_c).g = c.getGreenC();
166  (*ptr_c).b = c.getBlueC();
167  (*ptr_c).a = c.getAlphaC();
168 
169  ptr_v++;
170  ptr_c++;
171  }
172  }
173 
174  glBindBuffer( GL_ARRAY_BUFFER, _vbo_verts );
175  glUnmapBuffer( GL_ARRAY_BUFFER );
176  glBindBuffer( GL_ARRAY_BUFFER, _vbo_color );
177  glUnmapBuffer( GL_ARRAY_BUFFER );
178  glBindBuffer( GL_ARRAY_BUFFER, 0x0 );
179  }
