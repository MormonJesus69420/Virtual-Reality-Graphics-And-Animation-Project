205  template <typename T, int n>
206  void toNDfrom1D( T* indices, const T* dims, const T& idx_1d ) {
207 
208 
209  T rest = idx_1d;
210  for( int i = 0; i < n; i++ ) {
211 
212  // Product of dims
213  T pod = 1;
214  for( int j = n-1; j > i; j-- )
215  pod *= dims[j];
216 
217  // Compute indice of i
218  indices[i] = rest / pod;
219 
220  // Update 'rest'
221  rest -= indices[i] * pod;

299  template <typename T, int m, int res_fact_m, int n, int res_fact_n>
300  void sqMapping( const T* index_m, T* index_n, int& err ) {
301 
302  err = INDEX_NO_ERROR;
303 
304  if( (m * res_fact_m) != (n * res_fact_n) )
305  err |= INDEX_DIM_RES_MISSMATCH;
306 
307  const IndexND<T,m> &idx_m = *(const IndexND<T,m>*)index_m;
308 
309  const int res_m = std::pow( 2, res_fact_m );
310  for( int i = 0; i < m; i++ )
311  if( idx_m.s[i] >= res_m )
312  err |= INDEX_DIM_OUT_OF_BOUNDS;
313 
314  if( err != INDEX_NO_ERROR )
315  return;
316 
317  sqToND<T,n,res_fact_n>( sqTo1D<T,m,res_fact_m>( index_m ), index_n );
318  }

333  template <typename T, int n>
334  void sqWaweletTo1D( T& gamma, T& kappa, const T& j, const T& i, const T *k ) {
335 
336  const IndexND<T,n> &idx = *(const IndexND<T,n>*)k;
337 
338  const int log2_i = log2(i);
339 
340  gamma = j * n + log2_i;
341 
342  kappa = std::pow( 2, j * n ) * ( i - std::pow( 2, log2_i ) );
343  for( int l = 1; l <= n; l++ ) {
344 
345  kappa += idx.s[l-1] * std::pow( 2, j * ( n - l ) );
346  }

361  template <typename T, int n>
362  void sqWaweletToND( T& j, T& i, T* k, const T& gamma, const T& kappa ) {
363 
365  // Formula comments
366  //
367  // 'n' in the formulas is here
368  // denoted by 'nu',
369  // which again is assigned
370  // 'kappa_0'.
371 
372  // Transfer vars
373  IndexND<T,n> &idx = *(IndexND<T,n>*)k;
374

376  const int lbs = std::pow( 2, ( gamma / n ) * n ); // Level-block size
377  // std::cout << "lbs: " << lbs << std::endl;
378  const int kappa_0 = kappa - ( kappa / lbs ) * lbs;
379  // std::cout << "kappa_0: " << kappa_0 << std::endl;
380 
381  // Compute j
382  j = gamma / n;
383 
384  // Compute i
385  i = std::pow( 2, gamma % n ) + kappa / lbs;
386 
387 
388  // Compute the k_l's
389  int nu = kappa_0;
390 
391  const int mu = gamma / n;
392  for( int itr = n - 1; itr >= 0; itr-- ) {
393 
394  // std::cout << "nu = kappa_" << (n - itr - 1) << ": " << nu << std::endl;
395  T fact = std::pow( 2, mu * itr );
396  idx.s[n - itr - 1] = nu / fact;

416  template <typename T, int n>
417  void generateMappingTableFrom1DToND( T* mapping_table, const T* dims ) {
418 
419  // Compute
420  // N - the product of the resolution of all dimensions
421  // L - the sum of the resolution of all dimensions, minus the dimension
422  T N = 1;
423  T L = -n;
424  for( int i = 0; i < n; i++ ) {
425  L += dims[i];
426  N *= dims[i];
427  }
428 
429 
430  // List of lists of n-dim vectors
431  std::vector< std::vector< IndexND<T,n> > > S( L+T(1), std::vector< IndexND<T,n> >() );
432 
433  // Indice of (k1, ..., kn)
434  IndexND<T,n> k_indices;
435 
436  for( int i = 0; i < N; i++ ) {
437 
438  // Compute indices, the index of (k1, ..., kn)
439  Cartesian::toNDfrom1D<T,n>( (T*)&k_indices, dims, i );
440 
441  // Sum of the indices
442  T sum = getSumOfIndices<T,n>( (T*)&k_indices );
443 
444  // Append indice to the appropriate list
445  S[sum].push_back( k_indices );
446  }
447 
448  // Order the lists S into one one-dimensional list :)
449  IndexND<T,n>* mt = (IndexND<T,n>*)mapping_table;
450  // Iterate through the lists S
451  for( typename::std::vector< std::vector< IndexND<T,n> > >::iterator itr_i = S.begin(); itr_i != S.end(); itr_i++ )
452  // Iterate through the given list S_i
453  for( typename::std::vector< IndexND<T,n> >::iterator itr_j = (*itr_i).begin(); itr_j != (*itr_i).end(); itr_j++ )
454  *mt++ = *itr_j;
455  }
