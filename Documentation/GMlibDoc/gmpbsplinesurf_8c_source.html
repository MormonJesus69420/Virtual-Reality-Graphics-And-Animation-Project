40  inline
41  PBSplineSurf<T>::PBSplineSurf( const DMatrix< Vector<T,3> >& c, const DVector<T>& u, const DVector<T>& v, int du, int dv) {
42 
43  init();
44 
45  // Set Control Points
46  _c = c;
47 
48  // Set order
49  _ku = u.getDim()-_c.getDim1();
50  _kv = v.getDim()-_c.getDim2();
51 
52  // Set knot-vector in u-direction
53  if(_ku == 1) { // closed in u-direction
54  _du = du;
55  initKnot( _u, _cu, _ku, u, _c.getDim1(), _du);
56  }
57  else {
58  _u = u;
59  _du = _ku-1;
60  _cu = false;
61  }
62 
63  // Set knot-vector in v-direction
64  if(_kv == 1) { // closed in v-direction
65  if(dv==0) _dv = du;
66  else _dv = dv;
67  initKnot( _v, _cv, _kv, v, _c.getDim2(), _dv);
68  }
69  else {
70  _v = v;
71  _dv = _kv-1;
72  _cv = false;
73  }
74  }

135  void PBSplineSurf<T>::setClosed( bool closed_u, bool closed_v, T du, T dv ) {
136 
137  bool changed = false;
138 
139  if(_cu != closed_u) {
140  DVector<T> nu = _u;
141  if(closed_u) {
142  if(du==T(0)) du = (_u[_u.getDim()-_ku]-_u[_du])/(_u.getDim() - 2*_ku + 1);
143  initKnot( _u, _cu, _ku, nu, _c.getDim1(), _du, du);
144  } else
145  initKnot2(_u, _cu, nu, _c.getDim1(), _du);
146  changed = true;
147  }
148 
149  if(_cv != closed_v){
150  DVector<T> nv = _v;
151  if(closed_v) {
152  if(dv==T(0)) dv = (_v[_v.getDim()-_kv]-_v[_dv])/(_v.getDim() - 2*_kv + 1);
153  initKnot( _v, _cv, _kv, nv, _c.getDim2(), _dv, dv);
154  } else
155  initKnot2(_v, _cv, nv, _c.getDim2(), _dv);
156  changed = true;
157  }

160  hideSelectors();

491  template <typename T>
492  void PBSplineSurf<T>::preSample( int dir, int m ) {
493  if(_part_viz) {
494  if( dir==1 ) {
495  if(!_partitioned[0]) {
496  _partitioned[0] = true;
497  makePartition( _vpu, _u, _ku, _pcu, m );
498  }
499  _ru.setDim(_vpu.getDim());
500  for(int i=0; i<_ru.getDim(); i++)
501  preSample( _ru[i], _u, _vpu[i].m, _du, _c.getDim1(), _u[_vpu[i].is], _u[_vpu[i].ie] );
502  }
503  if( dir==2 ) {
504  if(!_partitioned[1]) {
505  _partitioned[1] = true;
506  makePartition( _vpv, _v, _kv, _pcv, m );
507  }
508  _rv.setDim(_vpv.getDim());
509  for(int i=0; i<_rv.getDim(); i++)
510  preSample( _rv[i], _v, _vpv[i].m, _dv, _c.getDim2(), _v[_vpv[i].is], _v[_vpv[i].ie] );
511  }
512  }
513  else {
514  if( dir==1 )
515  preSample( _ru[0], _u, m, _du, _c.getDim1(), _u[_du], _u[_u.getDim()-_ku] );
516  if( dir==2 )
517  preSample( _rv[0], _v, m, _dv, _c.getDim2(), _v[_dv], _v[_v.getDim()-_kv] );
518  }
519  }

569  template <typename T>
570  inline
571  void PBSplineSurf<T>::multEval(DMatrix<Vector<T,3>>& p, const DMatrix<T>& bu, const DMatrix<T>& bv, const std::vector<int>& ii, const std::vector<int>& ij, int du, int dv) const {
572 
573  // Set Dimensions
574  p.setDim(du+1,dv+1);
575 
576  // We manually do these two operations!
577  // bv.transpose();
578  // p = bu * (c^bv);
579  // The reason why we do it manually is that we only copmpute the du first lines of bu and the dv first lines of bv
580 
581  // c = _c^bvT
582  DMatrix<Vector<T,3>> c(_ku, dv+1);
583 
584  for(int i=0; i< _ku; i++)
585  for(int j=0; j<=dv; j++) {
586  c[i][j] = _c(ii[i])(ij[0])*bv(j)(0);
587  for(int k=1; k<_kv; k++)
588  c[i][j] += _c(ii[i])(ij[k])*bv(j)(k);
589  }
590  // p = bu * c
591  for(int i=0; i<=dv; i++)
592  for(int j=0; j<=du; j++) {
593  p[i][j] = bu(i)(0)*c[0][j];
594  for(int k=1; k<_ku; k++)
595  p[i][j] += bu(i)(k)*c[k][j];
596  }
597  }

602  inline
603  void PBSplineSurf<T>::initKnot( DVector<T>& t, bool& c, int& k, const DVector<T>& g, int n, int d, T dt) {
604 
605  k = d+1;
606  t.setDim(n+d+k);
607  if(dt > T(0)) {
608  for(int i = d; i < n+1; i++) t[i] = g(i);
609  for(int i = n+1; i < n+k; i++) t[i] = t[i-1] + dt;
610  } else
611  for(int i = d; i < n+k; i++) t[i] = g(i-d);
612  for(int i = d-1, j = n+d; i >= 0; i--,j--) t[i] = t[i+1] - (t[j]-t[j-1]);
613  for(int i = n+k, j = k; i < t.getDim(); i++,j++) t[i] = t[i-1] + (t[j]-t[j-1]);
614  c = true;
615  }

632  template <typename T>
633  inline
634  void PBSplineSurf<T>::preSample( DVector< PreMat<T> >& p, const DVector<T>& t, int m, int d, int n, T start, T end ) {
635 
636  const T dt = ( end - start ) / T(m-1); // dt is the step in parameter values
637  p.setDim(m); // p is a vector of Bernstein-Hermite matrises at the sample points
638 
639  // Compute the Bernstein-Hermite matrix
640  for( int j = 0; j < m-1; j++ ) {
641  int i = EvaluatorStatic<T>::evaluateBSp( p[j].m, start+j*dt, t, d, false );// - d;
642  p[j].ind.init( i, d+1, n);
643  }
644  int i = EvaluatorStatic<T>::evaluateBSp( p[m-1].m, end, t, d, true );// - d;
645  p[m-1].ind.init( i, d+1, n);
646  }
