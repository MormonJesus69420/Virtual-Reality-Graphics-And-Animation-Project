121  void Scene::clear() {
122 
123  const bool running = isRunning();
124  if(running)
125  stop();
126 
127  // Remove selections
128  clearSelection();
129 
130  // Remove/(delete) sun
131  removeSun();
132 
133  // Lights: if in scene remove, then delete
134  for( int i = 0; i < _lights.getSize(); ++i ) {
135  SceneObject *obj = dynamic_cast<SceneObject*>(_lights(i));
136  _scene.remove(obj);
137  }
138  _lights.clear();
139 
140  // Cameras: if in scene remove, then delete
141  for( int i = 0; i < _cameras.getSize(); ++i ) _scene.remove(_cameras(i));
142  _cameras.clear();
143 
144  // Clear rest of scene (remove and delete)
145  _scene.clear();
146 
147  if(running)
148  start();
149  }

349  void Scene::simulate() {
350 
351  if( !_timer_active ) return;
352 
353  if( GMutils::compValueF(_timer_time_elapsed,0.0) ) prepare();
354 
355  double dt, timer_dt;
356  timer_dt = _timer.getSec(true);
357  if( _timer_fixed_dt_enabled ) dt = _timer_fixed_dt;
358  else dt = _timer_time_scale * timer_dt;
359 
360  if(dt) {
361 
362  _timer_time_elapsed += dt;
363  if ( _event_manager ) _event_manager->processEvents(dt);
364  for( int i=0; i< _scene.getSize(); i++ )
365  _scene[i]->simulate(dt);
366  }
367  }
