68  void PCurvePointsVisualizer<T,n>::render(const SceneObject* obj, const DefaultRenderer* renderer) const {
69 
70  // GL States
71  glPointSize( _size);
72 
73  _prog.bind(); {
74  // Model view and projection matrices
75  _prog.uniform( "u_mvpmat", obj->getModelViewProjectionMatrix(renderer->getCamera()) );
76  _prog.uniform( "u_color", _color );
77 
78  // Vertex attribute location
79  GL::AttributeLocation vert_loc = _prog.getAttributeLocation( "in_vertex" );
80 
81  // Bind and draw
82  _vbo.bind();
83  _vbo.enable( vert_loc, 3, GL_FLOAT, GL_FALSE, sizeof(GL::GLVertex), reinterpret_cast<const GLvoid*>(0x0) );
84  glDrawArrays( GL_POINTS, 0, _no_vertices );
85  _vbo.disable( vert_loc );
86  _vbo.unbind();
87  } _prog.unbind();
88  }
