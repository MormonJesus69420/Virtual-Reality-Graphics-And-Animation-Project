76  // Build an orthogonal matrix
77  float near_plane = -1.0f;
78  float far_plane = 1.0f;
79 
80  float l, r, b, t, n, f;
81  l = 0.0f;
82  r = 1.0f;
83  b = 0.0f;
84  t = 1.0f;
85  n = near_plane;
86  f = far_plane;
87 
88  float A, B, C;
89  A = - ( r + l ) / ( r - l );
90  B = - ( t + b ) / ( t - b );
91  C = - ( f + n ) / ( f - n );
92 
93  _ortho_mat[0][0] = 2 / (r - l);
94  _ortho_mat[0][1] = 0.0f;
95  _ortho_mat[0][2] = 0.0f;
96  _ortho_mat[0][3] = A;
97 
98  _ortho_mat[1][0] = 0.0f;
99  _ortho_mat[1][1] = 2 / ( t - b );
100  _ortho_mat[1][2] = 0.0f;
101  _ortho_mat[1][3] = B;
102 
103  _ortho_mat[2][0] = 0.0f;
104  _ortho_mat[2][1] = 0.0f;
105  _ortho_mat[2][2] = - 2.0f / (f-n);
106  _ortho_mat[2][3] = C;
107 
108  _ortho_mat[3][0] = 0.0f;
109  _ortho_mat[3][1] = 0.0f;
110  _ortho_mat[3][2] = 0.0f;
111  _ortho_mat[3][3] = 1.0f;
