54 void PSurfVisualizer<T,n>::fillMap(GL::Texture& map, const DMatrix<DMatrix<Vector<T,n> > > &p, int d1, int d2, bool closed_u, bool closed_v) {
55 
56  int m1 = closed_u ? p.getDim1()-1 : p.getDim1();
57  int m2 = closed_v ? p.getDim2()-1 : p.getDim2();
58 
59  DVector< Vector<float,3> > tex_data(m1*m2);
60  Vector<float,3> *ptr = tex_data.getPtr();
61  for( int j = 0; j < m1; ++j ) {
62  for( int i = 0; i < m2; ++i ) {
63 
64  *ptr++ = p(i)(j)(d1)(d2);
65  }
66  }
67 
68  // Create Normal map texture and set texture parameters
69  map.texImage2D( 0, GL_RGB16F, m2, m1, 0, GL_RGB, GL_FLOAT, tex_data.getPtr()->getPtr() );
70  map.texParameteri( GL_TEXTURE_MIN_FILTER, GL_LINEAR );
71  map.texParameteri( GL_TEXTURE_MAG_FILTER, GL_LINEAR );
72  if( closed_v ) map.texParameterf(GL_TEXTURE_WRAP_S, GL_REPEAT);
73  else map.texParameterf(GL_TEXTURE_WRAP_S, GL_CLAMP_TO_EDGE);
74 
75  if( closed_u ) map.texParameterf(GL_TEXTURE_WRAP_T, GL_REPEAT);
76  else map.texParameterf(GL_TEXTURE_WRAP_T, GL_CLAMP_TO_EDGE);
77 }

141  const DMatrix<DMatrix<Vector<T,n> > > &p) {
142 
143  GLsizeiptr no_vertices = p.getDim1() * p.getDim2() * sizeof(GL::GLVertexTex2D);
144 
145  vbo.bufferData( no_vertices, 0x0, GL_STATIC_DRAW );
146  GL::GLVertexTex2D *ptr = vbo.mapBuffer<GL::GLVertexTex2D>();
147  for( int i = 0; i < p.getDim1(); i++ ) {
148  float s = i/float(p.getDim1()-1);
149  for( int j = 0; j < p.getDim2(); j++, ptr++ ) {
150  // vertex position
151  ptr->x = p(i)(j)(0)(0)(0);
152  ptr->y = p(i)(j)(0)(0)(1);
153  ptr->z = p(i)(j)(0)(0)(2);
154  // tex coords
155  ptr->s = s;
156  ptr->t = j/float(p.getDim2()-1);
157  }
158  }
159  vbo.unmapBuffer();

238  for( int k = 0; k < 3; k++ ) {
