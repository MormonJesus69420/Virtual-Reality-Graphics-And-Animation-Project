93 inline
94 void PTriangCurve<T>::eval1( T t, int d ) const
95 {
96  this->_p.setDim( d + 1 );
97 
98  Vector<T,3> dh = _p2 - _p1;
99  Vector<T,3> h = _p1 + t*dh;
100 
101  DVector<Vector<T,3> > m = _s->evaluateParent(h, d);
102 
103  this->_p[0] = m[0];
104  if(d>0)
105  {
106  this->_p[1] = dS(m[1], m[2], m[3], dh);
107 
108  if(d>1)
109  {
110  Vector<T,3> v1 = dS(m[4], m[5], m[6], dh);
111  Vector<T,3> v2 = dS(m[5], m[7], m[8], dh);
112  Vector<T,3> v3 = dS(m[6], m[8], m[9], dh);
113  this->_p[2] = dS(v1, v2, v3, dh);
114  }
115  }
116 }

120 template <typename T>
121 inline
122 void PTriangCurve<T>::eval2( T t, int d ) const
123 {
124  this->_p.setDim( d + 1 );
125  double tt = _basis(t);
126 
127  Vector<T,3> dh = _p2 - _p1;
128  Vector<T,3> h = _p1 + t*dh;
129  Vector<T,3> v = _v1 + tt*(_v2 - _v1);
130 
131  DVector<Vector<T,3> > m = _s->evaluateParent(h, d+1);
132 
133  this->_p[0] = dS(m[1], m[2], m[3], v);
134  if(d>0)
135  {
136  Vector<T,3> dv = _basis.getDer1()*(_v2 - _v1);
137  Vector<T,3> v1 = dS(m[4], m[5], m[6], dh);
138  Vector<T,3> v2 = dS(m[5], m[7], m[8], dh);
139  Vector<T,3> v3 = dS(m[6], m[8], m[9], dh);
140  this->_p[1] = dS(v1, v2, v3, v) + dS(m[1], m[2], m[3], dv);
141 
142  if(d>1)
143  {
144  Vector<T,3> d2v = _basis.getDer2()*(_v2 - _v1);
145  Vector<T,3> v0 = dS(v1, v2, v3, dv); // d(dS(dh))(dv)
146  Vector<T,3> v4,v5,v6;
147  // [S_uuu S_uuv S_uuw]
148  v1 = dS(m[10], m[11], m[12], dh);
149  // [S_uuv S_uvv S_uvw]
150  v2 = dS(m[11], m[13], m[14], dh);
151  // [S_uuw S_uvw S_uww]
152  v3 = dS(m[12], m[14], m[15], dh);
153  // [S_uvv S_vvv S_vvw]
154  v4 = dS(m[13], m[16], m[17], dh);
155  // [S_uvw S_vvw S_vww]
156  v5 = dS(m[14], m[17], m[18], dh);
157  // [S_uww S_vww S_www]
158  v6 = dS(m[15], m[18], m[19], dh);
159 
160  Vector<T,3> d1 = dS(v1, v2, v3, dh); // d(dS_u(dh))(v)
161  Vector<T,3> d2 = dS(v2, v4, v5, dh); // d(dS_v(dh))(v)
162  Vector<T,3> d3 = dS(v3, v5, v6, dh); // d(dS_w(dh))(v)
163  Vector<T,3> d4 = dS(d1, d2, d3, v) + 2*v0 + dS(m[1], m[2], m[3], d2v);
164 
165  this->_p[2] = d4;
166  }
167  }
168 }

172 template <typename T>
173 inline
174 void PTriangCurve<T>::eval123( T t)
175 {
176  this->_p.setDim(6);
177  double tt = _basis(t);
178  double dt = _basis.getDer1();
179  double d2t= _basis.getDer2();
180 
181  Vector<T,3> dh = _p2 - _p1;
182  Vector<T,3> h = _p1 + t*dh;
183  Vector<T,3> v = _v1 + tt*(_v2 - _v1);
184  Vector<T,3> dv = dt*(_v2 - _v1);
185  Vector<T,3> d2v = d2t*(_v2 - _v1);
186 
187  DVector<Vector<T,3> > m = _s->evaluateParent(h, 3);
188 
189  Vector<T,3> v1 = dS(m[4], m[5], m[6], dh); // uu, uv, uw
190  Vector<T,3> v2 = dS(m[5], m[7], m[8], dh); // uv, vv, vw
191  Vector<T,3> v3 = dS(m[6], m[8], m[9], dh); // uw, vw, ww
192  Vector<T,3> v0 = dS(v1, v2, v3, dv); // d(dS(dh))(dv)
193  Vector<T,3> v4,v5,v6;
194 
195  this->_p[0] = m[0];
196  this->_p[2] = dS(m[1], m[2], m[3], dh);
197  this->_p[4] = dS(v1, v2, v3, dh);
198 
199  this->_p[1] = dS(m[1], m[2], m[3], v);
200  this->_p[3] = dS(v1, v2, v3, v) + dS(m[1], m[2], m[3], dv);
201  // [S_uuu S_uuv S_uuw]
202  v1 = dS(m[10], m[11], m[12], dh);
203  // [S_uuv S_uvv S_uvw]
204  v2 = dS(m[11], m[13], m[14], dh);
205  // [S_uuw S_uvw S_uww]
206  v3 = dS(m[12], m[14], m[15], dh);
207  // [S_uvv S_vvv S_vvw]
208  v4 = dS(m[13], m[16], m[17], dh);
209  // [S_uvw S_vvw S_vww]
210  v5 = dS(m[14], m[17], m[18], dh);
211  // [S_uww S_vww S_www]
212  v6 = dS(m[15], m[18], m[19], dh);
213  Vector<T,3> d1 = dS(v1, v2, v3, dh); // d(dS_u(dh))(dh)
214  Vector<T,3> d2 = dS(v2, v4, v5, dh); // d(dS_v(dh))(dh)
215  Vector<T,3> d3 = dS(v3, v5, v6, dh); // d(dS_w(dh))(dh)
216 
217  this->_p[5] = dS(d1, d2, d3, v) + 2*v0 + dS(m[1], m[2], m[3], d2v);
218 }

281 template <typename T>
282 void PTriangCurve<T>::resample( DVector< DVector< Vector<T, 3> > >& p, int m, int d, T start, T end )
283 {
284  if (_der_curve && _plot)
285  {
286  T du = (end-start)/(m-1);
287  p.setDim(m);
288 
289  for( int i = 0; i < m - 1; i++ )
290  {
291  p[i].setDim(d+2);
292  eval1(start + i * du, 0);
293  p[i][0] = this->_p[0];
294  eval2(start + i * du, d);
295  for(int j=0; j<=d;j++)
296  p[i][j+1] = this->_p[j];
297  }
298  p[m-1].setDim(d+2);
299  eval1(end, 0);
300  p[m-1][0] = this->_p[0];
301  eval2(end, d);
302  for(int j=0; j<=d;j++)
303  p[m-1][j+1] = this->_p[j];
304 
305  switch( this->_dm )
306  {
307  case GM_DERIVATION_EXPLICIT:
308  // Do nothing, evaluator algorithms for explicite calculation of derivatives
309  // should be defined in the eval( ... ) function enclosed by
310  // if( this->_derivation_method == this->EXPLICIT ) { ... eval algorithms for derivatives ... }
311  break;
312 
313  case GM_DERIVATION_DD:
314  default:
315  // _evalDerDD( p, d, du );
316  break;
317  };
318  }
319  else
320  PCurve<T,3>::resample( p, m, d, start, end );
321 }
