105  void Program::updateLinkerLog() {
106 
107  InfoIter itr = getInfoIter();
108  itr->linker_log.clear();
109 
111 
112  int len = 0;
113  int written = 0;
114  char *log;
115 
116  GL_CHECK(::glGetProgramiv( getId(), GL_INFO_LOG_LENGTH, &len ));
117 
118  if( len > 0 ) {
119 
120  len = len+1;
121  log = new char[len];
122 
123  GL_CHECK(::glGetProgramInfoLog( getId(), len, &written, log ));
124 
125  itr->linker_log.append( log, len );
126  delete log;
127  }
128  }

130  std::vector<GLuint> Program::getAttachedShaders() const {
131 
132  if(!isValid())
133  return std::vector<GLuint>();

135  GLint no_as;

137 
138  std::vector<GLuint> shader_ids(no_as);
139  GL_CHECK(::glGetAttachedShaders( getId(), no_as, 0x0, &shader_ids[0] ) );
