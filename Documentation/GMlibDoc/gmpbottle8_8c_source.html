78  template <typename T>
79  void PBottle8<T>::eval(T u, T v, int d1, int d2, bool /*lu*/, bool /*lv*/ ) const {
80 
81  this->_p.setDim( d1+1, d2+1 );
82 
83  T cu = cos(u);
84  T su = sin(u);
85  T cv = cos(v);
86  T sv = sin(v);
87  T c2v = cos(T(2)*v);
88  T s2v = sin(T(2)*v);
89  T c5u = cos(T(.5)*u);
90  T s5u = sin(T(.5)*u);
91  T c5 = c5u*s2v;
92  T s5 = s5u*s2v;
93 
94  this->_p[0][0][0] = (_r+c5u*sv-s5)*cu;
95  this->_p[0][0][1] = (_r+c5u*sv-s5)*su;
96  this->_p[0][0][2] = s5u*sv+c5;
97 
98  if( this->_dm == GM_DERIVATION_EXPLICIT ) {
99 
100  if(d1) { //u
101  this->_p[1][0][0] = -T(.5)*cu*s5u*sv - T(.5)*cu*c5 - su*_r - su*c5u*sv + su*s5;
102  this->_p[1][0][1] = -T(.5)*su*s5u*sv - T(.5)*su*c5 + cu*_r + cu*c5u*sv - cu*s5;
103  this->_p[1][0][2] = -T(.5)*(-c5u*sv+s5);
104  }
105  if(d1>1) { //uu
106  this->_p[2][0][0] = -T(.25)*cu*c5u*sv + T(.25)*cu*s5 + su*s5u*sv + su*c5 - cu*_r - cu*c5u*sv + cu*s5;
107  this->_p[2][0][1] = -T(.25)*su*c5u*sv + T(.25)*su*s5 - cu*s5u*sv - cu*c5 - su*_r - su*c5u*sv + su*s5;
108  this->_p[2][0][2] = -T(.25)*(s5u*sv+c5);
109  }
110  if(d2) { //v
111  this->_p[0][1][0] = (c5u*cv - s5u*c2v*T(2))*cu;
112  this->_p[0][1][1] = (c5u*cv - s5u*c2v*T(2))*su;
113  this->_p[0][1][2] = s5u*cv + T(2)*c5u*c2v;
114  }
115  if(d2>1) { //vv
116  this->_p[0][2][0] = (-c5u*sv + T(4))*s5*cu;
117  this->_p[0][2][1] = (-c5u*sv + T(4))*s5*su;
118  this->_p[0][2][2] = -s5u*sv - T(4)*c5;
119  }
120  if(d1 && d2) { //uv
121  this->_p[1][1][0] = -T(.5)*cu*s5u*cv - cu*c5u*c2v - su*c5u*cv + T(2)*su*s5u*c2v;
122  this->_p[1][1][1] = -T(.5)*su*s5u*cv - su*c5u*c2v + cu*c5u*cv - T(2)*cu*s5u*c2v;
123  this->_p[1][1][2] = -T(.5)*(-c5u*cv + T(2)*s5u*c2v);
124  }
125  if(d1>1 && d2) { //uuv
126  this->_p[2][1][0] = -T(.25)*cu*c5u*cv + T(.5)*cu*s5u*c2v + su*s5u*cv + T(2)*su*c5u*c2v - cu*c5u*cv + T(2)*cu*s5u*c2v;
127  this->_p[2][1][1] = -T(.25)*su*c5u*cv + T(.5)*su*s5u*c2v - cu*s5u*cv - T(2)*cu*c5u*c2v - su*c5u*cv + T(2)*su*s5u*c2v;
128  this->_p[2][1][2] = -T(.25)*(s5u*cv + T(2)*c5u*c2v);
129  }
130  if(d1 && d2>1) { //uvv
131  this->_p[1][2][0] = cu*s5u*T(.5)*sv + cu*c5*T(2) + su*c5u*sv - su*s5*T(4);
132  this->_p[1][2][1] = su*s5u*T(.5)*sv + su*c5*T(2) - cu*c5u*sv + cu*s5*T(4);
133  this->_p[1][2][2] = T(.5)*(-c5u*sv + T(4)*s5);
134  }
135  if(d1>1 && d2>1) { //uuvv
136  this->_p[2][2][0] = T(.25)*cu*c5u*sv - cu*s5 - su*s5u*sv - T(4)*su*c5 + cu*c5u*sv - T(4)*cu*s5;
137  this->_p[2][2][1] = T(.25)*su*c5u*sv - su*s5 + cu*s5u*sv + T(4)*cu*c5 + su*c5u*sv - T(4)*su*s5;
138  this->_p[2][2][2] = T(.25)*(s5u*sv + T(4)*c5);
139  }
140  }
