80  template <typename T>
81  void PBohemianDome<T>::eval(T u, T v, int d1, int d2, bool /*lu*/, bool /*lv*/ ) const {
82 
83  this->_p.setDim( d1+1, d2+1 );
84 
85  T cu = cos(u);
86  T cv = cos(v);
87  T su = sin(u);
88  T sv = sin(v);
89 
90  this->_p[0][0][0] = _r*cu;
91  this->_p[0][0][1] = _r*su+_w*cv;
92  this->_p[0][0][2] = _h*sv;
93 
94 
95  if( this->_dm == GM_DERIVATION_EXPLICIT ) {
96 
97  if(d1) { //u
98  this->_p[1][0][0] = -_r*su;
99  this->_p[1][0][1] = _r*cu;
100  this->_p[1][0][2] = T(0);
101  }
102  if(d1>1) { //uu
103  this->_p[2][0][0] = -_r*cu;
104  this->_p[2][0][1] = -_r*su;
105  this->_p[2][0][2] = T(0);
106  }
107  if(d2) { //v
108  this->_p[0][1][0] = T(0);
109  this->_p[0][1][1] = -_w*sv;
110  this->_p[0][1][2] = _h*cv;
111  }
112  if(d2>1) { //vv
113  this->_p[0][2][0] = T(0);
114  this->_p[0][2][1] = -_w*cv;
115  this->_p[0][2][2] = -_h*sv;
116  }
117  if(d1 && d2) { //uv
118  this->_p[1][1][0] = T(0);
119  this->_p[1][1][1] = T(0);
120  this->_p[1][1][2] = T(0);
121  }
122  if(d1>1 && d2) { //uuv
123  this->_p[2][1][0] = T(0);
124  this->_p[2][1][1] = T(0);
125  this->_p[2][1][2] = T(0);
126  }
127  if(d1 && d2>1) { //uvv
128  this->_p[1][2][0] = T(0);
