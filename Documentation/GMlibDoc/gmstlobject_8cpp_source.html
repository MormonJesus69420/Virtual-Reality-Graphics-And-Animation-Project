51  StlObject::StlObject( PSurf<float,3> *obj, int m1, int m2, GM_STL_VISUALIZATION gsv ) {
52 
53  // Resample DSurf
54  DMatrix< DMatrix< Vector<float, 3> > > p;
55  obj->resample( p, m1, m2, 1, 1,
56  obj->getParStartU(),
57  obj->getParStartV(),
58  obj->getParEndU(),
59  obj->getParEndV() );
60 
61  // Set Array Max Size (speedup)
62  _normals.setMaxSize( (p.getDim1()-1)*(p.getDim2()-1)*2 );
63  _vertices.setMaxSize( _normals.getMaxSize()*3 );
64 
65  for( int i = 0; i < p.getDim1() - 1; i++ ) {
66  for( int j = 1; j < p.getDim2(); j++ ) {
67 
68  Vector<float,3> &v1 = p[i][j-1][0][0];
69  Vector<float,3> &v2 = p[i+1][j-1][0][0];
70  Vector<float,3> &v3 = p[i][j][0][0];
71  Vector<float,3> &v4 = p[i+1][j-1][0][0];
72  Vector<float,3> &v5 = p[i+1][j][0][0];
73  Vector<float,3> &v6 = p[i][j][0][0];
74 
75  Vector<float,3> n1 = Vector<float,3>(v2 - v1) ^ (v3 - v1);
76  Vector<float,3> n2 = Vector<float,3>(v5 - v4) ^ (v6 - v4);
77 
78  _normals.insertAlways( n1 );
79  _vertices.insertAlways( v1 );
80  _vertices.insertAlways( v2 );
81  _vertices.insertAlways( v3 );
82 
83  _normals.insertAlways( n2 );
84  _vertices.insertAlways( v4 );
85  _vertices.insertAlways( v5 );
86  _vertices.insertAlways( v6 );
87  }
88  }
89 
90  _init();
91  replot( gsv );
92  }

221  int StlObject::_readStlBinary(const std::string& filename) {
222 
223  if( ( _stl_file = fopen( filename.c_str(), "rb" ) ) == NULL )
224  return(-1);
225 
226  unsigned long int t;
227 
228  // Skipping comment line
229  for( t=0; t<80 ; t++ )
230  fgetc( _stl_file );
231 
232  // Get number for triangles
233  unsigned long int n_triangles = _getUli();
234  float x,y,z;
235  _normals.setMaxSize( n_triangles+1 );
236  _vertices.setMaxSize( (n_triangles*3)+1 );
237 
238  // Read all the triangles
239  for( t=0; t<n_triangles; t++ ) {
240 
241  // and put in array
242  x=_getFloat(); y=_getFloat(); z=_getFloat();
243  _normals.insertAlways( UnitVector<float,3>(x,y,z) );
244 
245 
246  x=_getFloat(); y=_getFloat(); z=_getFloat();
247  _vertices.insertAlways( Point<float,3> (x,y,z) );
248 
249  x=_getFloat(); y=_getFloat(); z=_getFloat();
250  _vertices.insertAlways( Point<float,3> (x,y,z) );
251 
252  x=_getFloat(); y=_getFloat(); z=_getFloat();
253  _vertices.insertAlways( Point<float,3> (x,y,z) );
254 
255  // Get number of attribute bytes (is always zero and can be ignored)
256  _getUint();
257  }
258 
259  fclose( _stl_file );
260  return 1;
261  }

296  void StlObject::load( std::ifstream& stream, bool binary ) {
297 
298  // Binary
299  if( binary ) {
300 
301  // Read header
302  char hbuff[80];
303  stream.read( hbuff, 80 );
304 
305  _header = hbuff;
306 
307  unsigned int facets;
308  stream.read( reinterpret_cast<char*>(&facets), sizeof( unsigned int ) );
309 
310 
311  // Allocate memory
312  _normals.setSize(facets);
313  _vertices.setSize(_normals.getMaxSize()*3);
314 
315  // Read Normal and Vertices and Attribute bit of each face
316  for( unsigned int i = 0; i < facets; i++ ) {
317 
318  // Normal
319  stream.read( reinterpret_cast<char*>(_normals[i].getPtr()), 3 * sizeof( unsigned int ) );
320 
321  // Vertices
322  stream.read( reinterpret_cast<char*>(_vertices[i*3].getPtr()), 9 * sizeof( unsigned int ) );
323 
324  // Attribute
325  unsigned short attrib = 0;
326  stream.read( reinterpret_cast<char*>(&attrib), sizeof( unsigned short ) );
327  }
328  }
