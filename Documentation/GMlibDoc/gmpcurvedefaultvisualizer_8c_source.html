62  inline
63  void PCurveDefaultVisualizer<T,n>::render(const SceneObject* obj, const DefaultRenderer* renderer) const {
64 
65  // GL States
66  ::glLineWidth( _line_width );
67 
68  _prog.bind(); {
69  // Model view and projection matrices
70  _prog.uniform( "u_mvpmat", obj->getModelViewProjectionMatrix(renderer->getCamera()) );
71  _prog.uniform( "u_color", obj->getColor() );
72 
73  // Vertex attribute location
74  GL::AttributeLocation vertice_loc = _prog.getAttributeLocation( "in_vertex" );
75 
76  // Bind and draw
77  _vbo.bind();
78  _vbo.enable( vertice_loc, 3, GL_FLOAT, GL_FALSE, sizeof(GL::GLVertex), reinterpret_cast<const GLvoid*>(0x0) );
79  glDrawArrays( GL_LINE_STRIP, 0, _no_vertices );
80  _vbo.disable( vertice_loc );
81  _vbo.unbind();
82  } _prog.unbind();
83  }
