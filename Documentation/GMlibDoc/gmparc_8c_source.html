69  inline
70  PArc<T>::PArc( DVector< Vector<T,3> >& p, T s, T t, T e ) : PCurve<T,3>(20, 0, 3) {
71  Vector<T,3> x, y, z;
72 
73  _d = p[1].getLength();
74  _start = (s-t);
75  _end = (e-t);
76  _x = (t-s)/(e-s);
77 
78  x = p[1]/_d;
79  y = ( p[2] - ( x * p[2] ) * x ) / (_d * _d);
80  _k = y.getLength();
81 
82  if( _k > 1e-5 ) {
83  y /= _k;
84  z = y^x;
85  }
86  else {
87  y = x.getLinIndVec();
88  y.normalize();
89  z = y^x;
90  }
91 
92  // Do basis change
93  this->_side = y.template toType<float>();
94  this->_up = z.template toType<float>();
95  this->_dir = x.template toType<float>();
96  this->_pos = p[0];
97  this->basisChange( this->_side, this->_up, this->_dir, this->_pos );
98  }

114  inline
115  PArc<T>::PArc( const Vector<T,3>& p1, const Vector<T,3>& p2, const Vector<T,3>& p3, int n ) : PCurve<T,3>(20, 0, 2) {
116 
117  Vector<T,3> x, y, z;
118  Vector<T,3> p = n==1 ? p1 : (n==2 ? p2 : p3);
119  Vector<T,3> q1 = (p1+p2)/2.0;
120  Vector<T,3> q2 = (p3+p2)/2.0;
121  Vector<T,3> v1 = p2-p1;
122  Vector<T,3> v2 = p3-p2;
123 
124  T v11 = v1*v1;
125  T v12 = v1*v2;
126  T v22 = v2*v2;
127  T a = v12/v11;
128  T d = v22-a*v12;
129  T d1, d2;
130 
131  if(d<1.0e-5){ // special case, a stright line
132  _k = 0;
133  _d = 1;//2*q.getLength();
134  d1 = sqrt(v11);
135  d2 = sqrt(v22);
136  x = UnitVector<T,3>(v1);
137  y = v1.getLinIndVec();
138  z = UnitVector<T,3>(y^x);
139  y = x^z;
140  }
141  else {
142  T xx = ((q2-q1)*v2)/d;
143  Vector<T,3> p0 = q1+xx*(v2-a*v1);
144  Vector<T,3> a1 = p0-p1;
145  Vector<T,3> a2 = p0-p2;
146  Vector<T,3> a3 = p0-p3;
147  T r = a1.getLength();
148  _k = 1/r;
149  _d = 1;
150  d1 = r*acos((a1*a2)/(r*r));
151  d2 = r*acos((a2*a3)/(r*r));
152  y = _k*(p0 - p);
153  z = UnitVector<T,3>(n==1 ? y^v1 : y^v2);
154  x = z^y;
155  }
156  if(n==1){
157  _start = 0;
158  _end = d1+d2;
159  } else if(n==2) {
160  _start = -d1;
161  _end = d2;
162  } else if(n==3) {
163  _start = -d1-d2;
164  _end = 0;
165  }
166  _x = d1/(d1+d2);
167 
168  // Do basis change
169  this->_side = y.template toType<float>();
170  this->_up = z.template toType<float>();
171  this->_dir = x.template toType<float>();
172  this->_pos = p;
173  this->basisChange( this->_side, this->_up, this->_dir, this->_pos );
174  }

361  template <typename T>
362  void PArc<T>::eval( T t, int d, bool /*l*/ ) const {
363 
364  this->_p.setDim( d + 1 );
365 
366  if( _k < 1e-5 ) { // special case, a stright line
367  this->_p[0][0] = _d * t;
368  this->_p[0][1] = T(0);
369  this->_p[0][2] = T(0);
370  if( this->_dm == GM_DERIVATION_EXPLICIT ) {
371  if( d > 0 ) {
372  this->_p[1][0] = _d;
373  this->_p[1][1] = T(0);
374  this->_p[1][2] = T(0);
375  if( d > 1 ) {
376  this->_p[2][0] = T(0);
377  this->_p[2][1] = T(0);
378  this->_p[2][2] = T(0);
379  if( d > 2 ) {
380  this->_p[3][0] = T(0);
381  this->_p[3][1] = T(0);
382  this->_p[3][2] = T(0);
383  }
384  }
385  }
386  }
387  }
388  else { // a circle - the curvature _k is bigger than the tolerance
389  T kdt = _k * _d * t;
390  T skdt = sin(kdt)/_k;
391  T ckdt = cos(kdt)/_k;
392  this->_p[0][0] = skdt;
393  this->_p[0][1] = 1/_k - ckdt;
394  this->_p[0][2] = T(0);
395  if( this->_dm == GM_DERIVATION_EXPLICIT ) {
396  T g = _k * _d;
397  if( d > 0 ) {
398  this->_p[1][0] = g * ckdt;
399  this->_p[1][1] = g * skdt;
400  this->_p[1][2] = T(0);
401  if( d > 1 ) {
402  g *= g;
403  this->_p[2][0] = -g * skdt;
404  this->_p[2][1] = g * ckdt;
405  this->_p[2][2] = T(0);
406  if( d > 2 ) {
407  g *= g;
408  this->_p[3][0] = -g * ckdt;
409  this->_p[3][1] = -g * skdt;
410  this->_p[3][2] = T(0);
411  }
412  }
413  }
414  }
415  }
