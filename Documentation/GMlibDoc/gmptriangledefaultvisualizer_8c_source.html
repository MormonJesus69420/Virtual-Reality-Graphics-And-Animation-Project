48  template <typename T, int n>
49  inline
50  void PTriangleDefaultVisualizer<T,n>::render(const SceneObject *obj, const DefaultRenderer* renderer) const {
51 
52  const Camera* cam = renderer->getCamera();
53 
54  const HqMatrix<float,3> &mvmat = obj->getModelViewMatrix(cam);
55  const HqMatrix<float,3> &pmat = obj->getProjectionMatrix(cam);
56 
57  this->glSetDisplayMode();
58 
59  _prog.bind(); {
60 
61  // Model view and projection matrices
62  _prog.uniform( "u_mvmat", mvmat );
63  _prog.uniform( "u_mvpmat", pmat * mvmat );
64 
65  // Lights
66  _prog.bindBufferBase( "DirectionalLights", renderer->getDirectionalLightUBO(), 0 );
67  _prog.bindBufferBase( "PointLights", renderer->getPointLightUBO(), 1 );
68  _prog.bindBufferBase( "SpotLights", renderer->getSpotLightUBO(), 2 );
69 
70  // Get Material Data
71  const Material &m = obj->getMaterial();
72  _prog.uniform( "u_mat_amb", m.getAmb() );
73  _prog.uniform( "u_mat_dif", m.getDif() );
74  _prog.uniform( "u_mat_spc", m.getSpc() );
75  _prog.uniform( "u_mat_shi", m.getShininess() );
76 
77  GL::AttributeLocation vert_loc = _prog.getAttributeLocation( "in_vertex" );
78  GL::AttributeLocation normal_loc = _prog.getAttributeLocation( "in_normal" );
79 
80  _vbo.bind();
81  _vbo.enable( vert_loc, 3, GL_FLOAT, GL_FALSE, sizeof(GL::GLVertexNormal), reinterpret_cast<const GLvoid*>(0x0) );
82  _vbo.enable( normal_loc, 3, GL_FLOAT, GL_TRUE, sizeof(GL::GLVertexNormal), reinterpret_cast<const GLvoid*>(sizeof(GL::GLVertex) ) );
83 
84  draw();
85 
86  _vbo.disable( normal_loc );
87  _vbo.disable( vert_loc );
88  _vbo.unbind();
89 
90  } _prog.unbind();
91  }
