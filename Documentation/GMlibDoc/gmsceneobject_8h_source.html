611  void SceneObject::basisChange( const Vector<float,3>& x, const Vector<float,3>& y, const Vector<float,3>& z, const Vector<float,3>& p ) {
612 
613  static Vector<float,4> nx, ny, nz, np;
614  memcpy( nx.getPtr(), z.getPtr(), 12 );
615  memcpy( ny.getPtr(), x.getPtr(), 12 );
616  memcpy( nz.getPtr(), y.getPtr(), 12 );
617  memcpy( np.getPtr(), p.getPtr(), 12 );
618  nx[3] = ny[3] = nz[3] = 0.0f;
619  np[3] = 1.0f;
620 
621  _matrix.setCol( nx, 0 );
622  _matrix.setCol( ny, 1 );
623  _matrix.setCol( nz, 2 );
624  _matrix.setCol( np, 3 );
625  }
