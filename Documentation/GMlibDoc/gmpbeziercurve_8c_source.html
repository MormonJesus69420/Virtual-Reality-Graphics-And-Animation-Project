69  inline
70  PBezierCurve<T>::PBezierCurve( const DVector< Vector<T, 3> >& g, T s, T t, T e ):PCurve<T,3>(0,0,10) {
71 
72  init();
73  this->setDomain(s, e);
74 
75  // Generate the control points
76  DMatrix<T> bhp;
77  EvaluatorStatic<T>::evaluateBhp( bhp, g.getDim()-1, this->_map(t), T(1)/(e-s) );
78  bhp.invert();
79  _c = bhp * g;
80 
81  // Set origin in local coordinate system to c(t) = g(0)
82  for( int i = 0; i < g.getDim(); i++ )
83  _c[i] -= g[0];
84  this->translateParent( g[0] );
85 
86 // std::cout << "CP:" << _c << std::endl;
87 // std::cout << "g:" << g << std::endl;
88  }

469  void PBezierCurve<T>::updateSamples() const {
470 
471  while(_pos_change.size()>0) {
472  EditSet es = _pos_change.back();
473  for(int i=_pos_change.size()-2; i>=0; i--)
474  if (_pos_change[i].ind == es.ind) {

517  inline
518  void PBezierCurve<T>::multEval(DVector<Vector<T,3>>& p, const DMatrix<T>& B, int d) const {
519 
520  p.setDim(d+1);
521  // p = B * c;
522  // We do this manually because we only want to compute the d first rows of B.
523  for(int i=0; i<=d; i++) {
524  p[i] = B(i)(0)*_c[0];
525  for(int k=1; k<_c.getDim(); k++)
526  p[i] += B(i)(k)*_c[k];
527  }
528  }
