68 template <typename T>
69 inline
70 void PSubTriangle<T>::eval( T u, T v, T w, int d ) const {
71 
72  if(d==0) this->_p.setDim(1);
73  else if(d==1) this->_p.setDim(4);
74  else if(d==2) this->_p.setDim(10);
75  else this->_p.setDim(20); // d==3
76 
77  Point<T,2> h = u * _q[0] + v * _q[1] + w * _q[2];
78  // Note that that the derivatives: h_u = _q[0], h_v = _q[1], h_w = _q[2]
79  // and that all second derivatives, for example h_uu = 0 etc.
80 
81  DMatrix<Vector<T,3> > p = _s->evaluateParent(h,d);
82 
83  this->_p[0] = p[0][0];
84 
85  if(d>0)
86  { // S_u S_v
87  this->_p[1] = dS(p[1][0], p[0][1], _q[0]); // dS(h_u)
88  this->_p[2] = dS(p[1][0], p[0][1], _q[1]); // dS(h_v)
89  this->_p[3] = dS(p[1][0], p[0][1], _q[2]); // dS(h_w)
90 
91  if (d>1)
92  {
93  Vector<T,3> s1 = dS(p[2][0], p[1][1], _q[0]);
94  Vector<T,3> s2 = dS(p[1][1], p[0][2], _q[0]);
95  this->_p[4] = dS(s1, s2, _q[0]); // [dS_u(h_u),dS_v(h_u)](h_u) : S_uu
96  this->_p[5] = dS(s1, s2, _q[1]); // [dS_u(h_u),dS_v(h_u)](h_v) : S_uv
97  this->_p[6] = dS(s1, s2, _q[2]); // [dS_u(h_u),dS_v(h_u)](h_w) : S_uw
98 
99  s1 = dS(p[2][0], p[1][1], _q[1]);
100  s2 = dS(p[1][1], p[0][2], _q[1]);
101  this->_p[7] = dS(s1, s2, _q[1]); // [dS_u(h_v),dS_v(h_v)](h_v) : S_vv
102  this->_p[8] = dS(s1, s2, _q[2]); // [dS_u(h_v),dS_v(h_v)](h_w) : S_vw
103  this->_p[9] = dS2(p[2][0], p[1][1], p[0][2], _q[2]);// [dS_u(h_w),dS_v(h_w)](h_w) : S_ww
104  }
105  if (d>2)
106  {
107  Vector<T,3> s1 = dS(p[3][0], p[2][1], _q[0]); // u
108  Vector<T,3> s2 = dS(p[2][1], p[1][2], _q[0]);
109  Vector<T,3> s3 = dS(p[1][2], p[0][3], _q[0]);
110  Vector<T,3> s4 = dS(s1, s2, _q[0]); // u
111  Vector<T,3> s5 = dS(s2, s3, _q[0]);
112  this->_p[10] = dS(s4, s5, _q[0]); // u : S_uuu
113  this->_p[11] = dS(s4, s5, _q[1]); // v : S_uuv
114  this->_p[12] = dS(s4, s5, _q[2]); // w : S_uuw
115 
116  Vector<T,3> s6 = dS(s1, s2, _q[1]); // v
117  Vector<T,3> s7 = dS(s2, s3, _q[1]);
118  this->_p[13] = dS(s6, s7, _q[1]); // v : S_uvv
119  this->_p[14] = dS(s6, s7, _q[2]); // w : S_uvw
120  this->_p[15] = dS2(s1, s2, s3, _q[2]); // w : S_uww
121 
122  Vector<T,3> s10 = dS(p[3][0], p[2][1], _q[1]);// v
123  Vector<T,3> s11 = dS(p[2][1], p[1][2], _q[1]);
124  Vector<T,3> s12 = dS(p[1][2], p[0][3], _q[1]);
125 
126  Vector<T,3> s13= dS(s10, s11, _q[1]); // v
127  Vector<T,3> s14= dS(s11, s12, _q[1]);
128  this->_p[16] = dS(s13, s14, _q[1]); // v : S_vvv
129  this->_p[17] = dS(s13, s14, _q[2]); // w : S_vvw
130  this->_p[18] = dS2(s10, s11, s12, _q[2]); // w : S_vww
131 
132  Vector<T,3> s17 = dS(p[3][0], p[2][1], _q[2]);// w
133  Vector<T,3> s18 = dS(p[2][1], p[1][2], _q[2]);
134  Vector<T,3> s19 = dS(p[1][2], p[0][3], _q[2]);
135  this->_p[19] = dS2(s17, s18, s19, _q[2]); // w : S_www
136  }
137  }
