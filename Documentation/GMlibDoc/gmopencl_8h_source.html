207  template <class T, OpenCL::EventInfo::TYPE T_type>
208  T OpenCL::getEvent( const std::string& name ) const {
209 
210  GMutils::DerivedFrom<T,cl::Event>();
211 
212  static T neutral_event;
213 
214  EventMap::const_iterator itr = _events.find(name);
215  if( itr == _events.end() )
216  return neutral_event;
217 
218  if( (*itr).second.type != T_type )
219  return neutral_event;
220 
221  return reinterpret_cast<const T&>((*itr).second.event); // s@#%^*!
222  }
