80  inline
81  void PBezierTriangle<T>::eval( T u, T v, T w, int d ) const {
82 
83  this->_p.setDim(4);
84 
85  if( _c.getDim() == 3 ) {
86  this->_p[0] = _c[0]*u + _c[1]*v + _c[2]*w;
87  this->_p[1] = _c[0];
88  this->_p[2] = _c[1];
89  this->_p[3] = _c[2];
90  }
91  else if( _c.getDim() == 6 ) {
92  this->_p[0] = _c[0]*(u*u) + _c[1]*(2*u*v) + _c[2]*(2*u*w)
93  + _c[3]*(v*v) + _c[4]*(2*v*w) + _c[5]*(w*w);
94  this->_p[1] = _c[0]*(2*u) + _c[1]*(2*v) + _c[2]*(2*w);
95  this->_p[2] = _c[1]*(2*u) + _c[3]*(2*v) + _c[4]*(2*w);
96  this->_p[3] = _c[2]*(2*u) + _c[4]*(2*v) + _c[5]*(2*w);
97  }
98  else if( _c.getDim() == 10 ) {
99  this->_p[0] = _c[0]*(u*u*u) + _c[1]*(3*u*u*v) + _c[2]*(3*u*u*w) +
100  _c[3]*(3*u*v*v) + _c[4]*(6*u*v*w) + _c[5]*(3*u*w*w) +
101  _c[6]*(v*v*v) + _c[7]*(3*v*v*w) + _c[8]*(3*v*w*w) + _c[9]*(w*w*w);
102  this->_p[1] = _c[0]*(3*u*u) + _c[1]*(6*u*v) + _c[2]*(6*u*w) + _c[3]*(3*v*v) + _c[4]*(6*v*w) + _c[5]*(3*w*w);
103  this->_p[2] = _c[1]*(3*u*u) + _c[3]*(6*u*v) + _c[4]*(6*u*w) + _c[6]*(3*v*v) + _c[7]*(6*v*w) + _c[8]*(3*w*w);
104  this->_p[3] = _c[2]*(3*u*u) + _c[4]*(6*u*v) + _c[5]*(6*u*w) + _c[7]*(3*v*v) + _c[8]*(6*v*w) + _c[9]*(3*w*w);
105  }
