194  template <typename T>
195  void PLogSpiral<T>::eval( T t, int d, bool /*l*/ ) const {
196 
197  this->_p.setDim( d + 1 );
198 
199  const T ct = cos(t);
200  const T st = sin(t);
201  const T ex = _a*exp(_b*t);
202 
203  this->_p[0][0] = ex*ct;
204  this->_p[0][1] = ex*st;
205  this->_p[0][2] = _c*t;
206 
207  if( this->_dm == GM_DERIVATION_EXPLICIT ) {
208  if( d > 0 ) {
209  this->_p[1][0] = ex*(_b*ct-st);
210  this->_p[1][1] = ex*(_b*st+ct);
211  this->_p[1][2] = _c;
212  }
213  if( d > 1 ) {
214  this->_p[2][0] = ex*((_b*_b-1)*ct - 2*_b*st);
215  this->_p[2][1] = ex*((_b*_b-1)*st + 2*_b*ct);
216  this->_p[2][2] = T(0);
217  }
218  if( d > 2 ) {
219  this->_p[3][0] = ex*((1-3*_b*_b)*st + (_b*_b*_b-3*_b)*ct);
220  this->_p[3][1] = ex*((3*_b*_b-1)*ct + (_b*_b*_b-3*_b)*st);
221  this->_p[3][2] = T(0);
222  }
223  }
224  }
