112  template <typename T, int n>
113  void PSurfNormalsVisualizer<T,n>::makePlotAll( const DMatrix< DMatrix< Vector<T, 3> > >& p, const DMatrix< Vector<float, 3> >& normals ) {
114 
115  int no_normals = p.getDim1() * p.getDim2();
116  _no_elements = no_normals * 2;
117 
118  _vbo.bind();
119  _vbo.bufferData( no_normals * 2 * sizeof(GL::GLVertex), 0x0, GL_STATIC_DRAW );
120 
121  GL::GLVertex *ptr = _vbo.mapBuffer<GL::GLVertex>();
122  if( ptr ) {
123 
124  for( int i = 0; i < p.getDim1(); i++ ) {
125  for( int j = 0; j < p.getDim2(); j++ ) {
126 
127  const Point<T,3> &pos = p(i)(j)(0)(0);
128  (*ptr).x = float(pos(0));
129  (*ptr).y = float(pos(1));
130  (*ptr).z = float(pos(2));
131  ptr++;
132 
133  const Vector<T,3> N = normals(i)(j).getNormalized() * _size;
134  (*ptr).x = float(pos(0)) + N(0);
135  (*ptr).y = float(pos(1)) + N(1);
136  (*ptr).z = float(pos(2)) + N(2);
137  ptr++;
138  }
139  }
140  }
141 
142  _vbo.unmapBuffer();
143  _vbo.unbind();
144  }
