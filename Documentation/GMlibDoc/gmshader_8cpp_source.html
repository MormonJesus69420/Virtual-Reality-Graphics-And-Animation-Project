116  void Shader::doDelete(GLuint id) const {
117 
118  // Detach shader from all programs it is attached to
119  const std::list<GMlib::GL::Private::ProgramInfo> &programs = GMlib::GL::Program::getData();
120  std::list<GMlib::GL::Private::ProgramInfo>::const_iterator p_itr;
121  for( p_itr = programs.begin(); p_itr != programs.end(); ++p_itr ) {
122 
123  GLint no_as;
124  GL_CHECK(::glGetProgramiv( p_itr->id, GL_ATTACHED_SHADERS, &no_as ));
125 
126  std::vector<GLuint> shader_ids(no_as);
127  GL_CHECK(::glGetAttachedShaders( p_itr->id, no_as, 0x0, &shader_ids[0] ) );
128 
129  std::vector<GLuint>::const_iterator s_itr =
130  std::find( shader_ids.begin(), shader_ids.end(), id );
131  if( s_itr != shader_ids.end() )
132  GL_CHECK(::glDetachShader( p_itr->id, *s_itr ));
133  }
134 
135  // delete shader
136  GL_CHECK(::glDeleteShader( id ));
137  }
