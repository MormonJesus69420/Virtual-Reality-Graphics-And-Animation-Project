136  // Build an orthogonal matrix
137  float near_plane = -1.0f;
138  float far_plane = 1.0f;
139 
140  float l, r, b, t, n, f;
141  l = 0.0f;
142  r = 1.0f;
143  b = 0.0f;
144  t = 1.0f;
145  n = near_plane;
146  f = far_plane;
147 
148  float A, B, C;
149  A = - ( r + l ) / ( r - l );
150  B = - ( t + b ) / ( t - b );
151  C = - ( f + n ) / ( f - n );
152 
153  _ortho_mat[0][0] = 2 / (r - l);
154  _ortho_mat[0][1] = 0.0f;
155  _ortho_mat[0][2] = 0.0f;
156  _ortho_mat[0][3] = A;
157 
158  _ortho_mat[1][0] = 0.0f;
159  _ortho_mat[1][1] = 2 / ( t - b );
160  _ortho_mat[1][2] = 0.0f;
161  _ortho_mat[1][3] = B;
162 
163  _ortho_mat[2][0] = 0.0f;
164  _ortho_mat[2][1] = 0.0f;
165  _ortho_mat[2][2] = - 2.0f / (f-n);
166  _ortho_mat[2][3] = C;
167 
168  _ortho_mat[3][0] = 0.0f;
169  _ortho_mat[3][1] = 0.0f;

207  void DefaultRenderer::initRenderProgram() {
208 
209 
210  const std::string prog_name = "render_prog";
211  if( _render_prog.acquire(prog_name) ) return;
212 
213 
214  std::string vs_src(
215  "#version 150 core\n"
216  "\n"
217  "uniform mat4 u_mvpmat;\n"
218  "\n"
219  "in vec4 in_vertex;\n"
220  "in vec2 in_tex_coord;\n"
221  "\n"
222  "out vec2 ex_tex_coord;\n"
223  "out vec4 gl_Position;\n"
224  "\n"
225  "void main() {\n"
226  "\n"
227  " ex_tex_coord = in_tex_coord;\n"
228  " gl_Position = u_mvpmat * in_vertex;\n"
229  "}\n"
230  );
231 
232  std::string fs_src(
233  "#version 150 core\n"
234  "\n"
235  "uniform sampler2D u_tex;\n"
236  "uniform sampler2D u_tex_selected;\n"
237  "uniform float u_buf_w;\n"
238  "uniform float u_buf_h;\n"
239  "uniform vec4 u_select_color;\n"
240  "\n"
241  "in vec2 ex_tex_coord;\n"
242  "\n"
243  "out vec4 gl_FragColor;\n"
244  "\n"
245  "float avg( vec4 v ) {\n"
246  "\n"
247  " return ( v.r + v.g + v. b ) / 4.0;\n"
248  "}\n"
249  "\n"
250  "\n"
251  "float runFilter( mat3 mask, float w, float h ) {\n"
252  "\n"
253  " float result = 0; //vec4( 0.0, 0.0, 0.0, 0.0 );\n"
254  "\n"
255  " for( float i = -1.0; i <= 1.0; i = i + 1.0 ) {\n"
256  " for( float j = -1.0; j <= 1.0; j = j + 1.0 ) {\n"
257  "\n"
258  " float x = ex_tex_coord.x + ( (i + 0.5) / w );\n"
259  " float y = ex_tex_coord.y + ( (j + 0.5) / h );\n"
260  "\n"
261  " float sub_res = avg( texture2D( u_tex_selected, vec2( x, y ) ) );\n"
262  "\n"
263  " result += sub_res * avg(\n"
264  " vec4(\n"
265  " mask[int(i+1.0)][int(j+1.0)],\n"
266  " mask[int(i+1.0)][int(j+1.0)],\n"
267  " mask[int(i+1.0)][int(j+1.0)],\n"
268  " mask[int(i+1.0)][int(j+1.0)]\n"
269  " )\n"
270  " );\n"
271  " }\n"
272  " }\n"
273  "\n"
274  " return result;\n"
275  "}\n"
276  "\n"
277  "void main() {\n"
278  "\n"
279  " // x and y sobel filters\n"
280  " mat3 mask_x = mat3(\n"
281  " -1.0, 0.0, 1.0,\n"
282  " -2.0, 0.0, 2.0,\n"
283  " -1.0, 0.0, 1.0\n"
284  " );\n"
285  "\n"
286  " mat3 mask_y = mat3(\n"
287  " 1.0, 2.0, 1.0,\n"
288  " 0.0, 0.0, 0.0,\n"
289  " -1.0, -2.0, -1.0\n"
290  " );\n"
291  "\n"
292  " // summation vars.\n"
293  " float res_x = runFilter( mask_x, u_buf_w, u_buf_h );\n"
294  " float res_y = runFilter( mask_y, u_buf_w, u_buf_h );\n"
295  " float res = sqrt( res_x * res_x + res_y * res_y );\n"
296  "\n"
297  " gl_FragColor = texture( u_tex, ex_tex_coord );\n"
298  "\n"
299  " if( res > 0.0 )\n"
300  " gl_FragColor = u_select_color;\n"
301  "\n"
302 // " float bah = texture( u_tex_selected, ex_tex_coord ).b; \n"
303 // " if( bah > 0.0 )\n"
304 // " gl_FragColor = vec4( bah, 0.0, 0.0, 1.0 );\n"
305 // " gl_FragColor = vec4( res, 0.0, 0.0, 1.0 );\n"
306 // " gl_FragColor = vec4( res_y, 0.0, 0.0, 1.0 );\n"
307 // " gl_FragColor = vec4( 0.0, res_y, 0.0, 1.0 );\n"
308  "}\n"
309  );
310 
311  bool compile_ok, link_ok;
312 
313  GL::VertexShader vshader;
314  vshader.create("render_vs");
315  vshader.setPersistent(true);
316  vshader.setSource(vs_src);
317  compile_ok = vshader.compile();
318  if( !compile_ok ) {
319  std::cout << "Src:" << std::endl << vshader.getSource() << std::endl << std::endl;
320  std::cout << "Error: " << vshader.getCompilerLog() << std::endl;
321  }
322  assert(compile_ok);
323 
324  GL::FragmentShader fshader;
325  fshader.create("render_fs");
326  fshader.setPersistent(true);
327  fshader.setSource(fs_src);
328  compile_ok = fshader.compile();
329  if( !compile_ok ) {
330  std::cout << "Src:" << std::endl << fshader.getSource() << std::endl << std::endl;
331  std::cout << "Error: " << fshader.getCompilerLog() << std::endl;
332  }
333  assert(compile_ok);
334 
335  _render_prog.create(prog_name);
336  _render_prog.setPersistent(true);
337  _render_prog.attachShader(vshader);
338  _render_prog.attachShader(fshader);
339  link_ok = _render_prog.link();
340  if( !link_ok ) {
341  std::cout << "Error: " << _render_prog.getLinkerLog() << std::endl;
342  }
343  assert(link_ok);
344  }

529  void DefaultRenderer::renderToTarget() {
530 
531 
532 
533 
534 // GL_CHECK(::glViewport(0,0,_size(0),_size(1)));
535 
536  GL_CHECK(::glPolygonMode( GL_FRONT_AND_BACK, GL_FILL ));
537  GL_CHECK(::glDisable(GL_DEPTH_TEST));
538 
539  // Draw scene composition
540  {
541  _render_prog.bind();
542 
543  _render_prog.uniform( "u_mvpmat", _ortho_mat );
544  _render_prog.uniform( "u_tex", getRenderTexture(), static_cast<GLenum>(GL_TEXTURE0), 0 );
545  _render_prog.uniform( "u_tex_selected", getSelectTexture(), static_cast<GLenum>(GL_TEXTURE1), 1 );
546  _render_prog.uniform( "u_buf_w", static_cast<float>(_size(0)) );
547  _render_prog.uniform( "u_buf_h", static_cast<float>(_size(1)) );
548  _render_prog.uniform( "u_select_color", _select_color );
549 
550  GL::AttributeLocation vert_loc = _render_prog.getAttributeLocation( "in_vertex" );
551  GL::AttributeLocation tex_coord_loc = _render_prog.getAttributeLocation( "in_tex_coord" );
552 
553  _quad_vbo.bind();
554 
555  _quad_vbo.enable( vert_loc, 3, GL_FLOAT, GL_FALSE, sizeof(GL::GLVertexTex2D), reinterpret_cast<const GLvoid*>(0x0) );
556  _quad_vbo.enable( tex_coord_loc, 2, GL_FLOAT, GL_FALSE, sizeof(GL::GLVertexTex2D), reinterpret_cast<const GLvoid*>(3*sizeof(GLfloat)) );
557 
558  GL_CHECK(::glDrawArrays( GL_QUADS, 0, 4 ));
559 
560  _quad_vbo.disable(tex_coord_loc);
561  _quad_vbo.disable(vert_loc);
562 
563  _quad_vbo.unbind();
564 
565 
566  _render_prog.unbind();
567  }
568 
569  GL_CHECK(::glEnable(GL_DEPTH_TEST));
570  }

600  void DefaultRenderer::renderScene() {
601 
602  // Setup size of viewport viewport to fit size of render target
603  GL_CHECK(::glViewport(0, 0, _size(0), _size(1)));
604 
605  // Clear render buffers
606  _fbo.clearColorBuffer( getClearColor() );
607  _fbo.clear( GL_DEPTH_BUFFER_BIT );
608 
609  _fbo_select.clearColorBuffer( GMcolor::black() );
610  _fbo_select_depth.clear( GL_DEPTH_BUFFER_BIT );
611 
612  // Object rendering
613  _fbo.bind(); {
614 
615  // Render coordinate-system visualization
616  renderCoordSys();
617 
618  // Render the scene objects
619  for( int j = 0; j < _objs.getSize(); ++j )
620  render(_objs[j]);
621 
622  } _fbo.unbind();
623 
624  // Selection rendering - render to depth buffer

660  const Array<Light*> &lights_array = scene->getLights();
661  const Sun* sun = scene->getSun();
662 
663  const HqMatrix<float,3> cammat = camera->SceneObject::getMatrix() * camera->getMatrixToSceneInverse();
664 
665  /*
666  * Light types of "sun", "point" and "spot" is supported.
667  * It is assumed that the lights is grouped and sorted in that order
668  * and that the "number of info" is recorded in the header block:
669  * <ul>
670  * <li>Total number of lights.</li>
671  * <li>Number of suns.</li>
672  * <li>Number of point lights.</li>
673  * <li>Number of spot lights.</li>

687
