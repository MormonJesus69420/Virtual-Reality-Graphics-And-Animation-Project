107  inline
108  PBSplineCurve<T>::PBSplineCurve(const PCurve<T,3>& cp, int d, int n, int m):PCurve<T,3>(0,0,10) {
109 
110  init(d);
111  _cl = cp.isClosed();
112  _c.setDim(n);
113  generateKnotVector(cp.getParStart(), cp.getParEnd(), _cl);
114  std::vector<T> sample;
115  computeUniformParamVal(sample, (_cl ? (m+1) : m), cp.getParStart(), cp.getParEnd());
116  if(_cl) sample.pop_back();
117 
118  DVector<Vector<T, 3> > p(m);
119  for(int i=0;i<m; i++)
120  p[i] = cp(sample[i]);
121  DMatrix<T> A(m,n);
122  makeLeastSquareMatrix(A, sample, d, n);
123  DMatrix<T> AT = A;
124  AT.transpose();
125  DMatrix<T> AA = AT*A;
126  AA.invert();
127  DVector<Vector<T,3>> bb = AT*p;
128  _c = AA*bb;
129  }

603  inline
604  void PBSplineCurve<T>::transKnot( DVector<T>& t, bool& c, const DVector<T>& g, int n, int d) {
605 
606  if(_c.getDim() <= d ) {
607  std::cerr << " There is too few control points for this polynomial degree" << std::endl;
608  return;
609  }
610  t.setDim(g.getDim()-d);
611  for(int i = d; i <= n; i++) t[i] = g(i);
612  for(int i = d-1; i >= 0; i--) t[i] = t[i+1];
613  for(int i = n+1; i < t.getDim(); i++) t[i] = t[i-1];
614  c = false;
615  }

632  template <typename T>
633  inline
634  void PBSplineCurve<T>::transKnotClose( DVector<T>& t, bool& c, const DVector<T>& g, int n, int d, T dt) {
635 
636  int k = d+1;
637  t.setDim(n+k+d);
638  if(dt > T(0)) {
639  for(int i = d; i < n+1; i++) t[i] = g(i);
640  for(int i = n+1; i < n+k; i++) t[i] = t[i-1] + dt;
641  } else
642  for(int i = d; i < n+k; i++) t[i] = g(i);
643 
644  for(int i=d-1, j=n+d; i >= 0; i--,j--) t[i] = t[i+1] - (t[j]-t[j-1]);
645  for(int i=n+k, j=k; i < t.getDim(); i++,j++) t[i] = t[i-1] + (t[j]-t[j-1]);
646  c = true;
647  }

664  void PBSplineCurve<T>::spesKnotClose( DVector<T>& t, bool& c, const DVector<T>& g, int n, int d) {
665 
666  t.setDim(n+d+d+1);
667  for(int i=d; i <= n+d; i++) t[i] = g(i-d);
668  for(int i=d-1, j=n+d; i >= 0; i--,j--) t[i] = t[i+1] - (t[j]-t[j-1]);
669  for(int i=n+d+1, j=d+1; i < t.getDim(); i++,j++) t[i] = t[i-1] + (t[j]-t[j-1]);

684  inline
685  void PBSplineCurve<T>::updateKnotClosed(int d) {
686 
687  int n = _c.getDim();
688  DVector<T> t = _t;
689 
690  _t.setDim(n+d+d+1);
691  int dd = d - _d;
692  if(dd<0)
693  for(int i=0; i < _t.getDim(); i++) _t[i] = t[i-dd];
694  else {
695  for(int i = dd; i < t.getDim()+dd; i++) _t[i] = t[i-dd];
696  for(int i = dd-1, j = t.getDim()+dd; i >= 0; i--, j++) {
697  _t[i] = _t[i+1] - (_t[n+i+1] - _t[n+i]);
698  _t[j] = _t[j-1] + (_t[j-n] - _t[j-n-1]);
699  }
700  }
701  _d = d;
702  _k = d+1;
703  }

923  template <typename T>
924  inline
925  void PBSplineCurve<T>::generateKnotVector( T s, T e, bool closed ) {
926 
927  if(closed) {
928  _t.setDim( _c.getDim() + _k + _d);
929  T dt = (e-s)/_c.getDim();
930 
931  for(int i = 0; i < _t.getDim(); i++ )
932  _t[i] = s+(i-_d)*dt;
933  }
934  else {
935  _t.setDim( _c.getDim() + _k );
936  T dt = (e-s)/(_c.getDim()-_d);
937 
938  int step = _t.getDim() - (2 * _k);
939  int i = 0;
940  for( ; i < _k; i++ ) // Set the start knots
941  _t[i] = s;
942  for( int j = 1; j <= step; j++ ) // Set the "step"-knots
943  _t[i++] = s+j*dt;
944  for( ; i < _t.getDim(); i++ ) // Set the end knots
945  _t[i] = e;
946  }
947  _cl = closed;
948  }
