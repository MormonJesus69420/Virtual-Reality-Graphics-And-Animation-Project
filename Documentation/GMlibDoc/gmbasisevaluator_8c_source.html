116  template <typename T>
117  T BasisEvaluator<T>::getIntegral( T a, T b, T sum, T eps ) {
118 
119  T t = b - a;
120 
121  T mat[16][16];
122  mat[0][0] = sum * t;
123 
124  T s;
125  int i,j, k;
126  for( i = 1; i < 16; i++ ) {
127 
128  s = T(0);
129  k = 1 << i;
130  t /= 2.0;
131 
132  for( j = 1; j < k; j += 2 )
133  s += getPhi( a + t * j );
134 
135  mat[0][i] = t * ( sum += s );
136 
137  for (j=1; j<=i; j++) {
138 
139  b = 1 << (j << 1);
140  mat[j][i-j] = ( b*mat[j-1][i-j+1] - mat[j-1][i-j] ) / (b-1.0);
141  }
142 
143  if( std::fabs( mat[i][0] - mat[i-1][0] ) < eps )
144  return mat[i][0];
145  }
146 
147  return mat[15][0];
148  }

177  template <typename T>
178  void BasisEvaluator<T>::init( int m ) {
179 
180  int i,j;
181  T p0, p1;
182  T f0, f1;
183 
184  _b.setDim(m+1);
185  _a.setDim( m, 5 );
186  _m = m;
187  _dt = T(1) / m;
188  _scale = T(1);
189  _b[0] = p0 = p1 = f0 = T(0);
190 
191  T t;
192  for( i = 1; i < m; i++ ) {
193 
194  t = _dt * i;
195  p1 = getPhi(t);
196  f1 = getF2(t) * p1;
197  _b[i] = getIntegral( t-_dt, t, 0.5*(p0+p1), 1e-17 );
198 
199  interpolate( i-1, p0, p1, _dt*f0, _dt*f1 );
200 
201  p0 = p1;
202  f0 = f1;
203  }
204 
205  _b[m] = getIntegral( _dt*(m-1), 1, 0.5*p0, 1e-17 );
206  interpolate( m-1, p0, p1, _dt*f0, T(0) );
207 
208  for( i = 1; i <= m; i <<= 1 )
209  for( j = m; j >= i; j -= 1 )
210  _b[j] += _b[j-i];
211 
212  _scale = T(1)/_b[m];
213  }

216  void BasisEvaluator<T>::interpolate( int i, double p0, double p1, double f0, double f1 ) {
217 
218  _a[i][0] = p0;
219  _a[i][1] = f0;
220  _a[i][2] = 3 * (p1-p0) - 2*f0 - f1;
