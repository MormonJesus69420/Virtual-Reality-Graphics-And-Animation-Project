80  template <typename T>
81  void PAsteroidalSphere<T>::eval( T u, T v, int d1, int d2, bool /*lu*/, bool /*lv*/ ) const {
82 
83  this->_p.setDim( d1+1, d2+1 );
84 
85  T cu = cos(u);
86  T cv = cos(v);
87  T su = sin(u);
88  T sv = sin(v);
89 
90 
91  this->_p[0][0][0] = T(pow(_a * cu*cv, T(3)));
92  this->_p[0][0][1] = T(pow(_b * su*cv, T(3)));
93  this->_p[0][0][2] = T(pow(_c * sv, T(3)));
94 
95  if( this->_dm == GM_DERIVATION_EXPLICIT ) {
96 
97  if(d1) { //u
98  this->_p[1][0][0] = -T(3)*_a*_a*_a*cu*cu*cv*cv*cv*su;
99  this->_p[1][0][1] = -T(3)*_b*_b*_b*(-1.0+cu*cu)*cv*cv*cv*cu;
100  this->_p[1][0][2] = T(0);
101  }
102  if(d1>1) { //uu
103  this->_p[2][0][0] = -T(3)*_a*_a*_a*cu*cv*cv*cv*(T(3)*cu*cu - T(2));
104  this->_p[2][0][1] = T(3)*_b*_b*_b*su*cv*cv*cv*(T(3)*cu*cu - T(1));
105  this->_p[2][0][2] = T(0);
106  }
107  if(d2) { //v
108  this->_p[0][1][0] = -T(3)*_a*_a*_a*cu*cu*cu*cv*cv*sv;
109  this->_p[0][1][1] = -T(3)*_b*_b*_b*cu*cu*su*cv*cv*sv;
110  this->_p[0][1][2] = T(3)*_c*_c*_c*cv*cv*cv;
111  }
112  if(d2>1) { //vv
113  this->_p[0][2][0] = -T(3)*_a*_a*_a*cu*cu*cu*cv*(T(3)*cv*cv - T(2));
114  this->_p[0][2][1] = T(3)*_b*_b*_b*(cu*cu - T(1))*su*cv*(T(3)*cv*cv - T(2));
115  this->_p[0][2][2] = T(3)*_c*_c*_c*sv*(T(3)*cv*cv - T(1));
116  }
117  if(d1 && d2) { //uv
118  this->_p[1][1][0] = T(9)*_a*_a*_a*cu*cu*cv*cv*su*sv;
119  this->_p[1][1][1] = T(9)*_b*_b*_b*(cu*cu - T(1))*cv*cv*cu*sv;
120  this->_p[1][1][2] = T(0);
121  }
122  if(d1>1 && d2) { //uuv
123  this->_p[2][1][0] = T(9)*_a*_a*_a*cu*cv*cv*sv*(T(3)*cu*cu - T(2));
124  this->_p[2][1][1] = -T(9)*_b*_b*_b*su*cv*cv*sv*(T(3)*cu*cu - T(1));
125  this->_p[2][1][2] = T(0);
126  }
127  if(d1 && d2>1) { //uvv
128  this->_p[1][2][0] = T(9)*_a*_a*_a*cu*cu*cv*su*(T(3)*cv*cv - T(2));
129  this->_p[1][2][1] = T(9)*_b*_b*_b*(cu*cu - T(1))*cv*cu*(T(3)*cv*cv - T(2));
130  this->_p[1][2][2] = T(0);
131  }
132  if(d1>1 && d2>1) { //uuvv
133  this->_p[2][2][0] = T(9)*_a*_a*_a*cu*cv*(T(4) - T(6)*cv*cv - T(6)*cu*cu + T(9)*cu*cu*cv*cv);
134  this->_p[2][2][1] = -T(9)*_b*_b*_b*su*cv*(T(9)*cu*cu*cv*cv - T(6)*cu*cu + T(2) - T(3)*cv*cv);
135  this->_p[2][2][2] = T(0);
136  }
137  }
