205  void prOut(T_Stream& out) const {
206 
207  if(_cu) { // closed in u-direction
208  DVector<T> u(_u.getDim()-2*_du);
209  for(int i = 0; i < u.getDim(); i++) u[i] = _u(i+_du);
210  out << u << std::endl << std::endl;

213  out << _u << std::endl << std::endl;
214 
215  if(_cv) { // closed in v-direction
216  DVector<T> v(_v.getDim()-2*_dv);
217  for(int i = 0; i < v.getDim(); i++) v[i] = _v(i+_dv);
218  out << v << std::endl << std::endl;

222 
223  out << _c << std::endl << std::endl;

226  template <typename T_Stream>
227  void prIn(T_Stream& in) {
228 
229  in >> _u >> _v;
230  in >> _c;
231 
232  // Set order
233  _ku = _u.getDim()-_c.getDim1();
234  _kv = _v.getDim()-_c.getDim2();
235 
236  if(_ku == 1) { // closed in u-direction
237  DVector<T> u = _u;
238  in >> _du;
239  initKnot( _u, _cu, _ku, u, _c.getDim1(), _du);
240  }
241  else {
242  _cu = false;
243  _du = _ku-1;
244  }
245 
246  if(_kv == 1) { // closed in v-direction
247  DVector<T> v = _v;
248  in >> _dv;
249  initKnot( _v, _cv, _kv, v, _c.getDim2(), _dv);
250  }
251  else {
252  _cv = false;
253  _dv = _kv-1;
254  }
255  }
