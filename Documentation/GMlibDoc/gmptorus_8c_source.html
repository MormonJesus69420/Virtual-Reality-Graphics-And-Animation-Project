125  template <typename T>
126  void PTorus<T>::eval(T u, T v, int d1,int d2, bool /*lu*/, bool /*lv*/ ) const {
127 
128  this->_p.setDim(d1+1,d2+1);
129 
130  T su = sin(u);
131  T sv = sin(v);
132  T cu = cos(u);
133  T cv = cos(v);
134  T bcva = _b*cv+_a;
135  T cucv = _b*cu*cv;
136  T cusv = _b*cu*sv;
137  T sucv = _b*su*cv;
138  T susv = _b*su*sv;
139  sv *= _c;
140  cv *= _c;
141  cu *= bcva;
142  su *= bcva;
143 
144  //S
145  this->_p[0][0][0] = cu;
146  this->_p[0][0][1] = su;
147  this->_p[0][0][2] = sv;
148 
149  if( this->_dm == GM_DERIVATION_EXPLICIT ) {
150 
151  if(d1) { //Su
152  this->_p[1][0][0] = -su;
153  this->_p[1][0][1] = cu;
154  this->_p[1][0][2] = T(0);
155  }
156  if(d1>1) { //Suu
157  this->_p[2][0][0] = -cu;
158  this->_p[2][0][1] = -su;
159  this->_p[2][0][2] = T(0);
160  }
161  if(d2) { //Sv
162  this->_p[0][1][0] = -cusv;
163  this->_p[0][1][1] = -susv;
164  this->_p[0][1][2] = cv;
165  }
166  if(d2>1) { //Svv

168  this->_p[0][2][1] = -sucv;
169  this->_p[0][2][2] = -sv;
170  }

173  this->_p[1][1][1] = -cusv;
174  this->_p[1][1][2] = T(0);
175  }
176  if(d1>1 && d2) { //Suuv
