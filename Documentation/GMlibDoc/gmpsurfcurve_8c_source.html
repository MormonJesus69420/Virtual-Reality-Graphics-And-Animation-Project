97  template <typename T>
98  void PSurfCurve<T>::resample( DVector< DVector< Vector<T, 3> > >& p, int m, int d, T start, T end )
99  {
100  if (_der_curve && _plot)
101  {
102  T du = (end-start)/(m-1);
103  p.setDim(m);
104 
105  for( int i = 0; i < m - 1; i++ )
106  {
107  p[i].setDim(d+1);
108  eval1(start + i * du, 1);
109  p[i][0] = this->_p[0];
110  eval2(start + i * du, d-1);
111  for(int j=1; j<d;j++)
112  p[i][j] = this->_p[j-1];
113  }
114  p[m-1].setDim(d+1);
115  eval1(end, 1);
116  p[m-1][0] = this->_p[0];
117  eval2(end, d-1);
118  for(int j=1; j<d;j++)
119  p[m-1][j] = this->_p[j-1];
120 
121  switch( this->_dm )
122  {
123  case GM_DERIVATION_EXPLICIT:
124  // Do nothing, evaluator algorithms for explicite calculation of derivatives
125  // should be defined in the eval( ... ) function enclosed by
126  // if( this->_derivation_method == this->EXPLICIT ) { ... eval algorithms for derivatives ... }
127  break;
128 
129  case GM_DERIVATION_DD:
130  default:
131  // _evalDerDD( p, d, du );
132  break;
133  };
134  }
135  else
136  PCurve<T,3>::resample( p, m, d, start, end );
137  }

184  template <typename T>
185  inline
186  void PSurfCurve<T>::eval1( T t, int d ) const {
187  this->_p.setDim( d + 1 );
188 
189  DMatrix< Vector<T,3> > m = _s->evaluateParent(_p1 + t*_dv , d);
190 
191  this->_p[0] = m[0][0];
192  if(d>0)
193  {
194  Matrix<T,3,2> d1;
195  d1.setCol(m[1][0], 0);
196  d1.setCol(m[0][1], 1);
197  this->_p[1] = d1*_dv;
198  }
199  if(d>1)
200  {
201  Matrix<T,3,2> d2;
202  d2.setCol(m[2][0], 0);
203  d2.setCol(m[1][1], 1);
204  Vector<T,3> v1 = d2*_dv;
205  Matrix<T,3,2> d3;
206  d3.setCol(m[1][1], 0);
207  d3.setCol(m[0][2], 1);
208  Vector<T,3> v2 = d3*_dv;
209  d2.setCol(v1, 0);
210  d2.setCol(v2, 1);
211  this->_p[2]= d2*_dv;
212  }
213  if(d>2)
214  {
215  Matrix<T,3,2> d4;
216  d4.setCol(m[3][0], 0);
217  d4.setCol(m[2][1], 1);
218  Vector<T,3> v1 = d4*_dv;
219  Matrix<T,3,2> d5;
220  d5.setCol(m[2][1], 0);
221  d5.setCol(m[1][2], 1);
222  Vector<T,3> v2 = d5*_dv;
223  d4.setCol(v1, 0);
224  d4.setCol(v2, 1);
225  v1 = d4*_dv;
226 
227  d4.setCol(m[1][2], 0);
228  d4.setCol(m[0][3], 1);
229  Vector<T,3> v3 = d4*_dv;
230  d5.setCol(v2, 0);
231  d5.setCol(v3, 1);
232  v2 = d5*_dv;
233  d4.setCol(v1, 0);
234  d4.setCol(v2, 1);
235  this->_p[3] = d4*_dv;
236  }
237  }
