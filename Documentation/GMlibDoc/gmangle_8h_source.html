385  inline
386  double Angle::translate( double b ) {
387 
388  double tmp_angle = _angle;
389 
390  // If current angle (tmpAngle) is lower than LowerBondary (b),
391  // then translate current angle forward with 2Pi until it
392  // is higher than the bondary
393  while( tmp_angle < b ) { tmp_angle += M_2PI; }
394 
395  // Change to UpperBondary
396  b += M_2PI;
397 
398  // If current angle (tmpAngle) is higher than UpperBondary (b),
399  // then translate current angle backward with 2Pi until it
400  // is lower than the bondary
401  while( tmp_angle > b ) { tmp_angle -= M_2PI; }
402 
403  return tmp_angle;
404  }
