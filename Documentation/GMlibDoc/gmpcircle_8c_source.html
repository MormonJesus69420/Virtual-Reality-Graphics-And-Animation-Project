136  template <typename T>
137  void PCircle<T>::eval( T t, int d, bool /*l*/ ) const {
138 
139  this->_p.setDim( d + 1 );
140 
141  const T ct = _r * cos(t);
142  const T st = _r * sin(t);
143 
144  this->_p[0][0] = ct;
145  this->_p[0][1] = st;
146  this->_p[0][2] = T(0);
147 
148  if( this->_dm == GM_DERIVATION_EXPLICIT ) {
149  if( d > 0 ) {
150  this->_p[1][0] = -st;
151  this->_p[1][1] = ct;
152  this->_p[1][2] = T(0);
153  }
154  if( d > 1 ) this->_p[2] = -this->_p[0];
155  if( d > 2 ) this->_p[3] = -this->_p[1];
156  if( d > 3 ) this->_p[4] = this->_p[0];
157  if( d > 4 ) this->_p[5] = this->_p[1];
158  if( d > 5 ) this->_p[6] = this->_p[2];
159  if( d > 6 ) this->_p[7] = this->_p[3];
160  }
161  }
