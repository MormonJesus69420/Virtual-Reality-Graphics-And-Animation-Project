145  void PointLight::calculateRadius(float constant, float linear, float quadratic) {
146 
147  double aa, solution;
148  if (GMutils::compValueF(linear,0.0f) && GMutils::compValueF(quadratic,0.0f))
149  {
150  setCullable(false); // Disable culling
151  solution = FLT_MAX;
152  }
153  else if (GMutils::compValueF(quadratic,0.0f))
154  {
155  if(constant <= _min_light_contribution)
156  solution = (_min_light_contribution-constant) / linear;
157  else
158  solution = 0;
159  }
160  else
161  {
162  if(constant <= _min_light_contribution)
163  aa= sqrt(linear*linear + 4 * (_min_light_contribution-constant) * quadratic);
164  else
165  aa = linear;
166  solution = (aa -linear)/(2 * quadratic);
167  }
168  _light_sphere.resetValue(solution);
169  }
170 }
