54  template <typename T>
55  inline
56  void TriangleFacetsDefaultVisualizer<T>::render(const SceneObject *obj, const DefaultRenderer *renderer) const {
57 
58 
59  const Camera* cam = renderer->getCamera();
60 
61  const HqMatrix<float,3> &mvmat = obj->getModelViewMatrix(cam);
62  const HqMatrix<float,3> &pmat = obj->getProjectionMatrix(cam);
63 
64  SqMatrix<float,3> nmat = mvmat.getRotationMatrix();
65 
66  this->glSetDisplayMode();
67 
68  _prog.bind(); {
69 
70  // Model view and projection matrices
71  _prog.uniform( "u_mvmat", mvmat );
72  _prog.uniform( "u_mvpmat", pmat * mvmat );
73  _prog.uniform( "u_nmat", nmat );
74 
75  // Lights
76  _prog.bindBufferBase( "DirectionalLights", renderer->getDirectionalLightUBO(), 0 );
77  _prog.bindBufferBase( "PointLights", renderer->getPointLightUBO(), 1 );
78  _prog.bindBufferBase( "SpotLights", renderer->getSpotLightUBO(), 2 );
79 
80  // Get Material Data
81  const Material &m = obj->getMaterial();
82  _prog.uniform( "u_mat_amb", m.getAmb() );
83  _prog.uniform( "u_mat_dif", m.getDif() );
84  _prog.uniform( "u_mat_spc", m.getSpc() );
85  _prog.uniform( "u_mat_shi", m.getShininess() );
86 
87  GL::AttributeLocation vert_loc = _prog.getAttributeLocation( "in_vertex" );
88  GL::AttributeLocation normal_loc = _prog.getAttributeLocation( "in_normal" );
89 
90  _vbo.bind();
91  _vbo.enable( vert_loc, 3, GL_FLOAT, GL_FALSE, sizeof(GL::GLVertexNormal), 0x0 );
92  _vbo.enable( normal_loc, 3, GL_FLOAT, GL_TRUE, sizeof(GL::GLVertexNormal), reinterpret_cast<const GLvoid*>(sizeof(GL::GLVertex)) );
93 
94  draw();
95 
96  _vbo.disable( vert_loc );
97  _vbo.disable( normal_loc );
98  _vbo.unbind();
99 
100  } _prog.unbind();
101  }

122  template <typename T>
123  void TriangleFacetsDefaultVisualizer<T>::initShader() {
124 
125  const std::string prog_name = "triangle_facets_prog";
126 
127  if(_prog.acquire(prog_name)) return;
128 
129  std::string vs_str =
130  GMlib::GL::OpenGLManager::glslDefHeaderVersionSource() +
131 
132  "uniform mat4 u_mvmat, u_mvpmat;\n"
133  "\n"
134  "in vec4 in_vertex;\n"
135  "in vec4 in_normal;\n"
136  "in vec2 in_tex;\n"
137  "\n"
138  "out vec4 gl_Position;\n"
139  "\n"
140  "smooth out vec3 ex_pos;\n"
141  "smooth out vec3 ex_normal;\n"
142  "\n"
143  "void main() {\n"
144  "\n"
145  " // Transform the normal to view space\n"
146  " mat3 nmat = inverse( transpose( mat3( u_mvmat ) ) );\n"
147  " ex_normal = nmat * vec3(in_normal);\n"
148  "\n"
149  " // Transform position into view space;\n"
150  " vec4 v_pos = u_mvmat * in_vertex;\n"
151  " ex_pos = v_pos.xyz * v_pos.w;\n"
152  "\n"
153  " // Compute vertex position\n"
154  " gl_Position = u_mvpmat * in_vertex;\n"
155  "}\n"
156  ;
157 
158  std::string fs_str =
159  GMlib::GL::OpenGLManager::glslDefHeaderVersionSource() +
160  GMlib::GL::OpenGLManager::glslFnComputeBlinnPhongLightingSource() +
161 
162  "uniform mat4 u_mvmat;\n"
163  "uniform mat3 u_nmat;\n"
164  "\n"
165  "uniform vec4 u_mat_amb;\n"
166  "uniform vec4 u_mat_dif;\n"
167  "uniform vec4 u_mat_spc;\n"
168  "uniform float u_mat_shi;\n"
169  "\n"
170  "smooth in vec3 ex_pos;\n"
171  "smooth in vec3 ex_normal;\n"
172  "\n"
173  "out vec4 gl_FragColor;\n"
174  "\n"
175  "void main() {\n"
176  "\n"
177  " vec3 normal = normalize( u_nmat * ex_normal );\n"
178  "\n"
179  " Material mat;\n"
180  " mat.ambient = u_mat_amb;\n"
181  " mat.diffuse = u_mat_dif;\n"
182  " mat.specular = u_mat_spc;\n"
183  " mat.shininess = u_mat_shi;\n"
184  "\n"
185  " gl_FragColor = computeBlinnPhongLighting( mat, ex_pos, normal );\n"
186  "}\n"
187  ;
188 
189  bool compile_ok, link_ok;
190 
191  GMlib::GL::VertexShader vshader;
192  vshader.create("phong_tf_vs");
193  vshader.setPersistent(true);
194  vshader.setSource(vs_str);
195  compile_ok = vshader.compile();
196  assert(compile_ok);
197 
198  GMlib::GL::FragmentShader fshader;
199  fshader.create("phong_tf_fs");
200  fshader.setPersistent(true);
201  fshader.setSource(fs_str);
202  compile_ok = fshader.compile();
203  if( !compile_ok ) {
204  std::cout << "Src:" << std::endl << fshader.getSource() << std::endl << std::endl;
205  std::cout << "Error: " << fshader.getCompilerLog() << std::endl;
206  }
207  assert(compile_ok);
208 
209  _prog.create(prog_name);
210  _prog.setPersistent(true);
211  _prog.attachShader( vshader );
212  _prog.attachShader( fshader );
213  link_ok = _prog.link();
214  assert(link_ok);
215 
216  }
