28 template <typename T> // Described on page 91-92 in "Blend book"
29 void EvaluatorStatic<T>::evaluateBhp( DMatrix<T>& mat, int d, T t, T scale ) {
30 
31  // Initiate result matrix
32  mat.setDim( d+1, d+1 );
33 
34  // Escape if the degree is 0
35  if( d < 1 ) {
36  mat[0][0] = T(1);
37  return;
38  }
39 
40  // Compute the Bernstein (polynomials), degree 1 -> d, one for each row.
41  // Starts from the second bottom row (degree 1), then goes upwards (degree 2,...,d).
42  mat[d-1][0] = 1 - t;
43  mat[d-1][1] = t;
44 
45  for( int i = d-2; i >= 0; i-- ) {
46  mat[i][0] = ( 1 - t) * mat[i+1][0];
47  for( int j = 1; j < d - i; j++ )
48  mat[i][j] = t * mat[i+1][j-1] + (1 - t) * mat[i+1][j];
49  mat[i][d-i] = t * mat[i+1][d-i-1];
50  }
51 
52  // Compute all the deriatives
53  mat[d][0] = -scale;
54  mat[d][1] = scale;
55 
56  for( int k = 2; k <= d; k++ ) {
57  const double s = k * scale;
58  for( int i = d; i > d - k; i-- ) {
59  mat[i][k] = s * mat[i][k-1];
60  for( int j = k - 1; j > 0; j-- )
61  mat[i][j] = s * ( mat[i][j-1] - mat[i][j] );
62  mat[i][0] = - s * mat[i][0];
63  }
64  }
65 }

81  template <typename T>
82  void EvaluatorStatic<T>::evaluateBSp2( DMatrix<T>& mat, T t, const DVector<T>& tv, int d, int ii, T scale ){
83 
84  // Knot-index ii must be: tv[ii] <= t < tv[ii+1] if right-evaluation
85  // else tv[ii] < t <= tv[ii+1] if left-evaluation
86 
87  // For the linar factor - mapping the knot intervalls to [0,1].
88  DVector<T> w(d);
89 
90  // Compute the B-splines (polynomials), degree 1 -> d, one for each row.
91  // Starts from the second bottom row (degree 1), then goes upwards (degree 2,...,d).
92  mat.setDim( d+1, d+1 );
93 
94  mat[d-1][1] = getW( tv, t, ii, 1 );
95  mat[d-1][0] = 1 - mat[d-1][1];
96 
97  for( int i = d - 2, k = 2; i >= 0; i--, k++ ) {
98  // Generate w, expression 5.45 (page 119) in "Blend book".
99  for( int j = 0; j < k; j++ )
100  w[j] = getW( tv, t, ii-k+j+1, k );
101 
102  // Compute the k b-splines
103  mat[i][0] = ( 1 - w[0]) * mat[i+1][0];
104  for( int j = 1; j < d - i; j++ )
105  mat[i][j] = w[j-1] * mat[i+1][j-1] + (1 - w[j]) * mat[i+1][j];
106  mat[i][d-i] = w[k-1] * mat[i+1][d-i-1];
107  }
108 
109  // Compute all deriatives for the derivatives B-splines (polynomials), 1st, 2nd,...,d-order derivatives.
110  mat[d][1] = delta( tv, ii, 1, scale );
111  mat[d][0] = -mat[d][1];
112 
113  for( int k = 2; k <= d; k++ ) {
114  // Generate w' (delta) for the derivatives, expression 5.47 (page 121) in "Blend book".
115  for( int j = 0; j < k; j++ )
116  w[j] = k * delta( tv, ii-k+j+1, k, scale);
117 
118  for( int i = d; i > d - k; i--) {
119  mat[i][k] = w[k-1] * mat[i][k-1];
120  for( int j = k - 1; j > 0; j--)
121  mat[i][j] = w[j-1]*mat[i][j-1] - w[j]*mat[i][j] ;
122  mat[i][0] = - w[0] * mat[i][0];
123  }
124  }
125  }

128  template <typename T>
129  void EvaluatorStatic<T>::evaluateH3d( DMatrix<T>& mat, int d, T t) {
130 
131  mat.setDim( d+1, 4 );
132 
133  T t2 = t*t;
134  T t3 = t*t2;
135 
136  // Compute the four Hermite polynomials (f_1, f_2, f'_1, f'_2)
137  // expression 4.16 (page 78) in "Blend book".
138  mat[0][1] = 3*t2 - 2*t3;
139  mat[0][0] = 1 - mat[0][1];
140  mat[0][2] = t - 2*t2 + t3;
141  mat[0][3] = t3 - t2;
142  if(d>0) { // First derivatives
143  mat[1][1] = 6*(t - t2);
144  mat[1][0] = -mat[1][1];
145  mat[1][2] = 1 - 4*t + 3*t2;
146  mat[1][3] = 3*t2 - 2*t;
147  if(d>1) { // Second derivatives
148  mat[2][1] = 6 - 12*t;
149  mat[2][0] = -mat[2][1];
150  mat[2][2] = 6*t - 4;
151  mat[2][3] = 6*t - 2;
152  if(d>2) { // Third derivatives
153  mat[3][1] = -12;
154  mat[3][0] = 12;
155  mat[3][2] = 6;
156  mat[3][3] = 6;
157  }
158  }
159  }

227  inline
228  int EvaluatorStatic<T>::knotIndex(const DVector<T>& tv, T t, int d, bool left) {
229  // Return knot-index i such that: tv[i] <= t < tv[i+1] if right-evaluation
230  // else tv[i] < t <= tv[i+1] if left-evaluation
231  int i = d;
232  int j = tv.getDim()-d-1; // = n
233  if(left){
234  do{
235  int k=(i+j)/2;
236  if(t > tv(k)) i=k;
237  else j=k;
238  }while(j-i>1);
239  }
240  else {
241  do{
242  int k=(i+j)/2;
243  if(t < tv(k)) j=k;
244  else i=k;
245  }while(j-i>1);
246  }
247  return i;
248  }

256  T EvaluatorStatic<T>::knotInterval( const DVector<T>& t, int d, int n ) {
257  // Return average knot-interval, that is average for intervalls different from zero
258  int i = 0;
259  for(int j=d; j<n; j++)
260  if(t[j+1] > t[j]) i++;
261  return (t[n]-t[d])/T(i);
