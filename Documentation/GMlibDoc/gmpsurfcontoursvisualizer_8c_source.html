127  inline
128  T PSurfContoursVisualizer<T>::getCurvatureGauss( DMatrix< Vector<T,3> >& p ) {
129 
130  UnitVector<T,3> N = p[1][0]^p[0][1];
131  Vector<T,3> du = p[1][0];
132  Vector<T,3> dv = p[0][1];
133  Vector<T,3> duu = p[2][0];
134  Vector<T,3> duv = p[1][1];
135  Vector<T,3> dvv = p[0][2];
136 
137  T E = du * du;
138  T F = du * dv;
139  T G = dv * dv;
140  T e = N * duu;
141  T f = N * duv;
142  T g = N * dvv;
143 
144  return (e*g - f*f) / (E*G - F*F);
145  }

147  template <typename T>
148  inline
149  T PSurfContoursVisualizer<T>::getCurvatureMean( DMatrix< Vector<T,3> >& p ) {
150 
151  UnitVector<T,3> N = p[1][0]^p[0][1];
152  Vector<T,3> du = p[1][0];
153  Vector<T,3> dv = p[0][1];
154  Vector<T,3> duu = p[2][0];
155  Vector<T,3> duv = p[1][1];
156  Vector<T,3> dvv = p[0][2];
157 
158  T E = du * du;
159  T F = du * dv;
160  T G = dv * dv;
161  T e = N * duu;
162  T f = N * duv;
163  T g = N * dvv;
164 
165  return 0.5 * (e*G - 2 * (f*F) + g*E) / (E*G - F*F);
166  }

170  T PSurfContoursVisualizer<T>::getCurvaturePrincipalMax( DMatrix< Vector<T,3> >& p ) {
171 
172  T K = getCurvatureGauss( p );
173  T H = getCurvatureMean( p );
174 
175  return H + sqrt( H*H - K );

200  template <typename T>
201  inline
202  void PSurfContoursVisualizer<T>::replot(
203  DMatrix< DMatrix< Vector<T, 3> > >& p,
204  DMatrix< Vector<T, 3> >& normals,
205  int m1, int m2, int d1, int d2,
206  bool /*closed_u*/, bool /*closed_v*/
207  ) {
208 
209  _tri_strips = PSurfVisualizer<T>::getNoTriangleStrips( p.getDim1(), p.getDim2() );
210  _indices_per_tri_strip = PSurfVisualizer<T>::getNoIndicesPerTriangleStrip( p.getDim1(), p.getDim2() );
211  _tri_strip_offset = sizeof(GLushort) * _indices_per_tri_strip;
212 
213  PSurfVisualizer<T>::fillTriangleStripIBO( _ibo, p.getDim1(), p.getDim2() );
214 
215 
216  // Color/Material Countours
217  DMatrix<double> cmap;
218 
219  T min, max;
220  T C;
221  cmap.setDim( p.getDim1(), p.getDim2() );
222  switch( _mapping ) {
223 
224  case GM_PSURF_CONTOURSVISUALIZER_U:
225  {
226  for( int i = 0; i < p.getDim1(); i++ )
227  for( int j = 0; j < p.getDim2(); j++ )
228  cmap[i][j] = double(i) / double(p.getDim1()-1);
229  }
230  break;
231  case GM_PSURF_CONTOURSVISUALIZER_V:
232  {
233  for( int i = 0; i < p.getDim1(); i++ )
234  for( int j = 0; j < p.getDim2(); j++ )
235  cmap[i][j] = double(j) / double(p.getDim2()-1);
236  }
237  break;
238 
239  default:
240  {
241  // Init min/max
242  min = max = getValue(p, 0, 0);
243 
244  // Extract all Gaussian Curvature data
245  for( int i = 0; i < p.getDim1(); i++ ) {
246  for( int j = 0; j < p.getDim2(); j++ ) {
247 
248  const T curvature = getValue(p, i, j);
249  if( curvature < min )
250  min = curvature;
251  if( curvature > max )
252  max = curvature;
253  }
254  }
255 
256  // Correct interval
257  if( (max-min) < 1e-5 ) {
258 
259  for( int i = 0; i < p.getDim1(); i++ )
260  for( int j = 0; j < p.getDim2(); j++ )
261  cmap[i][j] = 0.0;
262  }
263  else {
264  C = 1.0f / (max-min);
265  min /= (max-min);
266 
267  // Compute interpolated map values
268  for( int i = 0; i < p.getDim1(); i++ ) {
269  for( int j = 0; j < p.getDim2(); j++ ) {
270 
271  const T curvature = getValue(p, i, j);
272  cmap[i][j] = ( C * curvature ) - min;
273  }
274  }
275  }
276  }
277  break;
278  }
279 
280 
281  glBindBuffer( GL_ARRAY_BUFFER, _vbo );
282  Vertex data[p.getDim1()*p.getDim2()];
283 
284  // Fill vertex point data.
285  for( int i = 0; i < p.getDim1(); i++ ) {
286  for( int j = 0; j < p.getDim2(); j++ ) {
287 
288  const int idx = i * p.getDim2() + j;
289  data[idx].x = p[i][j][0][0][0];
290  data[idx].y = p[i][j][0][0][1];
291  data[idx].z = p[i][j][0][0][2];
292  }
293  }
294 
295 
296  // Fill vertex color data
297  switch( _method ) {
298  case GM_PSURF_CONTOURSVISUALIZER_NO_INTERPOLATION:
299  for( int i = 0; i < p.getDim1(); i++ ) {
300  for( int j = 0; j < p.getDim2(); j++ ) {
301 
302  const int idx = i * p.getDim2() + j;
303  const Color c = getColor( cmap[i][j] );
304  data[idx].r = c.getRedC();
305  data[idx].g = c.getGreenC();
306  data[idx].b = c.getBlueC();
307  data[idx].a = c.getAlphaC();
308  }
309  }
310  break;
311 
312  case GM_PSURF_CONTOURSVISUALIZER_LINEAR:
313  default:
314  for( int i = 0; i < p.getDim1(); i++ ) {
315  for( int j = 0; j < p.getDim2(); j++ ) {
316 
317  const int idx = i * p.getDim2() + j;
318  const Color c = getColorInterpolated( cmap[i][j] );
319  data[idx].r = c.getRedC();
320  data[idx].g = c.getGreenC();
321  data[idx].b = c.getBlueC();
322  data[idx].a = c.getAlphaC();
323  }
324  }
325  break;
326  }
327 
328  glBindBuffer( GL_ARRAY_BUFFER, _vbo );
329  glBufferData( GL_ARRAY_BUFFER, p.getDim1() * p.getDim2() * sizeof(Vertex), data, GL_STATIC_DRAW );
330  glBindBuffer( GL_ARRAY_BUFFER, 0x0 );
331  }

333  template <typename T>
334  inline
335  T PSurfContoursVisualizer<T>::getValue( DMatrix< DMatrix< Vector<T, 3> > >& p, int i, int j ) {
336 
337  switch( _mapping ) {
338  case GM_PSURF_CONTOURSVISUALIZER_X:
339  return p[i][j][0][0][0];
340  case GM_PSURF_CONTOURSVISUALIZER_Y:
341  return p[i][j][0][0][1];
342  case GM_PSURF_CONTOURSVISUALIZER_Z:
343  return p[i][j][0][0][2];
344 
345  default: break;
346  }
347 
348 
349  // Values requireing 1st derivatives
350  if( p[i][j].getDim1() < 2 || p[i][j].getDim2() < 2 )
351  return T(0);
352 
353  switch( _mapping ) {
354  case GM_PSURF_CONTOURSVISUALIZER_U:
355  return p[i][j][1][0].getLength();
356  case GM_PSURF_CONTOURSVISUALIZER_V:
357  return p[i][j][0][1].getLength();
358 
359  default: break;
360  }
361 
362 
363  // Values requireing 2nd derivatives
364  if( p[i][j].getDim1() < 3 || p[i][j].getDim2() < 3 )
365  return T(0);
366 
367  switch( _mapping ) {
368  case GM_PSURF_CONTOURSVISUALIZER_CURVATURE_GAUSS:
369  return getCurvatureGauss(p[i][j]);
370 
371  case GM_PSURF_CONTOURSVISUALIZER_CURVATURE_MEAN:
372  return getCurvatureMean(p[i][j]);
373 
374  case GM_PSURF_CONTOURSVISUALIZER_CURVATURE_PRINCIPAL_MAX:
375  return getCurvaturePrincipalMax( p[i][j] );
376 
377  case GM_PSURF_CONTOURSVISUALIZER_CURVATURE_PRINCIPAL_MIN:
378  return getCurvaturePrincipalMin( p[i][j] );
379 
380  default: break;
381  }
382 
383  return T(0);
384  }
