252  inline
253  void DVector<T>::increaseDim(int i, T val, bool at_end) {
254 
255  if(i>0)
256  {
257  int k,j = _n + i;
258  T* tmp = (j>4 ? new T[j] : _init);
259  if(at_end)
260  {
261  if(j>4) for (k=0; k<_n; k++) tmp[k] = _p[k];
262  for( k=_n; k<j; k++ ) tmp[k] = val;
263  }
264  else
265  {
266  for (k=j-1; k>=i; k--) tmp[k] = _p[k-i];
267  for (k=0; k<i; k++) tmp[k] = val;
268  }
269  if(_p != _init) delete [] _p;
270  _p = tmp;
271  _n = j;
272  }
