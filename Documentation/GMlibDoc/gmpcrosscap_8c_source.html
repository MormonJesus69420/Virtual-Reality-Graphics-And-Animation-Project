79  template <typename T>
80  void PCrossCap<T>::eval(T u, T v, int d1, int d2, bool /*lu*/, bool /*lv*/ ) const {
81 
82  this->_p.setDim( d1+1, d2+1 );
83 
84  T a = _r*_r*cos(u)*sin(u)*sin(v);
85  T b = _r*_r*sin(2*v)*cos(u)*cos(u);
86  T c = _r*_r*cos(2*v)*cos(u)*cos(u);
87 
88  this->_p[0][0][0] = a;
89  this->_p[0][0][1] = b;
90  this->_p[0][0][2] = c;
91 
92  T d, e, f, g;
93 
94  if( this->_dm == GM_DERIVATION_EXPLICIT ) {
95 
96  if(d1) //u
97  {
98  d = 2*_r*_r*sin(2*v)*cos(u)*sin(u);
99  e = 2*_r*_r*cos(2*v)*cos(u)*sin(u);
100  this->_p[1][0][0] = -_r*_r*sin(u)*sin(u)*sin(v) + _r*_r*cos(u)*cos(u)*sin(v);
101  this->_p[1][0][1] = -d;
102  this->_p[1][0][2] = -e;
103  }
104  if(d1>1)//uu
105  {
106  f = 2*_r*_r*sin(2*v)*sin(u)*sin(u);
107  g = 2*_r*_r*cos(2*v)*sin(u)*sin(u);
108  this->_p[2][0][0] = -4*a;
109  this->_p[2][0][1] = f - 2*b;
110  this->_p[2][0][2] = g - 2*c;
111  }
112  if(d2) //v
113  {
114  this->_p[0][1][0] = _r*_r*cos(u)*sin(u)*cos(v);
115  this->_p[0][1][1] = 2*c;
116  this->_p[0][1][2] = -2*b;
117  }
118  if(d2>1) //vv
119  {
120  this->_p[0][2][0] = -a;
121  this->_p[0][2][1] = -4*b;
122  this->_p[0][2][2] = -4*c;
123  }
124  if(d1 && d2) //uv
125  {
126  this->_p[1][1][0] = -_r*_r*sin(u)*sin(u)*cos(v) + _r*_r*cos(u)*cos(u)*cos(v);
127  this->_p[1][1][1] = -2*e;
128  this->_p[1][1][2] = 2*d;
129  }
130  if(d1>1 && d2)//uuv
131  {
132  this->_p[2][1][0] = -4*this->_p[0][1][0];
133  this->_p[2][1][1] = 2*g - 4*c;
134  this->_p[2][1][2] = -2*f + 4*b;
135  }
136  if(d1 && d2>1) //uvv
137  {
138  this->_p[1][2][0] = -this->_p[1][0][0];
139  this->_p[1][2][1] = 4*d;
140  this->_p[1][2][2] = 4*e;
141  }
142  if(d1>1 && d2>1) //uuvv
143  {
144  this->_p[2][2][0] = 4*a;
145  this->_p[2][2][1] = -4*f + 8*b;
146  this->_p[2][2][2] = -4*g + 8*c;
147  }
148  }
149  }
