42  template <typename T>
43  inline
44  PBSplineBasisCurve<T>::PBSplineBasisCurve( const DVector<T>& t ) : PCurve<T,3>(20,0,4) {
45 
46  // k is the order and we expand the knot vector with (k-1) knots at start and (k-1) knots at the end
47  // which are only dummy, but we need them to be able to use standard B-splines tools.
48  int k = t.getDim()-1;
49  _t.setDim(3*k-1);
50 
51  for(int i=0; i<k-1; i++)
52  _t[i] = t(0)- T(k-i-1);
53  for(int i=k-1; i < 2*k; i++)
54  _t[i] = t(i-k+1);
55  for(int i=2*k; i<_t.getDim(); i++)
56  _t[i] = t(k)+T(i-2*k+1);
57  }

111  template <typename T>
112  void PBSplineBasisCurve<T>::eval( T t, int d, bool l ) const {
113 
114  this->_p.setDim( d + 1 );
115 
116  DMatrix<T> B;
117  int k = (_t.getDim()+1)/3;
118  int i = 2*(k-1) - EvaluatorStatic<T>::evaluateBSp( B, t, _t, k-1, l);
119 
120  this->_p[0][0] = B[0][i];
121  this->_p[0][1] = t;
122  this->_p[0][2] = 0;
123 
124  if(d>0)
125  {
126  this->_p[1][0] = B[1][i];
127  this->_p[1][1] = 1;
128  this->_p[1][2] = 0;
129  if(d>1)
130  {
131  this->_p[2][0] = B[2][i];
132  this->_p[2][1] = 0;
133  this->_p[2][2] = 0;
134  if(d>2)
135  {
136  this->_p[3][0] = B[3][i];
137  this->_p[3][1] = 0;
138  this->_p[3][2] = 0;
139  if(d>3)
140  {
141  this->_p[4][0] = B[4][i];
142  this->_p[4][1] = 0;
143  this->_p[4][2] = 0;

145  }
