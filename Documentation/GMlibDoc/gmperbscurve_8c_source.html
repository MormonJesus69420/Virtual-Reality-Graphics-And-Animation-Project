288  template <typename T>
289  void PERBSCurve<T>::eval( T t, int d, bool left ) const
290  {
291  int k = EvaluatorStatic<T>::knotIndex(_t, t, 1, left);
292 
293  IndexBsp ii( k, 2, _c.getDim());
294 
295  // Evaluating first Local Curve @ (t-_t[k-1])/(_t[k+1]-_t[k-1])
296  DVector<Vector<T,3>> c0 = _c[ii[0]]->evaluateParent(t, d);
297 
298  // If t == _t[k], the sample is at the knot, set the values to the values of the first local curve.
299  if(std::abs(t - _t[k]) < 1e-5) { this->_p = c0; return; }
300 
301  // Evaluating second Local Curve @ (t-_t[k])/(_t[k+2]-_t[k)
302  this->_p = _c[ii[1]]->evaluateParent(t - (ii[0]<ii[1] ? T(0):this->getParDelta()), d);
303 
304  // Blend c0 and c1
305  Vector<T,3>& B = getB(t, k, d);
306  compBlend( d, B, c0, this->_p );
307  }

312  inline
313  void PERBSCurve<T>::compBlend(int d, const Vector<T,3>& B, DVector<Vector<T,3>>& c0, DVector<Vector<T,3>>& c1) const {
314 
315  c0 -= c1;
316  DVector<T> a(d+1);
317  for( int i = 0; i <= d; i++ )
318  {
319  // Compute the pascal triangle numbers
320  a[i] = 1;
321  for( int j = i-1; j > 0; j-- )
322  a[j] += a[j-1];
323  // Compute the sample position data
324  for( int j = 0; j <= i; j++ )
325  c1[i] += (a[j] * B(j)) * c0[i-j];
326  }
327  }

333  Vector<T,3>& PERBSCurve<T>::getB(T t, int k, int d) const {
334 
335  static Vector<T,3> B;
336 
337  _evaluator->set( _t[k], _t[k+1] - _t[k] );
338  B[0] = 1 - (*_evaluator)(t);
339  switch(d) {
340  case 2: B[2] = - _evaluator->getDer2();
341  case 1: B[1] = - _evaluator->getDer1();
342  }
343  return B;
344  }

463  void PERBSCurve<T>::generateCuLenKnotVector( PCurve<T,3>* g, int n, bool closed ) {
464 
465  const T sp = g->getParStart();
466  const T ep = g->getParEnd();
467 
468  if(closed) {
469  const T dt = (ep-sp)/n;
470  _t.setDim(n+3);
471  for(int i=0; i<n+3; i++ )
472  _t[i] = sp + (i-1) * dt;
473  }
474  else {
475  const T dt = (ep-sp)/(n-1);
476  _t.setDim(n+2);
477  _t[0] = _t[1] = sp; // Set the start knots
478  for(int i = 2; i < n; i++)
479  _t[i] = sp + (i-1)*dt; // Set the "step"-knots
480  _t[n] = _t[n+1] = ep; // Set the end knots
481  }
482  _cl = closed;
483  }

582  template <typename T>
583  void PERBSCurve<T>::preSample( int d ) {
584 
585  // For all partisions,
586  // prepare the array of (start, end) index of sample points influented by a local curve
587  _cp_index.resize(_c.getDim());
588  for(uint i=0; i < _cp_index.size(); i++) {
589  _cp_index[i].setDim(_pre_basis.size());
590  for(int j=0; j < _cp_index[i].getDim(); j++)
591  _cp_index[i][j] = {std::numeric_limits<int>::max(),-1};
592  }
593 
594  for(int i=0; i<_c.getDim(); i++) // For each local curve
595  _c[i]->updateMat();
596 
597  // for hver partisjon -
598  for(uint i=0; i<_pre_basis.size(); i++) { // For each partition
599  this->_visu[i].sur_sphere.reset(); // Reset surounding sphere
600  this->_visu[i].sample_val.resize(_pre_basis[i].size()); // Set number of sample points
601  for(uint j=0; j<_pre_basis[i].size(); j++) { // for hvert sample-punkt i partisjonen
602  multEval( this->_visu[i].sample_val[j], _pre_basis[i][j].B, _pre_basis[i][j].ind, j, i);
603  this->_visu[i].sur_sphere += this->_visu[i].sample_val[j][0];
604  for(uint k=0; k<2; k++ ) {
605  int i_p = _pre_basis[i][j].ind[k];
606  if(int(j) < _cp_index[i_p][i][0]) _cp_index[i_p][i][0] = j;
607  if(int(j) > _cp_index[i_p][i][1]) _cp_index[i_p][i][1] = j;
608  }
609  }
610  }

621  template <typename T>
622  void PERBSCurve<T>::updatSamples() const {
623  if(_local_change.size() > 0) {
624  int k = _local_change.back();
625  std::vector<CpIndex> ind;
626  ind.resize(_pre_basis.size());
627  for(uint i=0; i<ind.size(); i++) // for all partisions
628  if(_cp_index[k][i][1]>=0)
629  ind[i].push_back(_cp_index[k][i]);
630  _c[k]->updateMat();
631  _local_change.pop_back();
632  while(!_local_change.empty()) { // for all changed curves
633  k = _local_change.back();
634  CpIndex lc_ind = _cp_index[k];
635  for(uint i=0; i<ind.size(); i++){ // for all partisions
636  if(_cp_index[k][i][1]>=0) {
637  if(ind[i].getDim()==0)
638  ind[i].push_back(_cp_index[k][i]);
639  else {
640  int r = ind[i].getDim();
641  for(int j=0; j<r; j++) { //for all interval i partision
642  int s = isIntersecting(ind[i][j], lc_ind[i]);
643  if(s==0) ind[i].push_back(lc_ind[i]);
644  else if(s==2) ind[i][j] = lc_ind[i];
645  else if(s==3) ind[i][j][0] = lc_ind[i][0];
646  else if(s==4) ind[i][j][1] = lc_ind[i][1];
647  }
648  }
649  }
650  }
651  _c[k]->updateMat();
652  _local_change.pop_back();
653  }
654  for(uint i=0; i<this->_visu.size(); i++) { // for all partisions
655  for(int j=0; j<ind[i].getDim(); j++){
656  for(int k = ind[i][j][0]; k <= ind[i][j][1]; k++){
657  multEval( this->_visu[i].sample_val[k], _pre_basis[i][k].B, _pre_basis[i][k].ind, k, i);
658  }
659  }
660  }
661  }

681  void PERBSCurve<T>::makePartition( int m ) const {
682 
683  if(_type_local == SUB_CURVE) {
684  this->_visu.resize((_origin->getSampler())->size());
685  _pre_basis.resize((_origin->getSampler())->size());
686  for(uint i=0; i<this->_visu.size(); i++) {
687  this->_visu[i] = _origin->getSampleValues(i);
688  _pre_basis[i].resize(this->_visu[i].size());
689 
690  for(uint j=0; j<_pre_basis[i].size()-1; j++) {
691  int k = EvaluatorStatic<T>::knotIndex(_t, this->_visu[i][j], 1, false);
692  IndexBsp ii( k, 2, _c.getDim());
693  _pre_basis[i][j].B = getB(this->_visu[i][j],k,2);
694  _pre_basis[i][j].ind[0] = ii[0];
695  _pre_basis[i][j].ind[1] = ii[1];
696  }
697  int k = EvaluatorStatic<T>::knotIndex(_t, this->_visu[i][_pre_basis[i].size()-1], 1, true);
698  IndexBsp ii( k, 2, _c.getDim());
699  _pre_basis[i][_pre_basis[i].size()-1].B = getB(this->_visu[i][_pre_basis[i].size()-1],k,2);
700  _pre_basis[i][_pre_basis[i].size()-1].ind[0] = ii[0];
701  _pre_basis[i][_pre_basis[i].size()-1].ind[1] = ii[1];
702  }
703  }
704  else {
705  VisPart<T> pu( _t, 2, _pct );
706  SampNr<T> su( _t, pu, m );
707  this->_visu.resize(su.size());
708  _pre_basis.resize(su.size());
709 
710  for(uint i=0; i<this->_visu.size(); i++) {
711  computeUniformParamVal(this->_visu[i], su[i], _t[pu[2*i]], _t[pu[2*i+1]]);
712  _pre_basis[i].resize(su[i]);
713 
714  for(int j=0; j<su[i]-1; j++) {
715  int k = EvaluatorStatic<T>::knotIndex(_t, this->_visu[i][j], 1, false);
716  IndexBsp ii( k, 2, _c.getDim());
717  _pre_basis[i][j].B = getB(this->_visu[i][j],k,2);
718  _pre_basis[i][j].ind[0] = ii[0];
719  _pre_basis[i][j].ind[1] = ii[1];
720  }
721  int k = EvaluatorStatic<T>::knotIndex(_t, this->_visu[i][su[i]-1], 1, true);
722  IndexBsp ii( k, 2, _c.getDim());
723  _pre_basis[i][su[i]-1].B = getB(this->_visu[i][su[i]-1],k,2);
724  _pre_basis[i][su[i]-1].ind[0] = ii[0];
725  _pre_basis[i][su[i]-1].ind[1] = ii[1];
726  }
727  }

773  template <typename T>
774  inline
775  int PERBSCurve<T>::isIntersecting(const Vector<int,2>& a, const Vector<int,2>& b) const {
776 
777  if(b[1] < (a[0]-1) || b[0] > (a[1]+1)) return 0;
778  if(a[0] <= b[0] && a[1] >= b[1]) return 1;
779  if(b[0] <= a[0] && b[1] >= a[1]) return 2;
780  if(b[0] < a[0]) return 3;
781  if(b[1] > a[1]) return 4;
782  else return -1;
783  }
