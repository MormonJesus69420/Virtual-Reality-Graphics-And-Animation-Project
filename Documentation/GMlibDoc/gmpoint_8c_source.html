181 #define PC_getAng(n) inline\
182  Angle Point<T,n>::getAngle( Point<T, n> v ) const {\
183  Point<T,n> u = (*this);\
184  u /= u.getLength();\
185  v /= v.getLength();\
186  double ang = u*v;\
187  if (ang > 1.0) ang = 1.0;\
188  else if (ang < -1.0) ang = -1.0;\
189  ang = acos(ang);\
190  if (n ==2)\
191  {\
192  double orient = u[0]*v[1]-u[1]*v[0];\
193  if (orient >= 0.0) return Angle(ang);\
194  else return Angle(2*M_PI - ang);\
195  }\
196  else return Angle(ang);\
197  }
