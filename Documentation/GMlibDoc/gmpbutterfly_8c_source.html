214  template <typename T>
215  void PButterfly<T>::eval( T t, int d, bool /*l*/ ) const {
216 
217  this->_p.setDim( d + 1 );
218 
219  const double ct = cos(t);
220  const double st = sin(t);
221  const double st12 = sin(t/12);
222  const double a = exp(ct) - 2*cos(4*t) - pow(st12, 5.0);
223 
224  this->_p[0][0] = _size * T(ct * a);
225  this->_p[0][1] = _size * T(st * a);
226  this->_p[0][2] = fabs(this->_p[0][1])*_flaps;
227 
228  if( this->_dm == GM_DERIVATION_EXPLICIT ) {
229  if( d > 0 ) { // The first derivative
230  const double ct12 = cos(t/12);
231  const double a1 = -exp(ct)*st + 8*sin(4*t) - (5.0/12)*pow(st12,4.0)*ct12;
232  this->_p[1][0] = _size * T( -st*a + ct*a1 );
233  this->_p[1][1] = _size * T( ct*a + st*a1 );
234  if(this->_p[0][1] > 0)
235  this->_p[1][2] = this->_p[1][1]*_flaps;
236  else
237  this->_p[1][2] = -this->_p[1][1]*_flaps;
238  if( d > 1 ) { // The second derivative
239  const double a2 = exp(ct)*st*st - exp(ct) * ct + 8.0*4.0*cos(4*t)
240  - (5.0/12)*((1.0/3)* pow(st12,3.0 )*pow(ct12,2.0)
241  - (1.0/12)* pow(st12,5.0));
242  this->_p[2][0] = _size * T( -ct*a - st*a1 - st*a1 + ct*a2 );
243  this->_p[2][1] = _size * T( -st*a + ct*a1 + ct*a1 + st*a2 );
244  if(this->_p[0][1] > 0)
245  this->_p[2][2] = this->_p[2][1]*_flaps;
246  else
247  this->_p[2][2] = -this->_p[2][1]*_flaps;
248  }
249  }
250  }
251  }
