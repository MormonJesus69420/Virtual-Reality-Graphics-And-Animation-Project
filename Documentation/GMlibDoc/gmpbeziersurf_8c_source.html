51  template <typename T>
52  inline
53  PBezierSurf<T>::PBezierSurf( const DMatrix< Vector<T, 3> >& c, T s_u, T u, T e_u, T s_v, T v, T e_v ) {
54 
55  init();
56 
57  // Generate the control points
58  DMatrix<T> bu, bv;
59  EvaluatorStatic<T>::evaluateBhp( bu, c.getDim1()-1, ( u - s_u ) / ( e_u - s_u ), T(1)/(e_u-s_u) );
60  EvaluatorStatic<T>::evaluateBhp( bv, c.getDim2()-1, ( v - s_v ) / ( e_v - s_v ), T(1)/(e_v-s_v) );
61  setScale(T(1)/(e_u-s_u),T(1)/(e_v-s_v));
62  bu.invert();
63  bv.invert();
64  bv.transpose();
65 
66  _c = bu * (c^bv);
67 
68  for( int i = 0; i < c.getDim1(); i++ )
69  for( int j = 0; j < c.getDim2(); j++ )
70  _c[i][j] -= c(0)(0);
71 
72  this->translateGlobal( c(0)(0) );
73  }

403  void PBezierSurf<T>::internalPreSample( DVector< DMatrix< T > >& p, int m, int d, T scale, T start, T end ) {
404 
405  // compute dt (step in parameter)
406  const T dt = ( end - start ) / T(m-1);
407

413  EvaluatorStatic<T>::evaluateBhp( p[j], d, j*dt, scale );

419  template <typename T>
420  inline
421  void PBezierSurf<T>::multEval(const DMatrix<T>& bu, const DMatrix<T>& bv, int du, int dv) const {
422 
423  int ku = this->getDegreeU()+1;
424  int kv = this->getDegreeV()+1;
425 
426  DMatrix<Vector<T,3>> c(ku, dv+1);
427 
428  // We do these two operations manually here!
429  // bv.transpose();
430  // this->_p = bu * (c^bv);
431 
432  // c= _c^bvT
433  for(int i=0; i< ku; i++)
434  for(int j=0; j<=dv; j++) {
435  c[i][j] = _c(i)(0)*bv(j)(0);
436  for(int k=1; k<kv; k++)
437  c[i][j] += _c(i)(k)*bv(j)(k);
438  }
439  // _p = bu * c
440  for(int i=0; i<=dv; i++)
441  for(int j=0; j<=du; j++) {
442  this->_p[i][j] = bu(i)(0)*c[0][j];
443  for(int k=1; k<ku; k++)
444  this->_p[i][j] += bu(i)(k)*c[k][j];
445  }
446  }
