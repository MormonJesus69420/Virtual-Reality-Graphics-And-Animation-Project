76  template <typename T>
77  void PEightSurface<T>::eval(T u, T v, int d1, int d2, bool /*lu*/, bool /*lv*/ ) const {
78 
79  this->_p.setDim( d1+1, d2+1 );
80 
81 
82  this->_p[0][0][0] = cos(u)*sin(v)*cos(v);
83  this->_p[0][0][1] = sin(u)*sin(v)*cos(v);
84  this->_p[0][0][2] = sin(v);
85 
86 
87  if( this->_dm == GM_DERIVATION_EXPLICIT ) {
88 
89  if(d1) { //u
90  this->_p[1][0][0] = -sin(u)*sin(v)*cos(v);
91  this->_p[1][0][1] = cos(u)*sin(v)*cos(v);
92  this->_p[1][0][2] = T(0);
93  }
94  if(d1>1) { //uu
95  this->_p[2][0][0] = -cos(u)*sin(v)*cos(v);
96  this->_p[2][0][1] = -sin(u)*sin(v)*cos(v);
97  this->_p[2][0][2] = T(0);
98  }
99  if(d2) { //v
100  this->_p[0][1][0] = cos(u)*(T(2)*cos(v)*cos(v)-T(1));
101  this->_p[0][1][1] = sin(u)*(T(2)*cos(v)*cos(v)-T(1));
102  this->_p[0][1][2] = cos(v);
103  }
104  if(d2>1) { //vv
105  this->_p[0][2][0] = -T(4)*cos(u)*sin(v)*cos(v);
106  this->_p[0][2][1] = -T(4)*sin(u)*sin(v)*cos(v);
107  this->_p[0][2][2] = -sin(v);
108  }
109  if(d1 && d2) { //uv
110  this->_p[1][1][0] = -sin(u)*(T(2)*cos(v)*cos(v)-T(1));
111  this->_p[1][1][1] = cos(u)*(T(2)*cos(v)*cos(v)-T(1));
112  this->_p[1][1][2] = T(0);
113  }
114  if(d1>1 && d2) { //uuv
115  this->_p[2][1][0] = -cos(u)*(T(2)*cos(v)*cos(v)-T(1));
116  this->_p[2][1][1] = -sin(u)*(T(2)*cos(v)*cos(v)-T(1));
117  this->_p[2][1][2] = T(0);
118  }
119  if(d1 && d2>1) { //uvv
120  this->_p[1][2][0] = T(4)*sin(u)*sin(v)*cos(v);
121  this->_p[1][2][1] = -T(4)*cos(u)*sin(v)*cos(v);
122  this->_p[1][2][2] = T(0);
123  }
124  if(d1>1 && d2>1) { //uuvv
125  this->_p[2][2][0] = T(4)*cos(u)*sin(v)*cos(v);
126  this->_p[2][2][1] = T(4)*sin(u)*sin(v)*cos(v);
127  this->_p[2][2][2] = T(0);
128  }
129  }
130  }
