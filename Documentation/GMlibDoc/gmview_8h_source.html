365  void ViewBorder::move(int x, int y) {
366 
367  int l_size = _child_lt.getSize(_vertical);
368  int r_size = _child_rb.getSize(_vertical);
369 
370  if(_vertical) {
371 
372  l_size = _x1-l_size;
373  r_size += _x2;
374  if((x > l_size && x < _x1)||(x > _x2 && x < r_size))
375  _displacement = (double)(x-l_size)/(r_size-l_size);
376  }
377  else
378  {
379  l_size = _y1-l_size;
380  r_size += _y2;
381  if((y > l_size && y < _y1)||(y > _y2 && y < r_size))
382  _displacement = (double)(y-l_size)/(r_size-l_size);
383  }
384  }
