74  template <typename T>
75  void PTrianguloidTrefoil<T>::eval(T u, T v, int d1, int d2, bool /*lu*/, bool /*lv*/ ) const {
76 
77  this->_p.setDim( d1+1, d2+1 );
78 
79 
80  T su = sin(u);
81  T cu = cos(u);
82  T s2u = sin(2*u);
83  T s3u = sin(3*u);
84  T c2u = cos(2*u);
85  T c3u = cos(3*u);
86 
87  T sv = sin(v);
88  T cv = cos(v);
89  T cvp2 = cv+2;
90  T sv23 = sin(v+M_2PI_3);
91  T cv23 = cos(v+M_2PI_3);
92  T cv23p2 = cv23+2;
93 
94  DVector< Point<T,3> > fu( 1 );
95  DVector< Point<T,3> > fv( 1 );
96 
97  if( this->_dm == GM_DERIVATION_EXPLICIT ) {
98 
99  fu.setDim( d1+1 );
100  fv.setDim( d2+1 );
101 
102  switch(d1)
103  {
104  case 3: fu[3] = Point<T,3>(-54*c3u,-2*(cu+16*c2u), su-16*s2u);
105  case 2: fu[2] = Point<T,3>(-18*s3u,-2*(su+8*s2u), -cu+8*c2u);
106  case 1: fu[1] = Point<T,3>( 6*c3u, 2*(cu+4*c2u), -su+4*s2u);
107  default: fu[0] = Point<T,3>( 2*s3u, 2*(su+2*s2u), cu-2*c2u);
108  }
109 
110  switch(d2)
111  {
112  case 3: fv[3] = Point<T,3>(6*(sv*cv/pow(cvp2,3)+pow(sv,3)/pow(cvp2,4))+sv/(cvp2*cvp2),
113  6*(sv23*cv23/pow(cv23p2,3)+pow(sv23,3)/pow(cv23p2,4))+sv23/(cv23p2*cv23p2),
114  (sv*cv23p2+3*cv*sv23+3*sv*cv23+sv23*cvp2)/4);
115  case 2: fv[2] = Point<T,3>(2*sv*sv/pow(cvp2,3)+cv/(cvp2*cvp2),
116  2*(sv23*sv23/pow(cv23p2,3))+cv23/(cv23p2*cv23p2),
117  -(cv*cv23p2-2*sv*sv23+cvp2*cv23)/4);
118  case 1: fv[1] = Point<T,3>( sv/(cvp2*cvp2), sv23/(cv23p2*cv23p2), -(sv*cv23p2+cvp2*sv23)/4);
119  default: fv[0] = Point<T,3>( 1/cvp2, 1/cv23p2, cvp2*cv23p2/4);
120  }
121  }
122  else {
123 
124  fu[0] = Point<T,3>( 2*s3u, 2*(su+2*s2u), cu-2*c2u);
125  fv[0] = Point<T,3>( 1/cvp2, 1/cv23p2, cvp2*cv23p2/4);
126  }
127 
128  for(int i=0; i<fu.getDim();i++)
129  for(int j=0; j<fv.getDim(); j++)
130  this->_p[i][j] = fu[i]%fv[j];
131  }
