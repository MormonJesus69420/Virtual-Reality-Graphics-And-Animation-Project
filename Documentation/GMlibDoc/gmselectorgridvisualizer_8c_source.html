162  inline
163  void SelectorGridVisualizer<T>::_makeLines(bool cl) // Fill IBO
164  {
165  // Create the indice buffer
166  int n = 2*(_c.getDim()-1);
167  _no_indices = n + (cl? 2 : 0);
168  std::vector<GLushort> indices(_no_indices);
169  for(int i=0, j=0; i < n; i+=2, j++) {
170  indices[i] = j;
171  indices[i+1] = j + 1;
172  }
173  if(cl) {
174  indices[n] = 0;
175  indices[n+1] = _c.getDim() - 1;
176  }
177 
178  _ibo.bufferData( _no_indices * sizeof(GLushort), indices.data(), GL_DYNAMIC_DRAW );
179  }

182  template <typename T>
183  inline
184  void SelectorGridVisualizer<T>::_makeTriangs() // Fill IBO
185  {
186  // Create the indice buffer
187  int m = (sqrt(8*_c.getDim()+1)-1)/2;
188  DVector<GLushort> indices(_no_indices = 3*(m*(m-1)));
189  for( int k=0, i=0; i < m-1; i++) {
190  int o1 = (i*(i+1))/2;
191  int o2 = ((i+1)*(i+2))/2;
192  for( int j = 0, l = o2-o1; j < l; j++ ) {
193  indices[k++] = o1+j; indices[k++] = o2+j;
194  indices[k++] = o1+j; indices[k++] = o2+j+1;
195  indices[k++] = o2+j; indices[k++] = o2+j+1;
196  }
197  }
198 
199  _ibo.bufferData( _no_indices * sizeof(GLushort), indices.getPtr(), GL_DYNAMIC_DRAW );
200  }

202  template <typename T>
203  void SelectorGridVisualizer<T>::_makeQuads(int m1, int m2, bool cu, bool cv) {
204  // Fill IBO
205 
206  int n1 = m1 - (cu?0:1);
207  int n2 = m2 - (cv?0:1);
208  _no_indices = (m1 * n2 + n1 * m2) * 2;
209  DVector<GLushort> indices(_no_indices); // Create the indice buffer
210  GLushort *iptr = indices.getPtr();
211 
212  // "Lines" in v dir
213  for( int i = 0; i < m1; i++ ) {
214  for( int j = 0; j < m2-1; j++ ) {
215  *iptr++ = i * m2 + j;
216  *iptr++ = i * m2 + j + 1;
217  }
218  if(cv) {
219  *iptr++ = i * m2 + m2-1;
220  *iptr++ = i * m2;
221  }
222  }
223 
224  // "Lines" in u dir
225  for( int j = 0; j < m2; j++ ) {
226  for( int i = 0; i < m1-1; i++ ) {
227  *iptr++ = i * m2 + j;
228  *iptr++ = (i+1) * m2 + j;
229  }
230  if(cu) {
231  *iptr++ = (m1-1)*m2+j;
232  *iptr++ = j;
233  }
234  }
235 
236  _ibo.bufferData( _no_indices * sizeof(GLushort), indices.getPtr(), GL_DYNAMIC_DRAW );
237  }
