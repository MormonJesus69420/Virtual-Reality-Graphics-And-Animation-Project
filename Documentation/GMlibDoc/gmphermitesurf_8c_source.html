102  void PHermiteSurf<T>::eval( T u, T v, int d1, int d2, bool /*lu*/, bool /*lv*/ ) const {
103  this->_p.setDim( d1+1, d2+1 );
104 
105  DMatrix<T> hu, hv;
106  EvaluatorStatic<T>::evaluateH3d( hu, d1, u);
107  EvaluatorStatic<T>::evaluateH3d( hv, d2, v);
108 
109  for( int i = 0; i < hu.getDim1(); i++ )
110  for( int j = 0; j < hv.getDim1(); j++ )
111  this->_p[i][j] = hu(i) * ( _m^hv(j) );
112  }
