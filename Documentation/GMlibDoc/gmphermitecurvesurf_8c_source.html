105  m[j][i][0][1] = h[1][0]*v1[j][0] + h[1][1]*v2[j][0] + h[1][2]*v1[j][1] + h[1][3]*v2[j][1];

107  m[j][i][1][1] = h[1][0]*v1[j][2] + h[1][1]*v2[j][2] + h[1][2]*v1[j][3] + h[1][3]*v2[j][3];

110  m[j][i][2][0] = h[0][0]*v1[j][4] + h[0][1]*v2[j][4] + h[0][2]*v1[j][5] + h[0][3]*v2[j][5];

165 template <typename T>
166 void PHermiteCurveSurf<T>::eval(T u, T v, int d1,int d2, bool /*lu*/, bool /*lv*/ ) const {
167  if(_swap_par)
168  {
169  std::swap(u,v);
170  std::swap(d1,d2);
171  }
172 
173  DMatrix<Vector<T,3> > c(_cu.getSize(),d2+1);
174 
175  for(int i=0; i< _cu.getSize(); i++)
176  {
177  c[i] = _cu(i)->evaluateParent(v,d2);
178 
179 // cout << "c["<< i << "]=" << c[i] << endl;
180  }
181 
182  DMatrix< T > hp; // Storing the Hermite Polynomials
183  EvaluatorStatic<T>::evaluateH3d( hp, d1, u);
184 
185  DMatrix< Vector<T,3> > p = hp * c;
186 
187  if(_swap_par) this->_p = p.transpose();
188  else this->_p = p;
189 }
