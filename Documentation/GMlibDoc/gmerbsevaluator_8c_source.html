146  template <typename T>
147  inline
148  T ERBSEvaluator<T>::getF2( T t ) {
149  if( t == _lambda || t < 2.3e-308 || t == 1 )
150  return T(0);
151 
152  T h = (t-1/(1+_gamma)) * std::fabs(t-_lambda)/(t * (1-t));
153  if( t < _lambda )
154  h -= T(1);
155  else
156  h += T(1);
157  h *= -this->getScale()*_alpha*_beta*(1+_gamma)/std::pow(t*std::pow(1-t, _gamma), _alpha);
158 
159  if( (1+_gamma)*_alpha < 1 ) {
160 
161  const T g = std::pow(std::fabs(t-_lambda), 1-(1+_gamma)*_alpha)/h;
162  if( g < 2.3e-308 )
163  return 0.0;
164  else
165  return 1 / g;
166  }
167  else if( (1+_gamma)*_alpha > 1 )
168  return h * std::pow(std::fabs(t-_lambda), (1+_gamma)*_alpha-1);
169  else
170  return h;
171  }
