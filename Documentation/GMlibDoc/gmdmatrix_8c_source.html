111  void DMatrix<T>::increaseDim(int i, int j, T val, bool h_end, bool v_end) {
112 
113  if(i>0)
114  {
115  int k, m = _n + i;
116  DVector<T>* tmp = (m>4 ? new DVector<T>[m] : _init);
117  if(v_end)
118  {
119  if(m>4) for (k=0; k<_n; k++)
120  {
121  tmp[k] = _p[k];
122  tmp[k].increaseDim(j,val,h_end);
123  }
124  for (k=_n; k<m; k++)
125  {
126  tmp[k].setDim(j);
127  tmp[k].clear(val);
128  }
129  }
130  else
131  {
132  for (k=m-1; k>=i; k--)
133  {
134  tmp[k] = _p[k-i];
135  tmp[k].increaseDim(j,val,h_end);
136  }
137  for (k=0; k<i; k++)
138  {
139  tmp[k].setDim(j);
140  tmp[k].clear(val);
141  }
142  }
143  if(_p != _init) delete [] _p;
144  _p = tmp;
145  _n = m;
146  }

156  DMatrix<T>& DMatrix<T>::invert() {
157 
158  #ifdef GM_INTEL // This requires MKL from Intel (mkl_c.lib),
159  // or BLAS and Lapack binaries in a compatible format
160  if(getDim1()==getDim2() && getDim1()>1)
161  {
162  int nk=getDim1();
163  Array<float> aa;
164  aa.setSize(nk*nk);
165  for(int i=0; i<nk; i++) // daft cast
166  for(int j=0; j<nk; j++) aa[i+nk*j] = (float) (*this)[i][j];
167 
168  Array<float> work; work.setSize(nk*32); // temporary work array
169  int wsize=nk*32; // work array size - should be optimized
170  Array<int> ipiv; ipiv.setSize(nk); // pivot table (result), size max(1,mmm,nnn)
171  int info=0; // error message, i=info>0 means that a[i][i]=0 ): singular,
172  int mmm=nk; // dimentsion
173  int nnn=nk; // dimentsion i.e. a is mxn matrix
174  int lda=nk; // leading dimentsion
175  // float *a // input array
176 
177  // LAPACK When calling LAPACK routines from C-language programs, make sure that you follow Fortran rules:
178  // Pass variables by 'address' as opposed to pass by 'value'. Be sure to store your data Fortran-style,
179  // i.e. data stored column-major rather than row-major order
180  // Project neeeds mkl_c.lib and mkl_lapack32.h
181 
182  sgetrf(&mmm, &nnn, aa.ptr(), &lda, ipiv.ptr(), &info); // using Lapack LU-fact a is overwritten by LU
183  sgetri(&mmm, aa.ptr(), &lda, ipiv.ptr(), work.ptr(), &wsize, &info); // a should now contain the inverse
184 
185  // if(info==0)
186  for(int i=0; i<nk; i++)
187  for(int j=0; j<nk; j++) (*this)[i][j]=(T) aa[i+j*nk];
188  }
189  #else // gauss-jordan implementation from Numerical recipes
190  // Ordinary LU-decomp used.
191  DMatrix<T> a=(*this);
192  Array<int> indx(a.getDim2());
193 
194  const double TINY=1.0e-20;
195  int i,imax,j,k;
196  double big,dum,sum,temp;
197 
198  int n=a.getDim1(); //nrows
199  Array<double> vv(n);
200  double d=1.0;
201 #ifdef _OPENMP
202  #pragma omp parallel for
203 #endif
204  for (i=0;i<n;i++)
205  {
206  big=0.0;
207  for (j=0;j<n;j++) if ((temp=std::abs(a[i][j])) > big) big=temp;
208 // if (big == 0.0) return (*this); //nrerror("Singular matrix in routine ludcmp");
209  vv[i]=1.0/big;
210  }
211 #ifdef _OPENMP
212  #pragma omp parallel for
213 #endif
214  for (j=0;j<n;j++)
215  {
216 #ifdef _OPENMP
217  #pragma omp parallel for
218 #endif
219  for (i=0;i<j;i++)
220  {
221  sum=a[i][j];
222  for (k=0;k<i;k++) sum -= a[i][k]*a[k][j];
223  a[i][j]=sum;
224  }
225  big=0.0;
226 #ifdef _OPENMP
227  #pragma omp parallel for
228 #endif
229  for (i=j;i<n;i++)
230  {
231  sum=a[i][j];
232  for (k=0;k<j;k++) sum -= a[i][k]*a[k][j];
233  a[i][j]=sum;
234  if ((dum=vv[i]*std::abs(sum)) >= big) { big=dum; imax=i; }
235  }
236  if (j != imax)
237  {
238 #ifdef _OPENMP
239  #pragma omp parallel for
240 #endif
241  for (k=0;k<n;k++)
242  {
243  dum=a[imax][k];
244  a[imax][k]=a[j][k];
245  a[j][k]=dum;
246  }
247  d = -d;
248  vv[imax]=vv[j];
249  }
250  indx[j]=imax;
251  if (a[j][j] == 0.0) a[j][j]=TINY;
252  if (j != n-1)
253  {
254  dum=1.0/(a[j][j]);
255  for (i=j+1;i<n;i++) a[i][j] *= dum;
256  }
257  } // LU-decomp. finished, stored in a[][]
258 
259  DVector<T> b(a.getDim1()); // LU-back subst. begins
260 #ifdef _OPENMP
261  #pragma omp parallel for
262 #endif
263  for(int cols=0; cols<a.getDim2(); cols++)
264  {
265  int i2,ii=0,ip,j2;
266  for(i2=0; i2<getDim1(); i2++) if(i2==cols) b[i2]=T(1.0); else b[i2]=T(0.0);
267  T sum2;
268  int n2=a.getDim1();
269 #ifdef _OPENMP
270  #pragma omp parallel for
271 #endif
272  for (i2=0;i2<n2;i2++)
273  {
274  ip=indx[i2];
275  sum2=b[ip];
276  b[ip]=b[i2];
277  if (ii != 0)
278  for (j2=ii-1;j2<i2;j2++) sum2 -= a[i2][j2]*b[j2];
279  else if (sum2 != 0.0) ii=i2+1;
280  b[i2]=sum2;
281  }
282 #ifdef _OPENMP
283  #pragma omp parallel for
284 #endif
285  for (i2=n2-1;i2>=0;i2--)
286  {
287  sum2=b[i2];
288  for (j2=i2+1;j2<n2;j2++) sum2 -= a[i2][j2]*b[j2];
289  b[i2]=sum2/a[i2][i2];
290  }
291  // LU-back subst. finished,
292 #ifdef _OPENMP
293  #pragma omp parallel for
294 #endif
295  for(i2=0; i2<getDim1(); i2++) // inverse stored in this, a and b is disappearing?
296  (*this)[i2][cols]=b[i2];
297  }
298  #endif
299  return (*this);
300  }
