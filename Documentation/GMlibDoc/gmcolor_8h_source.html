423  inline
424  void Color::toHSV() {
425 
426  double r=getRedC(),g=getGreenC(),b=getBlueC();
427  double h,s,v,mx,mn;
428  mx = std::max<double>( std::max<double>(r,g), b );
429  mn = std::min<double>( std::min<double>(r,g), b );
430  v=mx;
431  if(mx > 1e-6) s = (mx-mn)/mx;
432  else s = 0.0;
433  if(s == 0.0) h = 0.0;
434  else
435  {
436  double rc = (mx-r)/(mx-mn);
437  double gc = (mx-g)/(mx-mn);
438  double bc = (mx-b)/(mx-mn);
439  if(std::fabs(r-mx) < 1e-7) h = bc-gc;
440  else if(std::fabs(g - mx) < 1e-7) h = rc-bc + 2.0;
441  else h = gc-rc + 4.0; // else if(std::abs(b -mx) < 1e-7)
442  h *= 60.0;
443  if(h<0.0) h += 360.0;
444  h /= 360.0; // 360 degree is skaled to 1.
445  }
446  setRed(h); setGreen(s); setBlue(v);
447  }

450  inline
451  void Color::toRGB() {
452 
453  double h=getRedC(),s=getGreenC(),v=getBlueC();
454  double r,g,b;
455  h *= 360.0;
456  if(s < 0.000001)
457  r = g = b = v;
458  else
459  {
460  if(h > 359.999999) h = 0.0;
461  h /= 60.0;
462  int i = static_cast<int>(h);
463  double f = h - double(i);
464  double p = v*(1-s);
465  double q = v*(1-s*f);
466  double t = v*(1-s*(1-f));
467  switch(i)
468  {
469  case 0: r=v; g=t; b=p; break;
470  case 1: r=q; g=v; b=p; break;
471  case 2: r=p; g=v; b=t; break;
472  case 3: r=p; g=q; b=v; break;
473  case 4: r=t; g=p; b=v; break;
474  default: r=v; g=p; b=q; break; // i == 5
475  }
476  }
477  setRed( r ); setGreen( g ); setBlue( b );
478  }
