342  void computeHelp2( std::vector<double[3]>& b, const G& t, bool closed, int k ) {
343 
344  int s = k-1;
345 
346  std::vector<double[2]> a(k);
347  for(int i=0; i<k; i++) {
348  a[i][0] = t[i+1]-t[i];
349  a[i][1] = a[i][0]*a[i][0];
350  }
351 
352  if(closed) {
353  double d = a[s][1]*a[0][0] + a[s][0]*a[0][1];
354  b[0][0] = b[k][0] = -a[0][1]/d;
355  b[0][1] = b[k][1] = (a[0][1]-a[s][1])/d;
356  b[0][2] = b[k][2] = a[s][1]/d;
357  }else {
358  double d = a[0][1]*a[1][0] + a[0][0]*a[1][1];
359  b[0][0] = -a[1][0]*(2*a[0][0]+a[1][0])/d;
360  b[0][1] = (a[0][0]+a[1][0])*(a[0][0]+a[1][0])/d;
361  b[0][2] = -a[0][1]/d;
362  d = a[s-1][1]*a[s][0] + a[s-1][0]*a[s][1];
363  b[k][0] = a[s][1]/d;
364  b[k][1] = -(a[s-1][0]+a[s][0])*(a[s-1][0]+a[s][0])/d;
365  b[k][2] = a[s-1][0]*(2*a[s][0]+a[s-1][0])/d;
366  }
367  for(int i=1; i<k; i++) {
368  double d = a[i-1][1]*a[i][0] + a[i-1][0]*a[i][1];
369  b[i][0] = -a[i][1]/d;
370  b[i][1] = (a[i][1]-a[i-1][1])/d;
371  b[i][2] = a[i-1][1]/d;
372  }

386  double relationCK(const T& p1, const T& p2, const T& p3) {
387  T v0 = p3-p1;
388  T v1 = p2-p1;
389  T v2 = p3-p2;
390 
391  double a1 = v1*v1;
392  double a2 = v2*v2;
393  double a3 = v1*v2;
394  double a4 = a3/a1;
395  double a5 = a2-a3*a4;
396 
397  if(a5 < 1e-5) return 1;
398 
399  double a6 = v0*v2;
400  return acos(a3/sqrt(a1*a2))*sqrt((a1+a6*a6/a5)/(v0*v0));
