105  template <typename T>
106  void PTriangleColorVisualizer<T>::replot(DVector<DMatrix<Vector<T, 3> > > &p, int m) {
107 
108  PTriangleVisualizer<T>::replot( p, m );
109 
110  // Allocate GPU memory
111  glBindBuffer( GL_ARRAY_BUFFER, _vbo_color );
112  glBufferData( GL_ARRAY_BUFFER, PTRIANGLEVERTEXATTRIB_SIZE * this->_no_vertices, 0x0, GL_DYNAMIC_DRAW );
113 
114 
115  // Fill GPU memory
116  PTriangleVertexAttributes *ptr = (PTriangleVertexAttributes*)glMapBuffer( GL_ARRAY_BUFFER, GL_WRITE_ONLY );
117 
118 
119  if(ptr) {
120 
121  DVector< Color > colors( p.getDim() ); {
122 
123  Color c1 = _c1;
124  Color c2 = _c2;
125  Color c3 = _c3;
126  if( _color_mode == GM_PTRIANGLE_COLORVISUALIZER_HSV ) {
127 
128  c1.toHSV();
129  c2.toHSV();
130  c3.toHSV();
131  }
132 
133  float u, v, w, du = 1.0f / (m-1);
134  int i,j,k;
135  for(k=0,i=0; i<m; i++) {
136  for(j=0;j<=i;j++) {
137  v = j*du;
138  u = (i-j)*du;
139  w = 1.0f-u-v;
140 
141  Color c = u * c1 + v * c2 + w * c3;
142  if( _color_mode == GM_PTRIANGLE_COLORVISUALIZER_HSV )
143  c.toRGB();
144  colors[k++] = c;
145  }
146  }
147  }
148 
149  for( int i = 0; i < this->_m; i++ ) {
150 
151  int o1 = 0.5 * i * (i+1);
152  int o2 = 0.5 * (i+1) * (i+2);
153  int o = i*(i+2);
154  int is = 2 * i + 3;
155 
156  for( int j = 0; j < is/2; j++ ) {
157 
158  int idx1 = (o+j*2);
159  int idxp1 = o2 +j;
160 
161  ptr[idx1].r = colors[idxp1].getRedC();
162  ptr[idx1].g = colors[idxp1].getGreenC();
163  ptr[idx1].b = colors[idxp1].getBlueC();
164  ptr[idx1].a = colors[idxp1].getAlphaC();
165 
166 
167  int idx2 = (o+j*2+1);
168  int idxp2 = o1 +j;
169 
170  ptr[idx2].r = colors[idxp2].getRedC();
171  ptr[idx2].g = colors[idxp2].getGreenC();
172  ptr[idx2].b = colors[idxp2].getBlueC();
173  ptr[idx2].a = colors[idxp2].getAlphaC();
174  }
175 
176  ptr[o+is-1].r = colors[o1+is-1].getRedC();
177  ptr[o+is-1].g = colors[o1+is-1].getGreenC();
178  ptr[o+is-1].b = colors[o1+is-1].getBlueC();
179  ptr[o+is-1].a = colors[o1+is-1].getAlphaC();
180  }
181 
182  }
183 
184  glUnmapBuffer( GL_ARRAY_BUFFER );
185  glBindBuffer( GL_ARRAY_BUFFER, 0x0 );
186  }
