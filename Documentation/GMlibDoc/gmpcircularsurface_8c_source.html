96  template <typename T>
97  void PCircularSurface<T>::eval(T u, T v, int d1, int d2, bool /*lu*/, bool /*lv*/ ) const {
98 
99  this->_p.setDim( d1+1, d2+1 );
100 
101  this->_p[0][0][0] = u * cos( v );
102  this->_p[0][0][1] = u * sin( v );
103  this->_p[0][0][2] = T(0);
104 
105 
106  if( this->_dm == GM_DERIVATION_EXPLICIT ) {
107 
108  if(d1) {
109  this->_p[1][0][0] = cos(v); // S_u
110  this->_p[1][0][1] = sin(v);
111  this->_p[1][0][2] = T(0);
112  }
113  if(d2) {
114  this->_p[0][1][0] = u * -sin(v); // S_v
115  this->_p[0][1][1] = u * cos(v);
116  this->_p[0][1][2] = T(0);
117  }
118  if(d1 && d2) {
119  this->_p[1][1][0] = -sin(v); // S_uv
120  this->_p[1][1][1] = cos(v);
121  this->_p[1][1][2] = T(0);
122  }
123  if(d1>1) {
124  this->_p[2][0][0] = T(0); // S_uu
125  this->_p[2][0][1] = T(0);
126  this->_p[2][0][2] = T(0);
127  }
128  if(d2>1) {
129  this->_p[0][2][0] = u * -cos(v); // S_vv
130  this->_p[0][2][1] = u * -sin(v);
131  this->_p[0][2][2] = T(0);
132  }
133  if(d1>1 && d2) {
134  this->_p[2][1][0] = T(0); // S_uuv
135  this->_p[2][1][1] = T(0);
136  this->_p[2][1][2] = T(0);
137  }
138  if(d1 && d2>1) {
139  this->_p[1][2][0] = -cos(v); // S_uvv
140  this->_p[1][2][1] = -sin(v);
141  this->_p[1][2][2] = T(0);
142  }
143  if(d1>1 && d2>1) {
144  this->_p[2][2][0] = T(0); // S_uuvv
145  this->_p[2][2][1] = T(0);
146  this->_p[2][2][2] = T(0);
147 
148  }
149  }
150  }
