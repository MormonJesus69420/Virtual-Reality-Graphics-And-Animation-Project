106  template <typename T, int n>
107  inline
108  void PSurfDerivativesVisualizer<T,n>::replot(
109  const DMatrix< DMatrix< Vector<T, n> > >& p,
110  const DMatrix< Vector<T,3> >& normals,
111  int m1, int m2, int d1, int d2,
112  bool closed_u, bool closed_v
113  ) {
114  GM_UNUSED(normals)
115  GM_UNUSED(m1) GM_UNUSED(m2)
116  GM_UNUSED(d1) GM_UNUSED(d2)
117  GM_UNUSED(closed_u) GM_UNUSED(closed_v)
118 
119 
120  assert( p(0)(0).getDim1() >= (_u+1) && p(0)(0).getDim2() >= (_v+1) );
121 
122 
123  int no_derivatives = p.getDim1() * p.getDim2();
124  _no_elements = no_derivatives * 2;
125 
126  _vbo.bufferData(no_derivatives * 2 * 3 * sizeof(GLfloat), 0x0, GL_DYNAMIC_DRAW);
127 
128  Vector<GLfloat,3>* ptr = _vbo.mapBuffer< Vector<GLfloat,3> >();
129  if( ptr ) {
130 
131  switch( _mode ) {
132  case GM_SURF_DERIVATIVESVISUALIZER_RELATIVE: {
133 
134  for( int i = 0; i < p.getDim1(); i++ )
135  for( int j = 0; j < p.getDim2(); j++ ) {
136 
137  const Point<T,3> &pos = p(i)(j)(0)(0);
138  const Vector<T,3> &v = p(i)(j)(_u)(_v) * _size;
139 
140  *(ptr++) = pos;
141  *(ptr++) = pos + v;
142  }
143  }
144  break;
145 
146  case GM_SURF_DERIVATIVESVISUALIZER_ABSOLUTE: {
147 
148  for( int i = 0; i < p.getDim1(); i++ )
149  for( int j = 0; j < p.getDim2(); j++ ) {
150 
151  const Point<T,3> &pos = p(i)(j)(0)(0);
152  const Vector<T,3> &v = p(i)(j)(_u)(_v).getNormalized() * _size;
153  *(ptr++) = pos;
154  *(ptr++) = pos + v;
155  }
156  }
157  break;
158 
159  case GM_SURF_DERIVATIVESVISUALIZER_NORMALIZED: {
160 
161  for( int i = 0; i < p.getDim1(); i++ )
162  for( int j = 0; j < p.getDim2(); j++ ) {
163 
164  const Point<T,3> &pos = p(i)(j)(0)(0);
165  const UnitVector<T,3> &v = p(i)(j)(_u)(_v);
166  *(ptr++) = pos;
167  *(ptr++) = pos + v;
168  }
169  }
170  break;
171  }
172  }
173 
174  _vbo.unmapBuffer();
175  }
