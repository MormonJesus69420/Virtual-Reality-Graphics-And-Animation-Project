192  int SceneObject::prepare(Array<HqMatrix<float,3> >& mat, Scene* s, SceneObject* parent) {
193 
194  int nr = 1;
195 
196  _scene = s;
197  _parent = parent;
198 
199  mat.push();
200  _matrix_scene_inv = _matrix_scene = mat.back();
201  _matrix_scene_inv.invertOrthoNormal();
202 
203  _present = mat.back() = mat.back() * getMatrix();
204  _global_total_sphere = _global_sphere = _present * _sphere;
205 
206  if(_scale.isActive()) {
207  _global_sphere *= _scale.getMax();
208  _global_total_sphere *= _scale.getMax();
209  }
210 
211  for( int i = 0; i < _children.getSize(); i++ ) {
212  nr += _children[i]->prepare(mat,s,this);
213  _global_total_sphere += _children[i]->getSurroundingSphere();
214  }
215  mat.pop();
216  return nr;
217  }

423  void SceneObject::move(char c, double d, bool propagate) {
424 
425  double dir_length = double();
426  Point<float,3> lock_pos;
427 
428  if(_locked)
429  {
430  lock_pos = getSceneLockPos();
431  dir_length = ( lock_pos - _pos ).getLength();
432  }
433 
434  switch (c) {
435  case 'u':
436  _pos += d*_up;
437  break;
438  case 'd':
439  _pos -= d*_up;
440  break;
441  case 'l':
442  _pos += d*_side;
443  break;
444  case 'r':
445  _pos -= d*_side;
446  break;
447  default:
448  break;
449  }
450 
451  if(_locked)
452  {
453  updateOrientation(lock_pos);
454  _pos = lock_pos - dir_length * _dir;
455  }
456 
457  basisChange(_side, _up, _dir, _pos);
458 
459  if(propagate && _parent )
460  _parent->edit( this );

557  void SceneObject::rotateParent(Angle a, const Vector<float,3>& rot_axel, bool propagate) {
558 
559  if(_locked)
560  {
561  Point<float,3> lock_pos = getSceneLockPos();
562  double dir_length = (lock_pos - _pos).getLength();
563  Vector<float,3> t,ra = rot_axel;
564  double dd= a.getRad()/M_PI_4;
565  while(dd > 1.0)
566  {
567  t = ra^_dir;
568  t.setLength(dir_length);
569  move(Vector<float,2>(t*_side,t*_up));
570  dd -= 1.0;
571  }
572  t = ra^_dir;
573  t.setLength(tan(dd*M_PI_4)*dir_length);
574  move(Vector<float,2>(t*_side,t*_up));
575  }
576  else
577  {
578  HqMatrix<float,3> m( a, rot_axel );
579  _up = m * _up;
580  _dir = m * _dir;
581  _side = m * _side;
582  }
583  basisChange(_side, _up, _dir, _pos);
584 
585  if(propagate && _parent )
586  _parent->edit( this );
587  }
