66  template <typename T>
67  inline
68  void PSweepSurf<T>::makeOmega(int n, T omega_0) {
69  T start = _spine->getParStart();
70  T delta = _spine->getParDelta()/n;
71  T del_2 = delta/2;
72 
73  DVector<Vector<T,3> > h = _spine->evaluate(start,3);
74  Vector<T,3> h12 = h[1]^h[2];
75  T h1_2 = h[1]*h[1];
76  T h12_2 = h12*h12;
77  T tau = (h12*h[3])/h12_2;
78  T prv = omega_0;
79 
80  _omega.setDim(n);
81  _omega[0] = prv;
82 
83  for(int i=1; i<=n; i++)
84  {
85  h = _spine->evaluate(start + delta*i,3);
86  h12 = h[1]^h[2];
87  h1_2 = h[1]*h[1];
88  h12_2 = h12*h12;
89  tau = (h12*h[3])/h12_2;
90  T k = tau*sqrt(h1_2);
91  _omega[i] = _omega[i-1] - del_2*(k+prv);
92  prv = k;
93  }
94  }

97  template <typename T>
98  inline
99  T PSweepSurf<T>::getOmega(T t) const {
100  int n = _omega.getDim()-1;
101  T d = _spine->getParDelta();
102  T u = t - _spine->getParStart();
103  int i = n*u/d;
104  T r = (n*u/d) - T(i);
105 
106  return _omega(i) + r*(_omega(i+1) - _omega(i));
107  }

133  template <typename T>
134  void PSweepSurf<T>::eval(T u, T v, int d1,int d2, bool /*lu*/, bool /*lv*/ ) const {
135 
136  this->_p.setDim(d1+1,d2+1);
137  int i;
138 
139  DVector<Vector<T,3> > h = _spine->evaluate(v,3);
140 
141  Vector<T,3> h12 = h[1]^h[2];
142 
143  T h1_2 = h[1]*h[1];
144  T h1_1 = sqrt(h1_2);
145  T h12_2 = h12*h12;
146  T h12_1 = sqrt(h12_2);
147 
148  T kappa = h12_1/(h1_2*h1_1);
149  T tau = (h12*h[3])/h12_2;
150 
151  Vector<T,3> _T = h[1]/h1_1;
152  Vector<T,3> _B = h12/h12_1;
153  Vector<T,3> _N = _B^_T;
154 
155  Matrix<T,3,3> Av;
156  Matrix<T,3,3> dAv;
157 
158  for(i=0; i<3; i++) Av[i][0] = _T[i];
159  for(i=0; i<3; i++) Av[i][1] = _N[i];
160  for(i=0; i<3; i++) Av[i][2] = _B[i];
161 
162  Vector<T,3> _dT = (h1_1*kappa)*_N;
163  Vector<T,3> _dN = (h1_1*tau)*_B - (h1_1*kappa)*_T;
164  Vector<T,3> _dB = (-h1_1*tau)*_N;
165 
166  for(i=0; i<3; i++) dAv[i][0] = _dT[i];
167  for(i=0; i<3; i++) dAv[i][1] = _dN[i];
168  for(i=0; i<3; i++) dAv[i][2] = _dB[i];
169 
170  if(_RMF)
171  {
172  T omega = getOmega(v);
173  Matrix<T,3,3> m;
174  m[0][0] = 1; m[0][1] = 0; m[0][2] = 0;
175  m[1][0] = 0; m[1][1] = cos(omega); m[1][2] = -sin(omega);
176  m[2][0] = 0; m[2][1] = sin(omega); m[2][2] = cos(omega);
177 
178  T th = -tau*h1_1;
179  Matrix<T,3,3> dm;
180  dm[0] = Vector<T,3>(0.0);
181  dm[1] = -th*m[2];
182  dm[2] = th*m[1];
183 
184  dAv = dAv*m + Av*dm;
185  Av = Av*m;
186  }
187 
188  DVector<Vector<T,3> > g = _profile->evaluate(u,d1);
189 
190  this->_p[0][0] = h[0] + Av*(g[0] - _spv); //S
191 
192  if( this->_dm == GM_DERIVATION_EXPLICIT)
193  {
194  if(d1)
195  this->_p[1][0] = Av*g[1]; //Su
196  if(d2)
197  this->_p[0][1] = h[1] + dAv*(g[0] - _spv); //Sv
198  }
199  }
