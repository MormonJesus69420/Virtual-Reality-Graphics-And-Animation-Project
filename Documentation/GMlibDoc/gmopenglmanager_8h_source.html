107  fillLightUBO( GL::UniformBufferObject& ubo, const GL::GLLightHeader& header, const std::vector<T>& data ) {
108 
109  ubo.bufferData( sizeof(GL::GLLightHeader) + data.size() * sizeof(T), 0x0, GL_DYNAMIC_DRAW );
110  ubo.bufferSubData( 0, sizeof(GL::GLLightHeader), &header );
111  ubo.bufferSubData( sizeof(GL::GLLightHeader), data.size() * sizeof(T), data.data() );
112  }

174
