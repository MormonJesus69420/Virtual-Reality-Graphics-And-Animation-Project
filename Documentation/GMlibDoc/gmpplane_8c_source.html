133  void PPlane<T>::eval(T u, T v, int d1, int d2, bool /*lu*/, bool /*lv*/ ) const {
134 
135  this->_p.setDim( d1+1, d2+1 );
136 
137  this->_p[0][0] = _pt + u*_u + v*_v ;
138 
139  if( this->_dm == GM_DERIVATION_EXPLICIT ) {
140 
141  // 1st
142  if(d1) this->_p[1][0] = _u; // S_u
143  if(d2) this->_p[0][1] = _v; // S_v
144  if(d1>1 && d2>1) this->_p[1][1] = Vector<T,3>(T(0)); // S_uv
145 
146  // 2nd
147  if(d1>1) this->_p[2][0] = Vector<T,3>(T(0)); // S_uu
148  if(d2>1) this->_p[0][2] = Vector<T,3>(T(0)); // S_vv
149  if(d1>1 && d2) this->_p[2][1] = Vector<T,3>(T(0)); // S_uuv
150  if(d1 && d2>1) this->_p[1][2] = Vector<T,3>(T(0)); // S_uvv
151  if(d1>1 && d2>1) this->_p[2][2] = Vector<T,3>(T(0)); // S_uuvv
152 
153  // 3rd
154  if(d1>2) this->_p[3][0] = Vector<T,3>(T(0)); // S_uuu
155  if(d2>2) this->_p[0][3] = Vector<T,3>(T(0)); // S_vvv
156  if(d1>2 && d2) this->_p[3][1] = Vector<T,3>(T(0)); // S_uuuv
157  if(d1 && d2>2) this->_p[1][3] = Vector<T,3>(T(0)); // S_uvvv
158  if(d1>2 && d2>1) this->_p[3][2] = Vector<T,3>(T(0)); // S_uuuvv
159  if(d1>1 && d2>2) this->_p[2][3] = Vector<T,3>(T(0)); // S_uuvvv
160  if(d1>2 && d2>2) this->_p[3][3] = Vector<T,3>(T(0)); // S_uuuvvv
161  }
