52  cl_mem d_A = clCreateBuffer(AppmlContext::getContext(), CL_MEM_READ_ONLY, m.getDim1() * m.getDim2() * sizeof(float), NULL, &AppmlContext::getErr());
53  cl_mem d_B = clCreateBuffer(AppmlContext::getContext(), CL_MEM_READ_ONLY, b.getDim1() * b.getDim2() * sizeof(float), NULL, &AppmlContext::getErr());
54  cl_mem d_C = clCreateBuffer(AppmlContext::getContext(), CL_MEM_READ_WRITE, work.getSize() * sizeof(float), NULL, &AppmlContext::getErr());
55 
56  for(int i=0; i<m.getDim1(); i++)
57  {
58  AppmlContext::getErr() = clEnqueueWriteBuffer(AppmlContext::getQueue(), d_A, CL_TRUE, i*m.getDim2()*sizeof(float), m(i).getDim()*sizeof(float), &m(i)(0), 0, NULL, NULL);
59  }
60  for(int i=0; i<b.getDim1(); i++)
61  {
62  AppmlContext::getErr() = clEnqueueWriteBuffer(AppmlContext::getQueue(), d_B, CL_TRUE, i*b.getDim2()*sizeof(float), b(i).getDim()*sizeof(float), &b(i)(0), 0, NULL, NULL);
63  }
64 
65  AppmlContext::getErr() = clAmdBlasSgemm(clAmdBlasRowMajor, clAmdBlasNoTrans, clAmdBlasNoTrans, m.getDim1(), b.getDim2(), m.getDim2(), 1.0f,
66  d_A, m.getDim2(), d_B, b.getDim2(), 0.0f, d_C, m.getDim2(), 1, &AppmlContext::getQueue(), 0, NULL, &AppmlContext::getEvent());
