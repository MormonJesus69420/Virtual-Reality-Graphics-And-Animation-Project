85  void PTCurve<T>::_resample( PCurve<T,3>* obj, DVector<DVector<Vector<T,3> > >& p, int m, int d ) {
86 
87  p.setDim( m );
88 
89  T st = obj->getParStart();
90  T en = obj->getParEnd();
91  T dt = ( en - st ) / ( m - 1 );
92 
93  for(int i = 0; i < m - 1; i++)
94  p[i] = obj->evaluateLocal( st + i * dt, d );
95 
96  p[m-1] = obj->evaluateLocal( en, d );
97  }

100  template <typename T>
101  inline
102  Point<T, 3> PTCurve<T>::_rotate_and_project() {
103 
104  Point<T, 3> p;
105  Point<T, 6> tA, tB;
106  double sphi = sin(_angle);
107  double cphi = cos(_angle);
108  int dim = 6;
109 
110  for(int i = 0; i < dim; i++) {
111  tB[i] = _tPoints(i);
112  tA[i] = tB[i];
113  }
114 
115  for(int i = 0; i < dim - 1; i++) {
116  for(int j = i + 1; j < dim; j++) {
117  tB[i] = cphi * tA[i] - sphi * tA[j];
118  tB[j] = sphi * tA[i] + cphi * tA[j];
119 
120  for(int k = 0; k < dim; k++)
121  tA[k] = tB[k];
122  }
123  }
124 
125  for(int i = 0; i < 3; i++)
126  p[i] = tA[i];
127 
128  return p;
129  }
